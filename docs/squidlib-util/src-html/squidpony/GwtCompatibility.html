009/**
010 * Static methods useful to be GWT-compatible, and also methods useful for filling gaps in Java's support for arrays.
011 * You can think of the purpose of this class as "GWT, and Compatibility". There's a replacement for a Math method that
012 * isn't available on GWT, a quick way to get the first element in an Iterable, copying, inserting, and filling methods
013 * for 2D arrays of primitive types (char, int, double, and boolean), and also a method to easily clone a Coord array.
014 * 
015 * @author smelC
016 * @author Tommy Ettinger
017 */
018public class GwtCompatibility {

021     * Gets an exact copy of an array of Coord. References are shared, which should be the case for all usage of Coord
022     * since they are immutable and thus don't need multiple variants on a Coord from the pool.
023         * @param input an array of Coord to copy
024         * @return A clone of {@code input}.
025         */
026        public static Coord[] cloneCoords(Coord[] input) {
027                final Coord[] result = new Coord[input.length];
028        //System.arraycopy, despite being cumbersome, is the fastest way to copy an array on the JVM.
029        System.arraycopy(input, 0, result, 0, input.length);
030                return result;
031        }

033        /**
034     * A replacement for Math.IEEEremainder, just because Math.IEEEremainder isn't GWT-compatible.
035     * Gets the remainder of op / d, which can be negative if any parameter is negative.
036         * @param op the operand/dividend
037         * @param d the divisor
038         * @return The remainder of {@code op / d}, as a double; can be negative
039         */
040        /* smelC: because Math.IEEEremainder isn't GWT compatible */
041        public static double IEEEremainder(double op, double d) {
042                final double div = Math.round(op / d);
043                return op - (div * d);
044        }

047     * Stupidly simple convenience method that produces a range from 0 to end, not including end, as an int array.
048     * @param end the exclusive upper bound on the range
049     * @return the range of ints as an int array
050     */
051    public static int[] range(int end)
052    {
053        if(end <= 0)
054            return new int[0];
055        int[] r = new int[end];
056        for (int i = 0; i < end; i++) {
057            r[i] = i;
058        }
059        return r;
060    }

069    {
070        if(end - start <= 0)
071            return new int[0];
072        int[] r = new int[end - start];
073        for (int i = 0, n = start; n < end; i++, n++) {
074            r[i] = n;
075        }
076        return r;
077    }

092    /**
093     * Stupidly simple convenience method that produces a char range from start to end, including end, as a char array.
094     * @param start the inclusive lower bound on the range, such as 'a'
095     * @param end the inclusive upper bound on the range, such as 'z'
096     * @return the range of chars as a char array
097     */
098    public static char[] charSpan(char start, char end)
099    {
100        if(end - start <= 0)
101            return empty;
102        if(end == 0xffff)
103        {
104
105            char[] r = new char[0x10000 - start];
106            for (char i = 0, n = start; n < end; i++, n++) {
107                r[i] = n;
108            }
109            r[0xffff - start] = 0xffff;
110            return r;
111        }
112        char[] r = new char[end - start + 1];
113        for (char i = 0, n = start; n <= end; i++, n++) {
114            r[i] = n;
115        }
116        return r;
117    }
118    /**
119     * Stupidly simple convenience method that produces a char array containing only letters that can be reasonably
120     * displayed (with SquidLib's default text display assets, at least). The letters are copied from a single source
121     * of 256 chars; if you need more chars or you don't need pure letters, you can use {@link #charSpan(char, char)}.
122     * @param charCount the number of letters to return in an array; the maximum this will produce is 256
123     * @return the range of letters as a char array
124     */
125    public static char[] letterSpan(int charCount)
126    {
127        if(charCount <= 0)
128            return empty;
129        char[] r = new char[Math.min(charCount, 256)];
130        System.arraycopy(letters, 0, r, 0, r.length);
131        return r;
132    }

134    /**
135     * Gets the first item in an Iterable of T, or null if it is empty. Meant for collections like LinkedHashSet, which
136     * can promise a stable first element but don't provide a way to access it. Not exactly a GWT compatibility method,
137     * but more of a Java standard library stand-in. Even though LinkedHashSet does not support this out-of-the-box,
138     * OrderedSet already provides a first() method.
139     * @param collection an Iterable of T; if collection is null or empty this returns null
140     * @param <T> any object type
141     * @return the first element in collection, or null if it is empty or null itself
142     */
143        public static <T> T first(Iterable<T> collection)
144    {
145        if(collection == null)
146            return null;
147        Iterator<T> it = collection.iterator();
148        if(it.hasNext())
149            return it.next();
150        return null;
151    }

153        /**
154         * Gets a copy of the 2D char array, source, that has the same data but shares no references with source.
155         * @param source a 2D char array
156         * @return a copy of source, or null if source is null
157         */
158        public static char[][] copy2D(char[][] source)
159        {
160                if(source == null)
161                        return null;
162                if(source.length < 1)
163                        return new char[0][0];
164                char[][] target = new char[source.length][source[0].length];
165                for (int i = 0; i < source.length && i < target.length; i++) {
166                        System.arraycopy(source[i], 0, target[i], 0, source[i].length);
167                }
168                return target;
169        }

225    /**
226     * Inserts as much of source into target at the given x,y position as target can hold or source can supply.
227     * Modifies target in-place and also returns target for chaining.
228     * Used primarily to place a smaller array into a different position in a larger array, often freshly allocated.
229     * @param source a 2D char array that will be copied and inserted into target
230     * @param target a 2D char array that will be modified by receiving as much of source as it can hold
231     * @param x the x position in target to receive the items from the first cell in source
232     * @param y the y position in target to receive the items from the first cell in source
233     * @return a modified copy of target with source inserted into it at the given position
234     */
235    public static char[][] insert2D(char[][] source, char[][] target, int x, int y)
236    {
237        if(source == null || target == null)
238            return target;
239        if(source.length < 1 || source[0].length < 1)
240            return copy2D(target);
241        for (int i = 0; i < source.length && x + i < target.length; i++) {
242            System.arraycopy(source[i], 0, target[x + i], y, Math.min(source[i].length, target[x+i].length - y));
243        }
244        return target;
245    }

373    /**
374     * Rearranges an ArrayList to use the given ordering, returning a copy; random orderings can be produced with
375     * {@link squidpony.squidmath.RNG#randomOrdering(int)} or
376     * {@link squidpony.squidmath.RNG#randomOrdering(int, int[])}. These orderings will never repeat an earlier element,
377     * and the returned ArrayList may be shorter than the original if {@code ordering} isn't as long as {@code list}.
378     * Using a random ordering is like shuffling, but allows you to repeat the shuffle exactly on other collections of
379     * the same size. A reordering can also be inverted with {@link #invertOrdering(int[])} or
380     * {@link #invertOrdering(int[], int[])}, getting the change that will undo another ordering.
381     * @param list an ArrayList that you want a reordered version of; will not be modified.
382     * @param ordering an ordering, typically produced by one of RNG's randomOrdering methods.
383     * @param <T> any generic type
384     * @return a modified copy of {@code list} with its ordering changed to match {@code ordering}.
385     */
386    public static <T> ArrayList<T> reorder (ArrayList<T> list, int... ordering) {
387        int ol;
388        if (list == null || ordering == null || (ol = Math.min(list.size(), ordering.length)) == 0)
389            return list;
390        ArrayList<T> alt = new ArrayList<T>(ol);
391        for (int i = 0; i < ol; i++) {
392            alt.add(list.get((ordering[i] % ol + ol) % ol));
393        }
394        return alt;
395    }

397    /**
398     * Given an ordering such as one produced by {@link squidpony.squidmath.RNG#randomOrdering(int, int[])}, this finds
399     * its inverse, able to reverse the reordering and vice versa.
400     * @param ordering the ordering to find the inverse for
401     * @return the inverse of ordering
402     */
403    public static int[] invertOrdering(int[] ordering)
404    {
405        int ol = 0;
406        if(ordering == null || (ol = ordering.length) == 0) return ordering;
407        int[] next = new int[ol];
408        for (int i = 0; i < ol; i++) {
409            if(ordering[i] < 0 || ordering[i] >= ol) return next;
410            next[ordering[i]] = i;
411        }
412        return next;
413    }

415    /**
416     * Given an ordering such as one produced by {@link squidpony.squidmath.RNG#randomOrdering(int, int[])}, this finds
417     * its inverse, able to reverse the reordering and vice versa. This overload doesn't allocate a new int
418     * array, and instead relies on having an int array of the same size as ordering passed to it as an
419     * additional argument.
420     * @param ordering the ordering to find the inverse for
421     * @param dest the int array to put the inverse reordering into; should have the same length as ordering
422     * @return the inverse of ordering; will have the same value as dest
423     */
424    public static int[] invertOrdering(int[] ordering, int[] dest)
425    {
426        int ol = 0;
427        if(ordering == null || dest == null || (ol = Math.min(ordering.length, dest.length)) == 0)
428            return ordering;
429        for (int i = 0; i < ol; i++) {
430            if(ordering[i] < 0 || ordering[i] >= ol) return dest;
431            dest[ordering[i]] = i;
432        }
433        return dest;
434    }
