001package squidpony;
002
003import regexodus.Matcher;
004import regexodus.Pattern;
005import regexodus.REFlags;
006import squidpony.annotation.Beta;
007import squidpony.squidmath.IntVLA;
008
009import java.util.ArrayList;
010import java.util.Iterator;
011
012/**
013 * A simple format parser for String-based configuration or data files where JSON is overkill.
014 * Supports only one type, String, but allows each String to have arbitrary nested levels of
015 * String children as if in sub-lists. You can interpret the Strings however you want, and
016 * quoting each String isn't necessary if they are just one word ("bare words" are allowed).
017 * The main way of using this is to get an ObText.ItemIterator value using {@link #iterator()},
018 * which acts as a normal Iterator over the top-level Strings (not children of anything), but
019 * to call its {@link ItemIterator#hasChild()} method when you expect potential child elements,
020 * then {@link ItemIterator#children()} to get another ItemIterator over the child elements if
021 * you want to explore deeper.
022 * <br>
023 * Format example:
024 * <pre>
025 * hello world
026 * 'how are you today?' [just great thanks]
027 * hooray!
028 *
029 * complexity?
030 * [it is possible [yes this is a good example]
031 * 'escapes like \[\'\] all work'
032 * ]
033 *
034 * comments are allowed // like this
035 * comments can have different forms # like this
036 * // block comments like in c are allowed
037 * / * but because this example is in javadoc, this example is not actually a comment * /
038 * // remove the spaces between each slash and asterisk to make the last line a comment.
039 * /[delimit/or block comments with delimiters/delimit]/
040 *
041 * '''
042 * raw strings (heredocs) look like this normally.
043 *     they permit characters without escapes, ]][][[ \/\/\ ,
044 *     except for triple quotes.
045 *     they keep newlines and indentation intact,
046 * except for up to one newline ignored adjacent to each triple quote.
047 * '''
048 *
049 * [[different[
050 * if you may need triple quotes
051 *     in the raw string, use a different syntax that allows delimiters.
052 * here, the delimiter is '''different''', just to be different.]different]]
053 * </pre>
054 * <br>
055 * Inspired strongly by STOB, http://stobml.org/ , but no code is shared and the format is
056 * slightly different. The main differences are that ObText supports nested block comments
057 * using the syntax {@code /[delimiter/contents/delimiter]/} where delimiter may be empty
058 * but must match on both sides, and contents is the body of the comment. ObText uses Python-
059 * like "heredoc" syntax for raw strings surrounded by triple-apostrophes '''like so''' with
060 * optional initial and final newlines in the raw string ignored. An alternate raw string
061 * syntax is present that allows delimiters, using the syntax
062 * {@code [[delimiter[contents]delimiter]]}, where again delimiter may be empty and contents
063 * is the body of the raw string. We use square brackets in place of STOB's curly braces to
064 * mark children associated with a string.
065 */
066@Beta
067public class ObText implements Iterable<String>{
068    public static final Pattern pattern = Pattern.compile(
069            "(?>'''(?:[\n\u000C\f\r\u0085\u2028\u2029]|\r\n)?({=s}.*?)(?:[\n\u000C\f\r\u0085\u2028\u2029]|\r\n)?''')" +
070            "|(?>\\[\\[({=q}[^\\[\\]]*)\\[(?:[\n\u000C\f\r\u0085\u2028\u2029]|\r\n)?({=s}.*?)(?:[\n\u000C\f\r\u0085\u2028\u2029]|\r\n)?\\]{\\q}\\]\\])" +
071            "|(?>({=q}[\"'])({=s}.*?)(?<!\\\\){\\q})" +
072            "|(?>(?>//|#)(?>\\V*))" +
073            "|(?>/\\*(?:.*?)\\*/)" +
074            "|(?>/\\[({=q}\\S*)/(?:.*?)/{\\q}\\]/)" +
075            "|({=s}[^\\s\\[\\]\"'#\\\\]+)" +
076            "|({=o}\\[)" +
077            "|({=c}\\])", REFlags.DOTALL | REFlags.UNICODE
078
079    );
080    public static final Matcher m = pattern.matcher();
081
082    protected final ArrayList<String> strings = new ArrayList<String>(64);
083    protected final IntVLA neighbors = new IntVLA(64);
084    private final IntVLA nesting = new IntVLA(16);
085    public ObText()
086    {
087
088    }
089    public ObText(CharSequence text)
090    {
091        parse(text);
092    }
093    public ObText parse(CharSequence text)
094    {
095        m.setTarget(text);
096        nesting.clear();
097        int t = -1;
098        while (m.find()) {
099            if (m.isCaptured("s")) {
100                strings.add(m.group("s"));
101                neighbors.add(1);
102            }
103            else if(m.isCaptured("o"))
104            {
105                nesting.add(neighbors.size - 1);
106            }
107            else if(m.isCaptured("c"))
108            {
109                neighbors.incr(t = nesting.pop(), neighbors.size - t - 1);
110                if(t < neighbors.size - 1)
111                    neighbors.set(neighbors.size-1, 0);
112            }
113        }
114        return this;
115    }
116
117    /**
118     * Returns an iterator over elements of type {@code T}.
119     *
120     * @return an Iterator.
121     */
122    @Override
123    public ItemIterator iterator() {
124        return new ItemIterator();
125    }
126
127    /**
128     * A one-way iterator through this ObText's String items. Note that this can get
129     * an iterator into a child sequence with {@link #children()}, which should
130     * only be called after checking that a child exists with {@link #hasChild()}.
131     * {@link #remove()} is not supported, but {@link #next()} and {@link #hasNext()}
132     * are, of course.
133     */
134    public class ItemIterator implements Iterator<String>
135    {
136        ItemIterator()
137        {
138        }
139        ItemIterator(int i)
140        {
141            current = i % neighbors.size;
142            index = current;
143        }
144        int index = 0, current = -1;
145        /**
146         * Returns {@code true} if the iteration has more elements.
147         * (In other words, returns {@code true} if {@link #next} would
148         * return an element rather than throwing an exception.)
149         *
150         * @return {@code true} if the iteration has more elements
151         */
152        @Override
153        public boolean hasNext() {
154            return index < neighbors.size && (current < 0 || neighbors.get(current) > 0);
155        }
156        /**
157         * Returns {@code true} if the ObText.Item has any child elements.
158         * (In other words, returns {@code true} if {@link #children} would
159         * return an ItemIterator rather than throwing an exception.)
160         *
161         * @return {@code true} if this has any children
162         */
163        public boolean hasChild() {
164            return index < neighbors.size - 1 && ((current < 0 && neighbors.get(0) > 1) || neighbors.get(current) > 1);
165        }
166
167        /**
168         * Returns the next element in the iteration.
169         *
170         * @return the next element in the iteration
171         * @throws java.util.NoSuchElementException if the iteration has no more elements
172         */
173        @Override
174        public String next() {
175            if(current < 0)
176            {
177                if(strings.isEmpty())
178                    throw new java.util.NoSuchElementException("No more sibling items in ObText object");
179                current = 0;
180                index = neighbors.get(0);
181                return strings.get(0);
182            }
183            if(index >= neighbors.size || neighbors.get(current) <= 0)
184                throw new java.util.NoSuchElementException("No more sibling items in ObText object");
185            index = neighbors.get(index) + (current = index);
186            return strings.get(current);
187        }
188        /**
189         * Returns the first child of this ObText.Item and descends into the sequence of child elements.
190         *
191         * @return the first child of this item
192         * @throws java.util.NoSuchElementException if the iteration has no children
193         */
194        public ItemIterator children() {
195            if(current < 0)
196            {
197                if(neighbors.size <= 0 || neighbors.get(0) == 1)
198                    throw new java.util.NoSuchElementException("No current children in ObText object");
199                return new ItemIterator(1);
200            }
201            if(current >= neighbors.size - 2 || neighbors.get(current) == 1)
202                throw new java.util.NoSuchElementException("No current children in ObText object");
203            return new ItemIterator(current+1);
204        }
205
206        @Override
207        public void remove() {
208            throw new UnsupportedOperationException("remove() not supported");
209
210        }
211    }
212
213}
