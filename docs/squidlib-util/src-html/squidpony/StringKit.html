014    public static String join(CharSequence delimiter, CharSequence... elements)
015    {
016        if(elements == null || elements.length == 0) return "";
017        StringBuilder sb = new StringBuilder(64);
018        sb.append(elements[0]);
019        for (int i = 1; i < elements.length; i++) {
020            sb.append(delimiter).append(elements[i]);
021        }
022        return sb.toString();
023    }

179    public static CharSequence encode(byte[] data)
180    {
181        if(data == null)
182            return "";
183        int len = data.length;
184        char[] chars = new char[len];
185        for (int i = 0, c = 0; c < len; i++, c++) {
186            chars[c] = (char) (data[i] | 0x2800);
187        }
188        return new String(chars);
189    }
190    public static CharSequence encode(short[] data)
191    {
192        if(data == null)
193            return "";
194        int len = data.length * 2;
195        char[] chars = new char[len];
196        short item;
197        for (int i = 0, c = 0; c < len; i++, c += 2) {
198            item = data[i];
199            chars[c] = (char) ((item & 0xff) | 0x2800);
200            chars[c+1] = (char)((item >>> 8) | 0x2800);
201        }
202        return new String(chars);
203    }

256    {
257        if(data == null)
258            return new byte[0];
259        int len = data.length();
260        byte[] bytes = new byte[len];
261        for (int i = 0; i < len; i++) {
262            bytes[i] = (byte)(data.charAt(i) & 0xff);
263        }
264        return bytes;

268    {

272        short[] shorts = new short[len];

294    public static float[] decodeFloats(CharSequence data)
295    {
296        if(data == null)
297            return new float[0];
298        int len = data.length() / 4, tmp;
299        float[] floats = new float[len];
300        for (int i = 0; i < len; i++) {
301            tmp = ((data.charAt(i * 4) & 0xff)
302                    | ((data.charAt(i * 4 + 1) & 0xff) << 8)
303                    | ((data.charAt(i * 4 + 2) & 0xff) << 16)
304                    | ((data.charAt(i * 4 + 3) & 0xff) << 24));
305            floats[i] = Float.intBitsToFloat(tmp);
306        }
307        return floats;
308    }

457    private static final Pattern decoder2D = Pattern.compile("(?:([^\t;]*)[\t;])+?", 0),
458            decoder3D = Pattern.compile("([^;]*;)#?", 0);
459    public static byte[][] decodeBytes2D(CharSequence data)
460    {
461        if(data == null)
462            return new byte[0][0];
463        Matcher matcher = decoder2D.matcher(data);
464        MatchIterator mi = matcher.findAll();
465        byte[][] values = new byte[mi.count()][];
466        int i = 0;
467        matcher.setTarget(data);
468        while (mi.hasNext())
469        {
470            values[i++] = decodeBytes(mi.next().group(1));
471        }
472        return values;
473    }
