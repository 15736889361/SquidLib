011    public static String join(CharSequence delimiter, CharSequence... elements)
012    {
013        if(elements == null || elements.length == 0) return "";
014        StringBuilder sb = new StringBuilder(64);
015        sb.append(elements[0]);
016        for (int i = 1; i < elements.length; i++) {
017            sb.append(delimiter).append(elements[i]);
018        }
019        return sb.toString();
020    }
021    public static String joinArrays(CharSequence delimiter, char[]... elements)
022    {
023        if(elements == null || elements.length == 0) return "";
024        StringBuilder sb = new StringBuilder(64);
025        sb.append(elements[0]);
026        for (int i = 1; i < elements.length; i++) {
027            sb.append(delimiter).append(elements[i]);
028        }
029        return sb.toString();
030    }

076    public static char[] apEncode(long number, int offset, char[] buf) {
077        if(buf != null && buf.length >= 16 - offset) {
078            buf[offset] = (char) ((number >>> 60) + 65);
079            buf[offset+1] = (char) ((0xf & (number >>> 56)) + 65);
080            buf[offset+2] = (char) ((0xf & (number >>> 52)) + 65);
081            buf[offset+3] = (char) ((0xf & (number >>> 48)) + 65);
082            buf[offset+4] = (char) ((0xf & (number >>> 44)) + 65);
083            buf[offset+5] = (char) ((0xf & (number >>> 40)) + 65);
084            buf[offset+6] = (char) ((0xf & (number >>> 36)) + 65);
085            buf[offset+7] = (char) ((0xf & (number >>> 32)) + 65);
086            buf[offset+8] = (char) ((0xf & (number >>> 28)) + 65);
087            buf[offset+9] = (char) ((0xf & (number >>> 24)) + 65);
088            buf[offset+10] = (char) ((0xf & (number >>> 20)) + 65);
089            buf[offset+11] = (char) ((0xf & (number >>> 16)) + 65);
090            buf[offset+12] = (char) ((0xf & (number >>> 12)) + 65);
091            buf[offset+13] = (char) ((0xf & (number >>> 8)) + 65);
092            buf[offset+14] = (char) ((0xf & (number >>> 4)) + 65);
093            buf[offset+15] = (char) ((0xf & number) + 65);
094        }
095        return buf;
096    }

146    public static long apDecodeLong(char[] data, int offset)
147    {
148        return (data == null || data.length < 16 + offset) ? 0 :
149                ((0xf & data[offset] - 65L) << 60)
150                        | ((0xf & data[offset + 1] - 65L) << 56)
151                        | ((0xf & data[offset + 2] - 65L) << 52)
152                        | ((0xf & data[offset + 3] - 65L) << 48)
153                        | ((0xf & data[offset + 4] - 65L) << 44)
154                        | ((0xf & data[offset + 5] - 65L) << 40)
155                        | ((0xf & data[offset + 6] - 65L) << 36)
156                        | ((0xf & data[offset + 7] - 65L) << 32)
157                        | ((0xf & data[offset + 8] - 65L) << 28)
158                        | ((0xf & data[offset + 9] - 65L) << 24)
159                        | ((0xf & data[offset + 10] - 65L) << 20)
160                        | ((0xf & data[offset + 11] - 65L) << 16)
161                        | ((0xf & data[offset + 12] - 65L) << 12)
162                        | ((0xf & data[offset + 13] - 65L) << 8)
163                        | ((0xf & data[offset + 14] - 65L) << 4)
164                        | ((0xf & data[offset + 15] - 65L));
