001package squidpony;
002
003import squidpony.annotation.Beta;
004import squidpony.squidmath.CrossHash;
005import squidpony.squidmath.K2;
006
007/**
008 * Used to standardize conversion for a given type, {@code T}, to and from a serialized String format.
009 * This abstract class should usually be made concrete by a single-purpose class (not the type T itself).
010 * Also includes a static registry of types as Class arrays (including the classes of generic type parameters in array
011 * elements after the first) to the  corresponding StringConvert objects that have been constructed for those types,
012 * although the registry must store the StringConvert objects without any further types (you can cast the StringConvert
013 * to a StringConvert with the desired generic type, or call {@link #restore(String)} on the un-parametrized type and
014 * get back an Object that can be cast to the correct type, but we aren't able to store the actual type, other than the
015 * Class array used to look up a StringConvert). The static method {@link #lookup(Class[])} can be used to find the
016 * StringConvert registered for a type combination. The static method {@link #get(CharSequence)} can be used to find a
017 * StringConvert by its name. The static utility method {@link #asArray(Class[])} can be used to help reduce the amount
018 * of arrays produced by varargs, but the array it returns must not be edited once used to construct a StringConvert.
019 */
020@Beta
021public abstract class StringConvert<T> {
022    protected final String name;
023    protected final Class[] classes;
024
025    /**
026     * Constructs a StringConvert using a vararg or array of Class objects. If an array is passed, it must not be
027     * altered after usage. If no arguments are passed, if types is null, or if the first item of types is null, then
028     * this uses a special class array consisting of {@link Void#TYPE}, and the name will be "void". If types has length
029     * 1, then the name will be the "simple name" of the first element in types, as produced by
030     * {@link Class#getSimpleName()} (note that this produces an empty string for anonymous classes). Otherwise, this
031     * considers items after the first to be generic type arguments of the first, and currently does not consider nested
032     * generic type parameters (this will be addressed later, if possible). In this last case, the name uses Java
033     * conventions for generics, turning the call {@code new StringConvert(OrderedMap.class, String.class, Long.class)}
034     * into a StringConvert with the name {@code "OrderedMap<String,Long>"} (no spaces will be present in the name, but
035     * thanks to some customization of the registry, you can give a String with spaces in it to
036     * {@link #get(CharSequence)} and still find the correct one).
037     * @param types a vararg of Class objects representing the type this can convert, including generic type parameters
038     *              of the first element, if there are any, at positions after the first
039     */
040    public StringConvert(final Class... types) {
041        if(types == null || types.length <= 0 || types[0] == null)
042        {
043            name = "void";
044            classes = new Class[]{Void.TYPE};
045        }
046        else if(types.length == 1) {
047            name = types[0].getSimpleName();
048            classes = types;
049        }
050        else
051        {
052            StringBuilder sb = new StringBuilder(64);
053            sb.append(types[0].getSimpleName()).append('<').append(types[1].getSimpleName());
054            for (int i = 2; i < types.length; i++) {
055                sb.append(',').append(types[i].getSimpleName());
056            }
057            name = sb.append('>').toString();
058            classes = types;
059        }
060
061    }
062    public String getName() {return name;}
063    public abstract String stringify(T item);
064    public abstract T restore(String text);
065
066    public static StringConvert<?> get(final CharSequence name)
067    {
068        int i = registry.indexOfA(name);
069        if(i < 0) return null;
070        return registry.getAAt(i);
071    }
072    /**
073     * Looks up the StringConvert for a given vararg of Class instances (if an array of Classes is used other than a
074     * vararg, it must not be altered in the future, nor reused in a way that modifies its elements). Returns null if no
075     * StringConvert is found. You should usually cast the returned StringConvert, if non-null, to the specific
076     * StringConvert generic type you want.
077     * @param types the vararg of types to look up
078     * @return the StringConvert registered for the given types, or null if none has been made
079     */
080    public static StringConvert<?> lookup(final Class... types)
081    {
082        return registry.getAFromB(types);
083    }
084
085    /**
086     * Simply takes a vararg of Class and returns it as an array.
087     * Can be handy to avoid re-creating arrays implicitly from varargs of Class items.
088     * @param types a vararg of Class
089     * @return types as an array
090     */
091    public static Class[] asArray(final Class... types)
092    {
093        return types;
094    }
095
096    public static final CrossHash.IHasher spaceIgnoringHasher = new CrossHash.IHasher() {
097        @Override
098        public int hash(Object data) {
099            if (data == null || !(data instanceof StringConvert || data instanceof CharSequence))
100                return 0;
101            final CharSequence s;
102            if(data instanceof StringConvert) s = ((StringConvert) data).getName();
103            else s = (CharSequence) data;
104            int result = 0x9E3779B9, a = 0x632BE5AB;
105            final int len = s.length();
106            for (int i = 0; i < len; i++) {
107                if(s.charAt(i) != ' ')
108                    result += (a ^= 0x85157AF5 * s.charAt(i));
109            }
110            return result * (a | 1) ^ (result >>> 11 | result << 21);
111        }
112
113        @Override
114        public boolean areEqual(Object left, Object right) {
115            if (left == right) return true;
116            if(!((left instanceof StringConvert || left instanceof CharSequence)
117                    && (right instanceof StringConvert || right instanceof CharSequence)))
118                return false;
119            final CharSequence ls, rs;
120            if(left instanceof StringConvert) ls = ((StringConvert) left).getName();
121            else ls = (CharSequence)left;
122            if(right instanceof StringConvert) rs = ((StringConvert) right).getName();
123            else rs = (CharSequence)right;
124            final int llen = ls.length(), rlen = rs.length();
125            char lc = ' ', rc = ' ';
126            for (int l = 0, r = 0; l < llen && r < rlen; l++, r++) {
127                while (l < llen && (lc = ls.charAt(l)) == ' ') ++l;
128                while (r < rlen && (rc = rs.charAt(r)) == ' ') ++r;
129                if(lc != rc)
130                    return false;
131            }
132            return true;
133        }
134    };
135    public static final K2<StringConvert, Class[]> registry =
136            new K2<>(128, 0.75f, spaceIgnoringHasher, CrossHash.objectArrayHasher);
