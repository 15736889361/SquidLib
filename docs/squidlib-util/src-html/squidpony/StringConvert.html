003/**
004 * Used to standardize conversion for a given type, {@code T}, to and from a serialized String format.
005 * This abstract class should usually be made concrete by a single-purpose class (not the type T itself).
006 */
007public abstract class StringConvert<T> {
008    public StringConvert() {};
009    public abstract String stringify(T item);
010    public abstract T restore(String text);
011}
