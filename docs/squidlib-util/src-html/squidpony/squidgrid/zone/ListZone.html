001package squidpony.squidgrid.zone;
002
003import java.util.List;
004
005import squidpony.squidgrid.zone.Zone.Skeleton;
006import squidpony.squidmath.Coord;
007
008/**
009 * A zone defined by a {@link List}.
010 * 
011 * @author smelC
012 */
013public class ListZone extends Skeleton {
014
015        protected final List<Coord> coords;
016
017        private static final long serialVersionUID = 1166468942544595692L;
018
019        public ListZone(List<Coord> coords) {
020                this.coords = coords;
021        }
022
023        @Override
024        public boolean isEmpty() {
025                return coords.isEmpty();
026        }
027
028        @Override
029        public int size() {
030                return coords.size();
031        }
032
033        @Override
034        public boolean contains(Coord c) {
035                return coords.contains(c);
036        }
037
038        @Override
039        public boolean contains(int x, int y) {
040                return coords.contains(Coord.get(x, y));
041        }
042
043        @Override
044        public List<Coord> getAll() {
045                return coords;
046        }
047
048        @Override
049        public String toString() {
050                return coords.toString();
051        }
052}
