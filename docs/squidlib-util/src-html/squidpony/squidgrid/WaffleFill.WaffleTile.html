011/**
012 * An implementation of the Wave Function Collapse algorithm for placing tiles in a way that satisfies constraints; the
013 * name both sounds like Wave Function and loosely describes how the algorithm works. All tiles start off in an
014 * indeterminate state, and as tiles are placed the states of their neighbors become more certain; the "Collapse" in
015 * Wave Function Collapse refers to the quantum-mechanics term for a decline in unpredictability as a superposition is
016 * measured, but here we use the term "Waffle" for its meaning as switching between positions or viewpoints until forced
017 * into a concrete state. This is meant to act like {@link DetailedMimic} or maybe {@link MimicFill}, but only requires
018 * tiles to be given to it with specific constraints on placement instead of requiring an existing design to mimic.
019 * <br>
020 * Massive credit goes to ExUtumno/mxgmn for developing the amazing Wave Function Collapse algorithm, which this code is
021 * closely based off of, but is not at all a verbatim copy. You can find out more at mxgmn's primary repository for WFC
022 * code, https://github.com/mxgmn/WaveFunctionCollapse , and see the incredible things more creative minds than my own
023 * have come up with using WFC or modifications on it. This file only implements the Simple Tiled Model for now.
024 * Created by Tommy Ettinger on 12/22/2016.
025 */

041    protected Boolean observe() {
042        double min = 1E+3, sum, mainSum, logSum, noise, entropy;
043        int argminx = -1, argminy = -1, amount;
044        boolean[] w;
045
046        for (int x = 0; x < FMX; x++)
047            for (int y = 0; y < FMY; y++) {
048                //if (OnBoundary(x, y)) continue;
049
050                //w = wave[x][y];
051                amount = 0;
052                sum = 0;
053
054                for (int t = 0; t < T; t++)
055                    if (wave[t].contains(x, y)) {
056                        amount += 1;
057                        sum += stationary[t];
058                    }
059
060                if (sum == 0) return false;
061
062                noise = random.nextDouble(1E-6);
063
064                if (amount == 1) entropy = 0;
065                else if (amount == T) entropy = logT;
066                else {
067                    mainSum = 0;
068                    logSum = Math.log(sum);
069                    for (int t = 0; t < T; t++) if (wave[t].contains(x, y)) mainSum += stationary[t] * logProb[t];
070                    entropy = logSum - mainSum / sum;
071                }
072
073                if (entropy > 0 && entropy + noise < min) {
074                    min = entropy + noise;
075                    argminx = x;
076                    argminy = y;
077                }
078            }
079
080        if (argminx == -1 && argminy == -1) {
081            observed = new int[FMX][FMY];
082            for (int x = 0; x < FMX; x++) {
083                for (int y = 0; y < FMY; y++)
084                    for (int t = 0; t < T; t++)
085                        if (wave[t].contains(x, y)) {
086                            observed[x][y] = t;
087                            break;
088                        }
089            }
090
091            return true;
092        }
093
094        double[] distribution = new double[T];
095        for (int t = 0; t < T; t++) {
096            distribution[t] = wave[t].contains(argminx, argminy) ? stationary[t] : 0;
097        }
098        int r = DetailedMimic.weightedRandom(distribution, random.nextDouble());
099        for (int t = 0; t < T; t++) {
100            wave[t].set(t == r, argminx, argminy);
101        }
102        changes.insert(argminx, argminy);
103
104        return null;
105    }

107    public boolean run(RNG rng, int limit) {
108        logT = Math.log(T);
109        logProb = new double[T];
110        for (int t = 0; t < T; t++) logProb[t] = Math.log(stationary[t]);
111
112        clear();
113
114        random = rng;
115
116        for (int l = 0; l < limit || limit == 0; l++) {
117            Boolean result = observe();
118            if (result != null) return result;
119            while (propagate()) ;
120        }
121
122        return true;
123    }
124
125    protected void clear() {
126        for (int t = 0; t < T; t++) wave[t].allOn();
127        changes.clear();
128    }

472    protected boolean propagate() {
473        boolean change = false, b;
474        for (int x2 = 0; x2 < FMX; x2++) {
475            for (int y2 = 0; y2 < FMY; y2++) {
476                for (int d = 0; d < 4; d++) {
477                    int x1 = x2, y1 = y2;
478                    if (d == 0) {
479                        if (x2 == 0) {
480                            if (!periodic) continue;
481                            else x1 = FMX - 1;
482                        } else x1 = x2 - 1;
483                    } else if (d == 1) {
484                        if (y2 == FMY - 1) {
485                            if (!periodic) continue;
486                            else y1 = 0;
487                        } else y1 = y2 + 1;
488                    } else if (d == 2) {
489                        if (x2 == FMX - 1) {
490                            if (!periodic) continue;
491                            else x1 = 0;
492                        } else x1 = x2 + 1;
493                    } else {
494                        if (y2 == 0) {
495                            if (!periodic) continue;
496                            else y1 = FMY - 1;
497                        } else y1 = y2 - 1;
498                    }
499
500                    if (!changes.contains(x1, y1)) continue;
501
502                    for (int t2 = 0; t2 < T; t2++) {
503                        if (!wave[t2].contains(x2, y2)) continue;
504
505                        b = false;
506                        int[] prop = propagator[d][t2];
507                        for (int i1 = 0; i1 < prop.length && !b; i1++) {
508                            b = wave[prop[i1]].contains(x1, y1);
509                        }
510
511                        if (!b) {
512                            changes.insert(x2, y2);
513                            change = true;
514                            wave[t2].remove(x2, y2);
515                        }
516                    }
517                }
518            }
519        }
520
521        return change;
522    }

524    public char[][] output()
525    {
526        char[][] result = new char[FMX * tilesize][FMY * tilesize];
527
528        if (observed != null)
529        {
530            for (int x = 0; x < FMX; x++) {
531                for (int y = 0; y < FMY; y++)
532                {
533                    WaffleTile tile = tiles.get(observed[x][y]);
534                    for (int yt = 0; yt < tilesize; yt++) {
535                        for (int xt = 0; xt < tilesize; xt++)
536                        {
537                            result[x * tilesize + xt][y * tilesize + yt] = tile.get(xt, yt);
538                        }
539                    }
540                }
541            }
542        }
543        else
544        {
545            double[] weights = new double[T];
546            int[][] sums = GreasedRegion.sum(wave);
547            char c;
548            for (int x = 0; x < FMX; x++)
549            {
550                for (int y = 0; y < FMY; y++)
551                {
552                    int amount = sums[x][y];
553                    double ssum = 0.0;
554                    for (int i = 0; i < T; i++) {
555                        if(wave[i].contains(x, y)) ssum += stationary[i];
556                    }
557                    double lambda = 1.0 / (ssum);
558
559                    for (int yt = 0; yt < tilesize; yt++)
560                    {
561                        for (int xt = 0; xt < tilesize; xt++) {
562                            c = '#';
563                            Arrays.fill(weights, 0.0);
564                            for (int t = 0; t < T; t++) {
565                                if (wave[t].contains(x, y)) {
566                                    weights[t] = stationary[t] * lambda;
567                                    c = '.';
568                                }
569                            }
570                            if(c == '.')
571                                c = tiles.get(DetailedMimic.weightedRandom(weights, random.nextDouble())).get(xt, yt);
572                            result[x * tilesize + xt][y * tilesize + yt] = c;
573                        }
574                    }
575                }
576            }
577        }
578
579        return result;
580    }

616            if ((routerID & 3) != 0 && (routerID & 12) != 0)
617                indexer = Router.Generator.chain(w, h,
618                        Router.Generator.rotate(w, h, routerID & 3),
619                        Router.Generator.flip(w, h, (routerID & 4) != 0, (routerID & 8) != 0));
620            else if ((routerID & 3) != 0)
621                indexer = Router.Generator.rotate(w, h, routerID & 3);
622            else if ((routerID & 12) != 0)
623                indexer = Router.Generator.flip(w, h, (routerID & 4) != 0, (routerID & 8) != 0);
624            else
625                indexer = Router.Generator.simple(w, h);

635            weight = Integer.parseInt(encoded.substring(data.length + 17, data.length + 25), 16);
636            symmetry = encoded.charAt(data.length + 25);

638            if ((routerID & 3) != 0 && (routerID & 12) != 0)
639                indexer = Router.Generator.chain(w, h,
640                        Router.Generator.rotate(w, h, routerID & 3),
641                        Router.Generator.flip(w, h, (routerID & 4) != 0, (routerID & 8) != 0));
642            else if ((routerID & 3) != 0)
643                indexer = Router.Generator.rotate(w, h, routerID & 3);
644            else if ((routerID & 12) != 0)

694        public WaffleTile(char[][] map, char symmetry, int rotation, boolean flipX, boolean flipY) {
695            w = map.length;
696            h = map[0].length;
697            this.symmetry = symmetry;
698            data = new char[w * h];
699            for (int x = 0; x < w; x++) {
700                for (int y = 0; y < h; y++) {
701                    data[x + y * w] = map[x][y];
702                }
703            }
704            if ((rotation & 3) != 0 && (flipX || flipY)) {
705                indexer = Router.Generator.chain(w, h,
706                        Router.Generator.rotate(w, h, rotation),
707                        Router.Generator.flip(w, h, flipX, flipY));
708            } else if ((rotation & 3) != 0) {
709                indexer = Router.Generator.rotate(w, h, rotation);
710            } else if (flipX || flipY) {
711                indexer = Router.Generator.flip(w, h, flipX, flipY);
712            } else {
713                indexer = Router.Generator.simple(w, h);
714            }
715            routerID = (char) (rotation & 3);
716            if (flipX)
717                routerID |= 4;
718            if (flipY)
719                routerID |= 8;
720        }
