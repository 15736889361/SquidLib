012/**
013 * An implementation of the Wave Function Collapse algorithm for placing tiles in a way that satisfies constraints; the
014 * name both sounds like Wave Function and loosely describes how the algorithm works. All tiles start off in an
015 * indeterminate state, and as tiles are placed the states of their neighbors become more certain; the "Collapse" in
016 * Wave Function Collapse refers to the quantum-mechanics term for a decline in unpredictability as a superposition is
017 * measured, but here we use the term "Waffle" for its meaning as switching between positions or viewpoints until forced
018 * into a concrete state. This is meant to act like {@link DetailedMimic} or maybe {@link MimicFill}, but only requires
019 * tiles to be given to it with specific constraints on placement instead of requiring an existing design to mimic.
020 * <br>
021 * Massive credit goes to ExUtumno/mxgmn for developing the amazing Wave Function Collapse algorithm, which this code is
022 * closely based off of, but is not at all a verbatim copy. You can find out more at mxgmn's primary repository for WFC
023 * code, https://github.com/mxgmn/WaveFunctionCollapse , and see the incredible things more creative minds than my own
024 * have come up with using WFC or modifications on it. This file only implements the Simple Tiled Model for now.
025 * Created by Tommy Ettinger on 12/22/2016.
026 */

481    protected boolean propagate() {
482        boolean change = false, b;
483        for (int x2 = 0; x2 < FMX; x2++) {
484            for (int y2 = 0; y2 < FMY; y2++) {
485                for (int d = 0; d < 4; d++) {
486                    int x1 = x2, y1 = y2;
487                    if (d == 0) {
488                        if (x2 == 0) {
489                            if (!periodic) continue;
490                            else x1 = FMX - 1;
491                        } else x1 = x2 - 1;
492                    } else if (d == 1) {
493                        if (y2 == FMY - 1) {
494                            if (!periodic) continue;
495                            else y1 = 0;
496                        } else y1 = y2 + 1;
497                    } else if (d == 2) {
498                        if (x2 == FMX - 1) {
499                            if (!periodic) continue;
500                            else x1 = 0;
501                        } else x1 = x2 + 1;
502                    } else {
503                        if (y2 == 0) {
504                            if (!periodic) continue;
505                            else y1 = FMY - 1;
506                        } else y1 = y2 - 1;
507                    }
508
509                    if (!changes.contains(x1, y1)) continue;
510
511                    for (int t2 = 0; t2 < T; t2++) {
512                        if (!wave[t2].contains(x2, y2)) continue;
513
514                        b = false;
515                        int[] prop = propagator[d][t2];
516                        for (int i1 = 0; i1 < prop.length && !b; i1++) {
517                            b = wave[prop[i1]].contains(x1, y1);
518                        }
519
520                        if (!b) {
521                            changes.insert(x2, y2);
522                            change = true;
523                            wave[t2].remove(x2, y2);
524                        }
525                    }
526                }
527            }
528        }
529
530        return change;
531    }
532    protected Boolean observe() {
533        double min = 1E+3, sum, mainSum, logSum, noise, entropy;
534        int argminx = -1, argminy = -1, amount;
535        boolean[] w;
536
537        for (int x = 0; x < FMX; x++)
538            for (int y = 0; y < FMY; y++) {
539                //if (OnBoundary(x, y)) continue;
540
541                //w = wave[x][y];
542                amount = 0;
543                sum = 0;
544
545                for (int t = 0; t < T; t++)
546                    if (wave[t].contains(x, y)) {
547                        amount += 1;
548                        sum += stationary[t];
549                    }
550
551                if (sum == 0)
552                    return false;
553
554                noise = random.nextDouble(1E-6);
555
556                if (amount == 1) entropy = 0;
557                else if (amount == T) entropy = logT;
558                else {
559                    mainSum = 0;
560                    logSum = Math.log(sum);
561                    for (int t = 0; t < T; t++) if (wave[t].contains(x, y)) mainSum += stationary[t] * logProb[t];
562                    entropy = logSum - mainSum / sum;
563                }
564
565                if (entropy > 0 && entropy + noise < min) {
566                    min = entropy + noise;
567                    argminx = x;
568                    argminy = y;
569                }
570            }
571
572        if (argminx == -1 && argminy == -1) {
573            observed = new int[FMX][FMY];
574            for (int x = 0; x < FMX; x++) {
575                for (int y = 0; y < FMY; y++)
576                    for (int t = 0; t < T; t++)
577                        if (wave[t].contains(x, y)) {
578                            observed[x][y] = t;
579                            break;
580                        }
581            }
582
583            return true;
584        }
585
586        double[] distribution = new double[T];
587        for (int t = 0; t < T; t++) {
588            distribution[t] = wave[t].contains(argminx, argminy) ? stationary[t] : 0;
589        }
590        int r = DetailedMimic.weightedRandom(distribution, random.nextDouble());
591        for (int t = 0; t < T; t++) {
592            wave[t].set(t == r, argminx, argminy);
593        }
594        changes.insert(argminx, argminy);
595
596        return null;
597    }

599    public boolean run(RNG rng, int limit) {
600        logT = Math.log(T);
601        logProb = new double[T];
602        for (int t = 0; t < T; t++) logProb[t] = Math.log(stationary[t]);
603
604        clear();
605
606        random = rng;
607
608        for (int l = 0; l < limit || limit == 0; l++) {
609            Boolean result = observe();
610            if (result != null) return result;
611            while (propagate()) ;
612        }
613
614        return true;
615    }

622    public char[][] output()
623    {
624        char[][] result = new char[FMX * tilesize][FMY * tilesize];
625
626        if (observed != null)
627        {
628            for (int x = 0; x < FMX; x++) {
629                for (int y = 0; y < FMY; y++)
630                {
631                    WaffleTile tile = tiles.get(observed[x][y]);
632                    for (int yt = 0; yt < tilesize; yt++) {
633                        for (int xt = 0; xt < tilesize; xt++)
634                        {
635                            result[x * tilesize + xt][y * tilesize + yt] = tile.get(xt, yt);
636                        }
637                    }
638                }
639            }
640        }
641        else
642        {
643            double[] weights = new double[T];
644            char c;
645            for (int x = 0; x < FMX; x++)
646            {
647                for (int y = 0; y < FMY; y++)
648                {
649                    double ssum = 0.0;
650                    for (int i = 0; i < T; i++) {
651                        if(wave[i].contains(x, y)) ssum += stationary[i];
652                    }
653                    double lambda = 1.0 / (ssum);
654
655                    for (int yt = 0; yt < tilesize; yt++)
656                    {
657                        for (int xt = 0; xt < tilesize; xt++) {
658                            c = '#';
659                            Arrays.fill(weights, 0.0);
660                            for (int t = 0; t < T; t++) {
661                                if (wave[t].contains(x, y)) {
662                                    weights[t] = stationary[t] * lambda;
663                                    c = '.';
664                                }
665                            }
666                            if(c == '.')
667                                c = tiles.get(DetailedMimic.weightedRandom(weights, random.nextDouble())).get(xt, yt);
668                            result[x * tilesize + xt][y * tilesize + yt] = c;
669                        }
670                    }
671                }
672            }
673        }
674
675        return result;
676    }

683    public class WaffleTile implements Serializable {
684        private static final long serialVersionUID = 1L;
685
686        public char[] data;
687        public Router indexer;
688        protected String name = null;
689        protected char routerID = 0;
690        public int weight = 100;
691        public char symmetry;
692        public transient int leftSide, rightSide;
693
694        public WaffleTile() {
695            symmetry = 'X';
696            data = new char[tilesize * tilesize];
697            Arrays.fill(data, '.');
698            indexer = routers[0];
699            leftSide = sideHash(false);
700            rightSide = sideHash(true);
701        }
702
703        public WaffleTile(WaffleTile other) {
704            this(other.data, other.routerID, other.weight, other.symmetry, other.name);
705        }
706
707        public WaffleTile(char[] data, char routerID, int weight, char symmetry, String name) {
708            this.data = new char[data.length];
709            System.arraycopy(data, 0, this.data, 0, data.length);
710            this.weight = weight;
711            this.symmetry = symmetry;
712            this.name = name;
713            this.routerID = routerID;
714            this.indexer = routers[this.routerID &= 15];
715            leftSide = sideHash(false);
716            rightSide = sideHash(true);
717        }
718
719        public WaffleTile(String encoded) {
720            data = new char[encoded.length() - 10];
721            encoded.getChars(0, data.length, data, 0);
722            routerID = encoded.charAt(data.length);
723            weight = Integer.parseInt(encoded.substring(data.length + 1, data.length + 9), 16);
724            symmetry = encoded.charAt(data.length + 9);
725            indexer = routers[routerID &= 15];
726            leftSide = sideHash(false);
727            rightSide = sideHash(true);
728        }
729
730        public WaffleTile(char symmetry, int weight, String... map) {
731            this.symmetry = symmetry;
732            data = new char[tilesize * tilesize];
733            for (int x = 0; x < tilesize; x++) {
734                for (int y = 0; y < tilesize; y++) {
735                    data[x + y * tilesize] = map[y].charAt(x);
736                }
737            }
738            routerID = 0;
739            indexer = routers[0];
740            leftSide = sideHash(false);
741            rightSide = sideHash(true);
742            this.weight = weight;
743        }
744
745
746        public char get(int x, int y)
747        {
748            return data[indexer.reroute(x, y)];
749        }
750
751        public int sideHash(final boolean rightSide)
752        {
753            int z = 0x632BE5AB, result = 1, x = rightSide ? tilesize - 1 : 0;
754            for (int y = 1; y < tilesize-1; y++) {
755                result += (z ^= data[indexer.reroute(x, y)] * 0x85157AF5) + 0x62E2AC0D;
756            }
757            return result ^ ((z ^ result) >>> 8) * 0x9E3779B9;
758        }
759
760        public String getName() {
761            return (name != null) ? name : (name = StringKit.hex(CrossHash.Lightning.hash64(data)));
762        }
763
764        public void setName(String name) {
765            this.name = name;
766        }
767
768        public String serializeToString() {
769            return String.valueOf(data) + routerID + StringKit.hex(weight) + symmetry;
770        }
771
772        @Override
773        public boolean equals(Object o) {
774            if (this == o) return true;
775            if (o == null || getClass() != o.getClass()) return false;
776
777            WaffleTile that = (WaffleTile) o;
778
779            if (routerID != that.routerID) return false;
780            if (weight != that.weight) return false;
781            if (symmetry != that.symmetry) return false;
782            if (!Arrays.equals(data, that.data)) return false;
783            return name != null ? name.equals(that.name) : that.name == null;
784        }
785
786        @Override
787        public int hashCode() {
788            int result = Arrays.hashCode(data);
789            result = 31 * result + (name != null ? name.hashCode() : 0);
790            result = 31 * result + (int) routerID;
791            result = 31 * result + weight;
792            result = 31 * result + (int) symmetry;
793            return result;
794        }

796        public WaffleTile rotateCopy(int rotation) {
797            char id = routerID;
798            id &= 12;
799            id |= (rotation & 3);
800            return new WaffleTile(data, id, weight, symmetry, getName());
801        }
