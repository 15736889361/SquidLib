012/**
013 * Generates a procedural world map and fills it with the requested number of factions, keeping some land unclaimed.
014 * The factions are given procedural names in an atlas that is linked to the chars used by the world map.
015 * Uses MultiSpill internally to produce the semi-random land and water shapes, hence the name.
016 * <a href="https://gist.github.com/tommyettinger/4a16a09bebed8e2fe8473c8ea444a2dd">Example output</a>.
017 * Created by Tommy Ettinger on 9/12/2016.
018 */
019@Beta
020public class SpillWorldMap {
021    public int width;
022    public int height;
023    public StatefulRNG rng;
024    public String name;
025    public char[][] politicalMap;
026    protected static final char[] letters = Category.L.contents();
027    public final OrderedMap<Character, String> atlas = new OrderedMap<>(16);
028
029    public SpillWorldMap()
030    {
031        width = 20;
032        height = 20;
033        name = "Permadeath Island";
034        rng = new StatefulRNG(CrossHash.Lightning.hash64(name));
035    }
036    /**
037     * Constructs a SpillWorldMap using the given width, height, and world name, and uses the world name as the
038     * basis for all future random generation in this object.
039     *
040     * @param width     the width of the map in cells
041     * @param height    the height of the map in cells
042     * @param worldName a String name for the world that will be used as a seed for all random generation here
043     */
044    public SpillWorldMap(int width, int height, String worldName) {
045        this.width = Math.max(width, 20);
046        this.height = Math.max(height, 20);
047        name = worldName;
048        rng = new StatefulRNG(CrossHash.Lightning.hash64(name));
049    }
050
051    /**
052     * Generates a basic physical map for this world, then overlays a more involved political map with the given number
053     * of factions trying to take land in the world (essentially, nations). The output is a 2D char array where each
054     * letter char is tied to a different faction, while '~' is always water, and '%' is always wilderness or unclaimed
055     * land. If makeAtlas is true, it also generates an atlas with the procedural names of all the factions and a
056     * mapping to the chars used in the output; the atlas will be in the {@link #atlas} member of this object but will
057     * be empty if makeAtlas has never been true in a call to this.
058     * <br>
059     * If width or height is larger than 256, consider enlarging the Coord pool before calling this with
060     * {@code Coord.expandPoolTo(width, height);}. This will have no effect if width and height are both less than or
061     * equal to 256, but if you expect to be using maps that are especially large (which makes sense for world maps),
062     * expanding the pool will use more memory initially and then (possibly) much less over time, easing pressure on
063     * the garbage collector as well, as re-allocations of large Coords that would otherwise be un-cached are avoided.
064     * @param factionCount the number of factions to have claiming land
065     * @param makeAtlas if true, this will assign random names to factions, accessible via {@link #atlas}
066     * @return a 2D char array where letters represent the claiming faction, '~' is water, and '%' is unclaimed
067     */
068    public char[][] generate(int factionCount, boolean makeAtlas) {
069        //, extra = 25 + (height * width >>> 4);
070        MultiSpill spreader = new MultiSpill(new short[width][height], Spill.Measurement.MANHATTAN, rng);
071        GreasedRegion bounds = new GreasedRegion(width, height).not().retract(4),
072                smallerBounds = bounds.copy().retract(5), area = new GreasedRegion(width, height),
073                tmpEdge = new GreasedRegion(width, height), tmpInner = new GreasedRegion(width, height),
074                tmpOOB = new GreasedRegion(width, height);
075        Coord[] pts = smallerBounds.randomPortion(rng, rng.between(24, 48));
076        int aLen = pts.length;
077        short[][] sm = new short[width][height], tmpSM;
078        Arrays.fill(sm[0], (short) (-1));
079        for (int i = 1; i < width; i++) {
080            System.arraycopy(sm[0], 0, sm[i], 0, height);
081        }
082        OrderedMap<Coord, Double> entries = new OrderedMap<>();
083        OrderedSet<Coord> oob;
084        for (int i = 0; i < aLen; i++) {
085            int volume = 10 + (height * width) / (aLen * 2);
086            area.clear().insert(pts[i]).spill(bounds, volume, rng).expand8way(2);
087            tmpInner.remake(area).retract(4).expand8way().and(area);
088            tmpEdge.remake(area).surface8way();
089            Coord[] edges = tmpEdge.separatedPortion(0.5);
090            int eLen = edges.length;
091            Double[] powers = new Double[eLen];
092            Arrays.fill(powers, 0.1);
093            entries = new OrderedMap<>(edges, powers);
094            eLen = entries.size();
095            for (int j = 0; j < 32; j++) {
096                entries.put(tmpInner.singleRandom(rng), (rng.nextDouble() + 1.5) * 0.4);
097            }
098            tmpOOB.remake(area).not();
099            oob = new OrderedSet<>(tmpOOB.asCoords());
100
101            spreader.start(entries, -1, oob);
102            tmpSM = spreader.spillMap;
103
104            for (int x = 0; x < width; x++) {
105                for (int y = 0; y < height; y++) {
106                    if (tmpSM[x][y] >= eLen)
107                        sm[x][y] = 0;
108                }
109            }
110
111        }
112        /*
113        // old version; produced a strange "swiss cheese" world map layout.
114        SobolQRNG sobol = new SobolQRNG(2);
115        sobol.skipTo(rng.between(1000, 6500));
116        //sobol.fillVector(filler);
117        //entries.put(Coord.get((int)(dim * filler[0]), (int)(dim * filler[1])), (filler[2] + 0.25) / 1.25);
118        for (int j = 0; j < 8; j++) {

120        }

122            entries.put(sobol.nextCoord(width - 16, height - 16).add(8), (rng.nextDouble() + 0.25) * 0.3);
123        }
124        count = entries.size();
125        double edgePower = 0.95;// count * 0.8 / extra;
126        for (int x = 1; x < width - 1; x += 4) {
127            entries.put(Coord.get(x, 0), edgePower);
128            entries.put(Coord.get(x, height - 1), edgePower);
129        }
130        for (int y = 1; y < height - 1; y += 4) {
131            entries.put(Coord.get(0, y), edgePower);
132            entries.put(Coord.get(width - 1, y), edgePower);
133        }
134        spreader.start(entries, -1, null);
135        short[][] sm = spreader.spillMap;
136        for (int x = 0; x < width; x++) {
137            for (int y = 0; y < height; y++) {
138                //blank[x][y] = (char) ('a' + Integer.bitCount(sm[x][y] + 7) % 26);
139                if (sm[x][y] >= count || (sm[x][y] & 7) == 0)
140                    sm[x][y] = -1;
141                else
142                    sm[x][y] = 0;
143            }
144        }
145        */
146        GreasedRegion map = new GreasedRegion(sm, 0, 0x7fff);
147        Coord[] centers = map.randomSeparated(0.1, rng, factionCount);
148        int controlled = (int) (map.size() * rng.between(0.9, 1.0));
149
150        spreader.initialize(sm);
151        entries.clear();
152        entries.put(Coord.get(-1, -1), 0.0);
153        for (int i = 0; i < factionCount; i++) {
154            entries.put(centers[i], rng.between(0.5, 1.0));
155        }
156        spreader.start(entries, controlled, null);
157        sm = spreader.spillMap;
158        politicalMap = new char[width][height];
159        for (int x = 0; x < width; x++) {
160            for (int y = 0; y < height; y++) {
161                politicalMap[x][y] = (sm[x][y] == -1) ? '~' : (sm[x][y] == 0) ? '%' : letters[(sm[x][y] - 1) & 511];
162            }
163        }
164        if (makeAtlas) {
165            atlas.clear();
166            atlas.put('~', "Water");
167            atlas.put('%', "Wilderness");
168            Thesaurus th = new Thesaurus(rng.nextLong());
169            th.addKnownCategories();
170            for (int i = 0; i < factionCount && i < 512; i++) {
171                atlas.put(letters[i], th.makeNationName());
172            }
173        }
174        return politicalMap;
175    }
176    /**
177     * Generates a basic physical map for this world, then overlays a more involved political map with the given number
178     * of factions trying to take land in the world (essentially, nations). The output is a 2D char array where each
179     * letter char is tied to a different faction, while '~' is always water, and '%' is always wilderness or unclaimed
180     * land. Does not generate an atlas, so you should come up with meanings for the letters yourself.
181     *
182     * @param factionCount the number of factions to have claiming land
183     * @return a 2D char array where letters represent the claiming faction, '~' is water, and '%' is unclaimed
184     */
185    public char[][] generate(int factionCount) {
186        return generate(factionCount, false);
187    }
