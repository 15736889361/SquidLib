007/**
008 * Can be used to generate world maps with a wide variety of data, starting with height, temperature and moisture.
009 * From there, you can determine biome information in as much detail as your game needs, with a default implementation
010 * available that assigns a single biome to each cell based on heat/moisture. The maps this produces are valid
011 * for spherical or toroidal world projections, and will wrap from edge to opposite edge seamlessly thanks to a
012 * technique from the Accidental Noise Library ( https://www.gamedev.net/blog/33/entry-2138456-seamless-noise/ ) that
013 * involves getting a 2D slice of 4D Simplex noise. Because of how Simplex noise works, this also allows extremely high
014 * zoom levels as long as certain parameters are within reason. You can access the height map with the
015 * {@link #heightData} field, the heat map with the {@link #heatData} field, the moisture map with the
016 * {@link #moistureData} field, and a special map that stores ints representing the codes for various ranges of
017 * elevation (0 to 8 inclusive, with 0 the deepest ocean and 8 the highest mountains) with {@link #heightCodeData}. The
018 * last map should be noted as being the simplest way to find what is land and what is water; any height code 4 or
019 * greater is land, and any height code 3 or less is water. This can produce rivers, and keeps that information in a
020 * GreasedRegion (alongside a GreasedRegion containing lake positions) instead of in the other map data. This class does
021 * not use Coord at all, but if you want maps with width and/or height greater than 256, and you want to use the river
022 * or lake data as a Collection of Coord, then you should call {@link squidpony.squidmath.Coord#expandPoolTo(int, int)}
023 * with your width and height so the Coords remain safely pooled. If you're fine with keeping rivers and lakes as
024 * GreasedRegions and not requesting Coord values from them, then you don't need to do anything with Coord. Certain
025 * parts of this class are not necessary to generate, just in case you want river-less maps or something similar;
026 * setting {@link #generateRivers} to false will disable river generation (it defaults to true).
027 * <br>
028 * The main trade-off this makes to obtain better quality is reduced speed; generating a 512x512 map on a circa-2016
029 * laptop (i7-6700HQ processor at 2.6 GHz) takes about 1 second (about 1.15 seconds for an un-zoomed map, 0.95 or so
030 * seconds to increase zoom at double resolution). If you don't need a 512x512 map, this takes commensurately less time
031 * to generate less grid cells, with 64x64 maps generating faster than they can be accurately seen on the same hardware.
032 * River positions are produced using a different method, and do not involve the Simplex noise parts other than using
033 * the height map to determine flow. Zooming with rivers is tricky, and generally requires starting from the outermost
034 * zoom level and progressively enlarging and adding detail to all rivers as zoom increases on specified points.
035 */

131    /**
132     * Generates a world using the specified RNG state as a long, with specific water and cooling modifiers that affect
133     * the land-water ratio and the average temperature, respectively.
134     * The worlds this produces will always have width and height as specified in the constructor (default 256x256).
135     * You can call {@link #zoomIn(int, int, int)} to double the resolution and center on the specified area, but the width
136     * and height of the 2D arrays this changed, such as {@link #heightData} and {@link #moistureData} will be the same.
137     * @param waterMod should be between 0.85 and 1.2; a random value will be used if this is negative
138     * @param coolMod should be between 0.85 and 1.4; a random value will be used if this is negative
139     * @param state the state to give this generator's RNG; if the same as the last call, this will reuse data
140     */
141    public void generate(double waterMod, double coolMod, long state)
142    {
143        if(cachedState != state || waterMod != waterModifier || coolMod != coolingModifier)
144        {
145            seed = state;
146            zoom = 0;
147            startCacheX.clear();
148            startCacheY.clear();
149            startCacheX.add(0);
150            startCacheY.add(0);
151
152        }
153        regenerate(startCacheX.peek(), startCacheY.peek(),
154                (width >> zoom), (height >> zoom), waterMod, coolMod, state);
155    }

167    /**
168     * Halves the resolution of the map and doubles the area it covers; the 2D arrays this uses keep their sizes. This
169     * version of zoomOut allows you to specify where the zoom should be centered, using the current coordinates (if the
170     * map size is 256x256, then coordinates should be between 0 and 255, and will refer to the currently used area and
171     * not necessarily the full world size).
172     * <br>
173     * Only has an effect if you have previously zoomed in using {@link #zoomIn(int, int, int)} or its overload.
174     * @param zoomCenterX the center X position to zoom out from; if too close to an edge, this will stop moving before it would extend past an edge
175     * @param zoomCenterY the center Y position to zoom out from; if too close to an edge, this will stop moving before it would extend past an edge
176     */
177    public void zoomOut(int zoomAmount, int zoomCenterX, int zoomCenterY)
178    {
179        if(zoomAmount == 0) return;
180        if(zoomAmount < 0) {
181            zoomIn(-zoomAmount, zoomCenterX, zoomCenterY);
182            return;
183        }
184        if(zoom > 0)
185        {
186            if(seed != cachedState)
187            {
188                generate(rng.nextLong());
189            }
190
191            zoom -= zoomAmount;
192            startCacheX.pop();
193            startCacheY.pop();
194            startCacheX.add(Math.min(Math.max(startCacheX.pop() + (zoomCenterX >> zoom + 1) - (width >> zoom + 2),
195                    0), width - (width >> zoom)));
196            startCacheY.add(Math.min(Math.max(startCacheY.pop() + (zoomCenterY >> zoom + 1) - (height >> zoom + 2),
197                    0), height - (height >> zoom)));
198            regenerate(startCacheX.peek(), startCacheY.peek(),width >> zoom, height >> zoom,
199                    waterModifier, coolingModifier, cachedState);
200            rng.setState(cachedState);
201        }
202
203    }

208     * Although there is no technical restriction on maximum zoom, zooming in more than 5 times (64x scale or greater)

216    /**
217     * Doubles the resolution of the map and halves the area it covers; the 2D arrays this uses keep their sizes. This
218     * version of zoomIn allows you to specify where the zoom should be centered, using the current coordinates (if the
219     * map size is 256x256, then coordinates should be between 0 and 255, and will refer to the currently used area and
220     * not necessarily the full world size).
221     * <br>
222     * Although there is no technical restriction on maximum zoom, zooming in more than 5 times (64x scale or greater)
223     * will make the map appear somewhat less realistic due to rounded shapes appearing more bubble-like and less like a
224     * normal landscape.
225     * @param zoomCenterX the center X position to zoom in to; if too close to an edge, this will stop moving before it would extend past an edge
226     * @param zoomCenterY the center Y position to zoom in to; if too close to an edge, this will stop moving before it would extend past an edge
227     */
228    public void zoomIn(int zoomAmount, int zoomCenterX, int zoomCenterY)
229    {
230        if(zoomAmount == 0) return;
231        if(zoomAmount < 0)
232        {
233            zoomOut(-zoomAmount, zoomCenterX, zoomCenterY);
234            return;
235        }
236        if(seed != cachedState)
237        {
238            generate(rng.nextLong());
239        }
240        zoom += zoomAmount;
241        if(startCacheX.isEmpty())
242        {
243            startCacheX.add(0);
244            startCacheY.add(0);
245        }
246        else {
247            startCacheX.add(Math.min(Math.max(startCacheX.peek() + (zoomCenterX >> zoom - 1) - (width >> zoom + 1),
248                    0), width - (width >> zoom)));
249            startCacheY.add(Math.min(Math.max(startCacheY.peek() + (zoomCenterY >> zoom - 1) - (height >> zoom + 1),
250                    0), height - (height >> zoom)));
251        }
252        regenerate(startCacheX.peek(), startCacheY.peek(),width >> zoom, height >> zoom,
253                waterModifier, coolingModifier, cachedState);
254        rng.setState(cachedState);
255    }

257    protected void regenerate(int startX, int startY, int usedWidth, int usedHeight,
258                              double waterMod, double coolMod, long state)
259    {
260        boolean fresh = false;
261        if(cachedState != state || waterMod != waterModifier || coolMod != coolingModifier)
262        {
263            minHeight = Double.POSITIVE_INFINITY;
264            maxHeight = Double.NEGATIVE_INFINITY;
265            minHeat0 = Double.POSITIVE_INFINITY;
266            maxHeat0 = Double.NEGATIVE_INFINITY;
267            minHeat1 = Double.POSITIVE_INFINITY;
268            maxHeat1 = Double.NEGATIVE_INFINITY;
269            minHeat = Double.POSITIVE_INFINITY;
270            maxHeat = Double.NEGATIVE_INFINITY;
271            minWet0 = Double.POSITIVE_INFINITY;
272            maxWet0 = Double.NEGATIVE_INFINITY;
273            minWet = Double.POSITIVE_INFINITY;
274            maxWet = Double.NEGATIVE_INFINITY;
275            cachedState = state;
276            fresh = true;
277        }
278        rng.setState(state);
279        int seedA = rng.nextInt(), seedB = rng.nextInt(), seedC = rng.nextInt(), t;
280
281        waterModifier = (waterMod <= 0) ? rng.nextDouble(0.25) + 0.89 : waterMod;
282        coolingModifier = (coolMod <= 0) ? rng.nextDouble(0.45) * (rng.nextDouble()-0.5) + 1.1 : coolMod;
283
284        double p, q,
285                ps, pc,
286                qs, qc,
287                h, temp,
288                i_w = 6.283185307179586 / width, i_h = 6.283185307179586 / height,
289                xPos = startX, yPos = startY, i_uw = usedWidth / (double)width, i_uh = usedHeight / (double)height;
290        double[] trigTable = new double[width << 1];
291        for (int x = 0; x < width; x++, xPos += i_uw) {
292            p = xPos * i_w;
293            trigTable[x<<1]   = Math.sin(p);
294            trigTable[x<<1|1] = Math.cos(p);
295        }
296        for (int y = 0; y < height; y++, yPos += i_uh) {
297            q = yPos * i_h;
298            qs = Math.sin(q);
299            qc = Math.cos(q);
300            for (int x = 0, xt = 0; x < width; x++) {
301                ps = trigTable[xt++];//Math.sin(p);
302                pc = trigTable[xt++];//Math.cos(p);
303                h = terrain.getNoiseWithSeed(pc +
304                                terrainRidged.getNoiseWithSeed(pc, ps, qc, qs, seedA + seedB),
305                        ps, qc, qs, seedA);
306                h *= waterModifier;
307                heightData[x][y] = h;
308                heatData[x][y] = (p = heat.getNoiseWithSeed(pc, ps, qc
309                                + otherRidged.getNoiseWithSeed(pc, ps, qc, qs, seedB + seedC)
310                        , qs, seedB));
311                moistureData[x][y] = (temp = moisture.getNoiseWithSeed(pc, ps, qc, qs
312                                + otherRidged.getNoiseWithSeed(pc, ps, qc, qs, seedC + seedA)
313                        , seedC));
314                minHeightActual = Math.min(minHeightActual, h);
315                maxHeightActual = Math.max(maxHeightActual, h);
316                if(fresh) {
317                    minHeight = Math.min(minHeight, h);
318                    maxHeight = Math.max(maxHeight, h);
319
320                    minHeat0 = Math.min(minHeat0, p);
321                    maxHeat0 = Math.max(maxHeat0, p);
322
323                    minWet0 = Math.min(minWet0, temp);
324                    maxWet0 = Math.max(maxWet0, temp);
325
326                }
327            }
328            minHeightActual = Math.min(minHeightActual, minHeight);
329            maxHeightActual = Math.max(maxHeightActual, maxHeight);
330
331        }
332        double heightDiff = 2.0 / (maxHeightActual - minHeightActual),
333                heatDiff = 0.8 / (maxHeat0 - minHeat0),
334                wetDiff = 1.0 / (maxWet0 - minWet0),
335                hMod,
336                halfHeight = (height - 1) * 0.5, i_half = 1.0 / halfHeight;
337        double minHeightActual0 = minHeightActual;
338        double maxHeightActual0 = maxHeightActual;
339        yPos = startY;
340        ps = Double.POSITIVE_INFINITY;
341        pc = Double.NEGATIVE_INFINITY;
342
343        for (int y = 0; y < height; y++, yPos += i_uh) {
344            temp = Math.abs(yPos - halfHeight) * i_half;
345            temp *= (2.4 - temp);
346            temp = 2.2 - temp;
347            for (int x = 0; x < width; x++) {
348                heightData[x][y] = (h = (heightData[x][y] - minHeightActual) * heightDiff - 1.0);
349                minHeightActual0 = Math.min(minHeightActual0, h);
350                maxHeightActual0 = Math.max(maxHeightActual0, h);
351                heightCodeData[x][y] = (t = codeHeight(h));
352                hMod = 1.0;
353                switch (t) {
354                    case 0:
355                    case 1:
356                    case 2:
357                    case 3:
358                        h = 0.4;
359                        hMod = 0.2;
360                        break;
361                    case 6:
362                        h = -0.1 * (h - forestLower - 0.08);
363                        break;
364                    case 7:
365                        h *= -0.25;
366                        break;
367                    case 8:
368                        h *= -0.4;
369                        break;
370                    default:
371                        h *= 0.05;
372                }
373                heatData[x][y] = (h = (((heatData[x][y] - minHeat0) * heatDiff * hMod) + h + 0.6) * temp);
374                if (fresh) {
375                    ps = Math.min(ps, h); //minHeat0
376                    pc = Math.max(pc, h); //maxHeat0
377                }
378            }
379        }
380        if(fresh)
381        {
382            minHeat1 = ps;
383            maxHeat1 = pc;
384        }
385        heatDiff = coolingModifier / (maxHeat1 - minHeat1);
386        qs = Double.POSITIVE_INFINITY;
387        qc = Double.NEGATIVE_INFINITY;
388        ps = Double.POSITIVE_INFINITY;
389        pc = Double.NEGATIVE_INFINITY;
390
391
392        for (int y = 0; y < height; y++) {
393            for (int x = 0; x < width; x++) {
394                heatData[x][y] = (h = ((heatData[x][y] - minHeat1) * heatDiff));
395                moistureData[x][y] = (temp = (moistureData[x][y] - minWet0) * wetDiff);
396                if (fresh) {
397                    qs = Math.min(qs, h);
398                    qc = Math.max(qc, h);
399                    ps = Math.min(ps, temp);
400                    pc = Math.max(pc, temp);
401                }
402            }
403        }
404        if(fresh)
405        {
406            minHeat = qs;
407            maxHeat = qc;
408            minWet = ps;
409            maxWet = pc;
410        }
411        if(generateRivers) {
412            if (fresh) {
413                addRivers();
414                riverData.connect8way().thin().thin();
415                partialRiverData.remake(riverData);
416                partialLakeData.remake(lakeData);
417            } else {
418                partialRiverData.remake(riverData);
419                partialLakeData.remake(lakeData);
420                for (int i = 1; i <= zoom; i++) {
421                    int stx = (startCacheX.get(i) - startCacheX.get(i - 1)) << (i - 1),
422                            sty = (startCacheY.get(i) - startCacheY.get(i - 1)) << (i - 1);
423                    if ((i & 3) == 3) {
424                        partialRiverData.zoom(stx, sty).connect8way();
425                        partialRiverData.or(workingData.remake(partialRiverData).fringe().quasiRandomRegion(0.4));
426                        partialLakeData.zoom(stx, sty).connect8way();
427                        partialLakeData.or(workingData.remake(partialLakeData).fringe().quasiRandomRegion(0.55));
428                    } else {
429                        partialRiverData.zoom(stx, sty).connect8way().thin();
430                        partialRiverData.or(workingData.remake(partialRiverData).fringe().quasiRandomRegion(0.5));
431                        partialLakeData.zoom(stx, sty).connect8way().thin();
432                        partialLakeData.or(workingData.remake(partialLakeData).fringe().quasiRandomRegion(0.7));
433                    }
434                }
435            }
436        }
437    }
438    public int codeHeight(final double high)
439    {
440        if(high < deepWaterUpper)
441            return 0;
442        if(high < mediumWaterUpper)
443            return 1;
444        if(high < shallowWaterUpper)
445            return 2;
446        if(high < coastalWaterUpper)
447            return 3;
448        if(high < sandUpper)
449            return 4;
450        if(high < grassUpper)
451            return 5;
452        if(high < forestUpper)
453            return 6;
454        if(high < rockUpper)
455            return 7;
456        return 8;
457    }

466    private static final Direction[] reuse = new Direction[9];
467    private void appendDirToShuffle(RNG rng) {
468        rng.randomPortion(Direction.CARDINALS, reuse);
469        reuse[rng.next(2)] = Direction.DIAGONALS[rng.next(2)];
470        reuse[4] = Direction.DIAGONALS[rng.next(2)];
471        reuse[5] = Direction.OUTWARDS[rng.next(3)];
472    }
473
474    protected void addRivers()
475    {
476        long rebuildState = rng.nextLong();
477        workingData.empty().insertRectangle(8, 8, width - 16, height - 16);
478        riverData.empty().refill(heightCodeData, 6, 100);
479        riverData.quasiRandomRegion(0.000003 * Math.max(width, height)).and(workingData);
480        int[] starts = riverData.asTightEncoded();
481        int len = starts.length, currentPos, choice, adjX, adjY, currX, currY, tcx, tcy, stx, sty, sbx, sby;
482        riverData.clear();
483        lakeData.clear();
484        PER_RIVER:
485        for (int i = 0; i < len; i++) {
486            workingData.clear();
487            currentPos = starts[i];
488            stx = tcx = currX = decodeX(currentPos);
489            sty = tcy = currY = decodeY(currentPos);
490            while (true) {
491
492                double best = 999999;
493                choice = -1;
494                appendDirToShuffle(rng);
495
496                for (int d = 0; d < 5; d++) {
497                    adjX = (currX + reuse[d].deltaX);
498                    if (adjX < 0 || adjX >= width)
499                    {
500                        if(rng.next(4) == 0)
501                            riverData.or(workingData);
502                        continue PER_RIVER;
503                    }
504                    adjY = (currY + reuse[d].deltaY);
505                    if (adjY < 0 || adjY >= height)
506                    {
507                        if(rng.next(4) == 0)
508                            riverData.or(workingData);
509                        continue PER_RIVER;
510                    }
511                    if (heightData[adjX][adjY] < best && !workingData.contains(adjX, adjY)) {
512                        best = heightData[adjX][adjY];
513                        choice = d;
514                        tcx = adjX;
515                        tcy = adjY;
516                    }
517                }
518                currX = tcx;
519                currY = tcy;
520                if (best >= heightData[stx][sty]) {
521                    tcx = rng.next(2);
522                    adjX = currX + ((tcx & 1) << 1) - 1;
523                    adjY = currY + (tcx & 2) - 1;
524                    lakeData.insert(currX, currY);
525                    lakeData.insert(currX + 1, currY);
526                    lakeData.insert(currX - 1, currY);
527                    lakeData.insert(currX, currY + 1);
528                    lakeData.insert(currX, currY - 1);
529                    if (!(adjY < 0 || adjY >= height || adjX < 0 || adjX >= width))
530                    {
531                        if(heightCodeData[adjX][adjY] <= 3) {
532                            riverData.or(workingData);
533                            continue PER_RIVER;
534                        }
535                        else if((heightData[adjX][adjY] -= 0.0002) < lowers[heightCodeData[adjX][adjY]-1])
536                        {
537                            if(rng.next(3) == 0)
538                                riverData.or(workingData);
539                            continue PER_RIVER;
540                        }
541                    }
542                    else
543                    {
544                        if(rng.next(5) == 0)
545                            riverData.or(workingData);
546                        continue PER_RIVER;
547                    }
548                    tcx = rng.next(2);
549                    adjX = currX + ((tcx & 1) << 1) - 1;
550                    adjY = currY + (tcx & 2) - 1;
551                    if (!(adjY < 0 || adjY >= height || adjX < 0 || adjX >= width))
552                    {
553                        if(heightCodeData[adjX][adjY] <= 3) {
554                            riverData.or(workingData);
555                            continue PER_RIVER;
556                        }
557                        else if((heightData[adjX][adjY] -= 0.0002) < lowers[heightCodeData[adjX][adjY]-1])
558                        {
559                            if(rng.next(3) == 0)
560                                riverData.or(workingData);
561                            continue PER_RIVER;
562                        }
563                    }
564                    else
565                    {
566                        if(rng.next(5) == 0)
567                            riverData.or(workingData);
568                        continue PER_RIVER;
569                    }
570
571                }
572                if(choice != -1 && reuse[choice].isDiagonal())
573                {
574                    tcx = currX - reuse[choice].deltaX;
575                    tcy = currY - reuse[choice].deltaY;
576                    if(heightData[tcx][currY] <= heightData[currX][tcy] && !workingData.contains(tcx, currY))
577                    {
578                        if(riverData.contains(tcx, currY))
579                        {
580                            riverData.or(workingData);
581                            continue PER_RIVER;
582                        }
583                        workingData.insert(tcx, currY);
584                        if (heightData[tcx][currY] <= 0.075)
585                        {
586                            riverData.or(workingData);
587                            continue PER_RIVER;
588                        }
589                    }
590                    else if(!workingData.contains(currX, tcy))
591                    {
592                        if(riverData.contains(currX, tcy))
593                        {
594                            riverData.or(workingData);
595                            continue PER_RIVER;
596                        }
597                        workingData.insert(currX, tcy);
598                        if (heightData[currX][tcy] <= 0.075)
599                        {
600                            riverData.or(workingData);
601                            continue PER_RIVER;
602                        }
603
604                    }
605                }
606                if(riverData.contains(currX, currY))
607                {
608                    riverData.or(workingData);
609                    continue PER_RIVER;
610                }
611                workingData.insert(currX, currY);
612                if (heightData[currX][currY] <= 0.075)
613                {
614                    riverData.or(workingData);
615                    continue PER_RIVER;
616                }
617            }
618        }
619
620        GreasedRegion tempData = new GreasedRegion(width, height);
621        int riverCount = riverData.size() >> 3, currentMax = riverCount >> 2, idx = 0, prevChoice;
622        for (int h = 5; h < 9; h++) { //, currentMax += riverCount / 18
623            workingData.empty().refill(heightCodeData, h).and(riverData);
624            RIVER:
625            for (int j = 0; j < currentMax && idx < riverCount; j++) {
626                double vdc = VanDerCorputQRNG.weakDetermine(idx++), best = -999999;
627                currentPos = workingData.atFractionTight(vdc);
628                if(currentPos < 0)
629                    break;
630                stx = sbx = tcx = currX = decodeX(currentPos);
631                sty = sby = tcy = currY = decodeY(currentPos);
632                appendDirToShuffle(rng);
633                choice = -1;
634                prevChoice = -1;
635                for (int d = 0; d < 5; d++) {
636                    adjX = (currX + reuse[d].deltaX);
637                    if (adjX < 0 || adjX >= width)
638                        continue;
639                    adjY = (currY + reuse[d].deltaY);
640                    if (adjY < 0 || adjY >= height)
641                        continue;
642                    if (heightData[adjX][adjY] > best) {
643                        best = heightData[adjX][adjY];
644                        prevChoice = choice;
645                        choice = d;
646                        sbx = tcx;
647                        sby = tcy;
648                        tcx = adjX;
649                        tcy = adjY;
650                    }
651                }
652                currX = sbx;
653                currY = sby;
654                if (prevChoice != -1 && heightCodeData[currX][currY] >= 4) {
655                    if (reuse[prevChoice].isDiagonal()) {
656                        tcx = currX - reuse[prevChoice].deltaX;
657                        tcy = currY - reuse[prevChoice].deltaY;
658                        if (heightData[tcx][currY] <= heightData[currX][tcy])
659                            tempData.insert(tcx, currY);
660                        else
661                            tempData.insert(currX, tcy);
662                    }
663                    tempData.insert(currX, currY);
664                }
665
666                while (true) {
667                    best = -999999;
668                    appendDirToShuffle(rng);
669                    choice = -1;
670                    for (int d = 0; d < 6; d++) {
671                        adjX = (currX + reuse[d].deltaX);
672                        if (adjX < 0 || adjX >= width)
673                            continue;
674                        adjY = (currY + reuse[d].deltaY);
675                        if (adjY < 0 || adjY >= height)
676                            continue;
677                        if (heightData[adjX][adjY] > best && !riverData.contains(adjX, adjY)) {
678                            best = heightData[adjX][adjY];
679                            choice = d;
680                            sbx = adjX;
681                            sby = adjY;
682                        }
683                    }
684                    currX = sbx;
685                    currY = sby;
686                    if (choice != -1 && heightCodeData[currX][currY] >= 4) {
687                        if (reuse[choice].isDiagonal()) {
688                            tcx = currX - reuse[choice].deltaX;
689                            tcy = currY - reuse[choice].deltaY;
690                            if (heightData[tcx][currY] <= heightData[currX][tcy])
691                                tempData.insert(tcx, currY);
692                            else
693                                tempData.insert(currX, tcy);
694                        }
695                        tempData.insert(currX, currY);
696                    }
697                    if (best <= heightData[stx][sty] || heightData[currX][currY] > rng.nextDouble(280.0)) {
698                        riverData.or(tempData);
699                        tempData.clear();
700
701                        lakeData.insert(currX, currY);
702                        sbx = rng.next(8);
703                        sbx &= sbx >>> 4;
704                        if ((sbx & 1) == 0)
705                            lakeData.insert(currX + 1, currY);
706                        if ((sbx & 2) == 0)
707                            lakeData.insert(currX - 1, currY);
708                        if ((sbx & 4) == 0)
709                            lakeData.insert(currX, currY + 1);
710                        if ((sbx & 8) == 0)
711                            lakeData.insert(currX, currY - 1);
712                        sbx = rng.next(2);
713                        lakeData.insert(currX + (-(sbx & 1) | 1), currY + ((sbx & 2) - 1)); // random diagonal
714                        lakeData.insert(currX, currY + ((sbx & 2) - 1)); // ortho next to random diagonal
715                        lakeData.insert(currX + (-(sbx & 1) | 1), currY); // ortho next to random diagonal
716
717                        continue RIVER;
718                    }
719                }
720            }
721
722        }
723
724        rng.setState(rebuildState);
725    }

727    /**
728     * A way to get biome information for the cells on a map when you only need a single value to describe a biome, such
729     * as "Grassland" or "TropicalRainforest".
730     * <br>
731     * To use: 1, Construct a SimpleBiomeMapper (constructor takes no arguments). 2, call
732     * {@link #makeBiomes(WorldMapGenerator)} with a WorldMapGenerator that has already produced at least one world map.
733     * 3, get biome codes from the {@link #biomeCodeData} field, where a code is an int that can be used as an index
734     * into the {@link #biomeTable} static field to get a String name for a biome type, or used with an alternate biome
735     * table of your design. Biome tables in this case are 54-element arrays organized into groups of 6 elements. Each
736     * group goes from the coldest temperature first to the warmest temperature last in the group. The first group of 6
737     * contains the dryest biomes, the next 6 are medium-dry, the next are slightly-dry, the next slightly-wet, then
738     * medium-wet, then wettest. After this first block of dry-to-wet groups, there is a group of 6 for coastlines, a
739     * group of 6 for rivers, and lastly a group for lakes. This also assigns moisture codes and heat codes from 0 to 5
740     * for each cell, which may be useful to simplify logic that deals with those factors.
741     */
742    public static class SimpleBiomeMapper
743    {
744        /**
745         * The heat codes for the analyzed map, from 0 to 5 inclusive, with 0 coldest and 5 hottest.
746         */
747        public int[][] heatCodeData,
748        /**
749         * The moisture codes for the analyzed map, from 0 to 5 inclusive, with 0 driest and 5 wettest.
750         */
751        moistureCodeData,
752        /**
753         * The biome codes for the analyzed map, from 0 to 53 inclusive. You can use {@link #biomeTable} to look up
754         * String names for biomes, or construct your own table as you see fit (see docs in {@link SimpleBiomeMapper}).
755         */
756        biomeCodeData;
757
758        public static final double
759                coldestValueLower = 0.0,   coldestValueUpper = 0.15, // 0
760                colderValueLower = 0.15,   colderValueUpper = 0.31,  // 1
761                coldValueLower = 0.31,     coldValueUpper = 0.5,     // 2
762                warmValueLower = 0.5,      warmValueUpper = 0.69,    // 3
763                warmerValueLower = 0.69,   warmerValueUpper = 0.85,  // 4
764                warmestValueLower = 0.85,  warmestValueUpper = 1.0,  // 5
765        
766                driestValueLower = 0.0,    driestValueUpper  = 0.27, // 0
767                drierValueLower = 0.27,    drierValueUpper   = 0.4,  // 1
768                dryValueLower = 0.4,       dryValueUpper     = 0.6,  // 2
769                wetValueLower = 0.6,       wetValueUpper     = 0.8,  // 3
770                wetterValueLower = 0.8,    wetterValueUpper  = 0.9,  // 4
771                wettestValueLower = 0.9,   wettestValueUpper = 1.0;  // 5
772
773        /**
774         * The default biome table to use with biome codes from {@link #biomeCodeData}. Biomes are assigned based on
775         * heat and moisture for the first 36 of 54 elements (coldest to warmest for each group of 6, with the first
776         * group as the dryest and the last group the wettest), then the next 6 are for coastlines (coldest to warmest),
777         * then rivers (coldest to warmest), then lakes (coldest to warmest).
778         */
779        public static final String[] biomeTable = {
780                //COLDEST //COLDER        //COLD            //HOT                  //HOTTER              //HOTTEST
781                "Ice",    "Ice",          "Grassland",      "Desert",              "Desert",             "Desert",             //DRYEST
782                "Ice",    "Tundra",       "Grassland",      "Grassland",           "Desert",             "Desert",             //DRYER
783                "Ice",    "Tundra",       "Woodland",       "Woodland",            "Savanna",            "Desert",             //DRY
784                "Ice",    "Tundra",       "SeasonalForest", "SeasonalForest",      "Savanna",            "Savanna",            //WET
785                "Ice",    "Tundra",       "BorealForest",   "TemperateRainforest", "TropicalRainforest", "Savanna",            //WETTER
786                "Ice",    "BorealForest", "BorealForest",   "TemperateRainforest", "TropicalRainforest", "TropicalRainforest", //WETTEST
787                "Rocky",  "Rocky",        "Beach",          "Beach",               "Beach",              "Beach",              //COASTS
788                "Ice",    "River",        "River",          "River",               "River",              "River",              //RIVERS
789                "Ice",    "River",        "River",          "River",               "River",              "River",              //LAKES
790        };
791
792        /**
793         * Simple constructor; pretty much does nothing. Make sure to call {@link #makeBiomes(WorldMapGenerator)} before
794         * using fields like {@link #biomeCodeData}.
795         */
796        public SimpleBiomeMapper()
797        {
798            heatCodeData = null;
799            moistureCodeData = null;
800            biomeCodeData = null;
801        }
802
803        /**
804         * Analyzes the last world produced by the given WorldMapGenerator and uses all of its generated information to
805         * assign biome codes for each cell (along with heat and moisture codes). After calling this, biome codes can be
806         * taken from {@link #biomeCodeData} and used as indices into {@link #biomeTable} or a custom biome table.
807         * @param world a WorldMapGenerator that should have generated at least one map; it may be at any zoom
808         */
809        public void makeBiomes(WorldMapGenerator world) {
810            if(world == null || world.width <= 0 || world.height <= 0)
811                return;
812            if(heatCodeData == null || (heatCodeData.length != world.width || heatCodeData[0].length != world.height))
813                heatCodeData = new int[world.width][world.height];
814            if(moistureCodeData == null || (moistureCodeData.length != world.width || moistureCodeData[0].length != world.height))
815                moistureCodeData = new int[world.width][world.height];
816            if(biomeCodeData == null || (biomeCodeData.length != world.width || biomeCodeData[0].length != world.height))
817                biomeCodeData = new int[world.width][world.height];
818            final double i_hot = (world.maxHeat == world.minHeat) ? 1.0 : 1.0 / (world.maxHeat - world.minHeat);
819            for (int x = 0; x < world.width; x++) {
820                for (int y = 0; y < world.height; y++) {
821                    final double hot = (world.heatData[x][y] - world.minHeat) * i_hot, moist = world.moistureData[x][y];
822                    final int heightCode = world.heightCodeData[x][y];
823                    int hc, mc;
824                    boolean isLake = world.generateRivers && world.partialLakeData.contains(x, y) && heightCode >= 4,
825                            isRiver = world.generateRivers && world.partialRiverData.contains(x, y) && heightCode >= 4;
826                    if (moist > wetterValueUpper) {
827                        mc = 5;
828                    } else if (moist > wetValueUpper) {
829                        mc = 4;
830                    } else if (moist > dryValueUpper) {
831                        mc = 3;
832                    } else if (moist > drierValueUpper) {
833                        mc = 2;
834                    } else if (moist > driestValueUpper) {
835                        mc = 1;
836                    } else {
837                        mc = 0;
838                    }
839
840                    if (hot > warmerValueUpper) {
841                        hc = 5;
842                    } else if (hot > warmValueUpper) {
843                        hc = 4;
844                    } else if (hot > coldValueUpper) {
845                        hc = 3;
846                    } else if (hot > colderValueUpper) {
847                        hc = 2;
848                    } else if (hot > coldestValueUpper) {
849                        hc = 1;
850                    } else {
851                        hc = 0;
852                    }
853
854                    heatCodeData[x][y] = hc;
855                    moistureCodeData[x][y] = mc;
856                    biomeCodeData[x][y] = isLake ? hc + 48 : (isRiver ? hc + 42 : ((heightCode == 4) ? hc + 36 : hc + mc * 6));
857                }
858            }
859        }
860    }
