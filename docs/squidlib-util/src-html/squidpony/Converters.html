001package squidpony;
002
003import regexodus.Matcher;
004import squidpony.squidmath.OrderedMap;
005import squidpony.squidmath.OrderedSet;
006
007/**
008 * Ways to produce concrete implementations of StringConvert for various data structures.
009 * Keeping the StringConvert producers separate from the data structures allows us to convert
010 * JDK types as well as to keep the parts that need ObText, and thus RegExodus, separate from
011 * the more general-use data structures.
012 * Created by Tommy Ettinger on 3/9/2017.
013 */
014public class Converters {
015    public static <K> StringConvert<OrderedSet<K>> convertOrderedSet(final StringConvert<K> convert)
016    {
017        return new StringConvert<OrderedSet<K>>() {
018            @Override
019            public String stringify(OrderedSet<K> item) {
020                StringBuilder sb = new StringBuilder(100);
021                K k;
022                for (int i = 0; i < item.size();) {
023                    k = item.getAt(i);
024                    if(item == k)
025                        return "";
026                    ObText.appendQuoted(sb, convert.stringify(k));
027                    if(++i < item.size())
028                        sb.append(' ');
029                }
030                return sb.toString();
031            }
032
033            @Override
034            public OrderedSet<K> restore(String text) {
035                Matcher m = ObText.pattern.matcher(text);
036                OrderedSet<K> d = new OrderedSet<>();
037                while (m.find())
038                {
039                    if(m.isCaptured("s"))
040                    {
041                        d.add(convert.restore(m.group("s")));
042                    }
043                }
044                return d;
045            }
046        };
047    }
048    public static <K, V> StringConvert<OrderedMap<K, V>> convertOrderedMap(final StringConvert<K> convertK, final StringConvert<V> convertV)
049    {
050        return new StringConvert<OrderedMap<K, V>>() {
051            @Override
052            public String stringify(OrderedMap<K, V> item) {
053                StringBuilder sb = new StringBuilder(100);
054                K k;
055                V v;
056                for (int i = 0; i < item.size();) {
057                    k = item.keyAt(i);
058                    if(k == item)
059                        return "";
060                    ObText.appendQuoted(sb, convertK.stringify(k));
061                    sb.append(' ');
062                    v = item.getAt(i);
063                    if(v == item)
064                        return "";
065                    ObText.appendQuoted(sb, convertV.stringify(v));
066                    if(++i < item.size())
067                        sb.append('\n');
068                }
069                return sb.toString();
070            }
071
072            @Override
073            public OrderedMap<K, V> restore(String text) {
074                Matcher m = ObText.pattern.matcher(text);
075                OrderedMap<K, V> d = new OrderedMap<>();
076                String t;
077                while (m.find())
078                {
079                    if(m.isCaptured("s"))
080                    {
081                        t = m.group("s");
082                        if(m.find() && m.isCaptured("s"))
083                        {
084                            d.put(convertK.restore(t), convertV.restore(m.group("s")));
085                        }
086                    }
087                }
088                return d;
089            }
090        };
091    }
092}
