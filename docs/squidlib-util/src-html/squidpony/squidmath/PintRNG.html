001package squidpony.squidmath;
002
003import squidpony.StringKit;
004import squidpony.annotation.Beta;
005
006import java.io.Serializable;
007
008/**
009 * A different kind of RandomnessSource that operates internally on ints, which may have significant advantages on GWT
010 * since that toolchain has to emulate longs (slowly) for almost any math with them. It is, however, expected to be
011 * slower on other platforms, at least platforms with 64-bit JVMs. Running on 32-bit desktop JVMs, 32-bit Dalvik on
012 * Android, or other increasingly-rare environments may benefit from this class, though.
013 * <br>
014 * Quality is not completely certain, but should be excellent in this version since it's based almost directly on PCG-
015 * Random's choices of numerical constants. Visual tests, at least, appear indistinguishable from other PRNGs. Period is
016 * considered very low, at 2^32, but all seeds should be valid, including 0. Generating 64 bits of random data takes a
017 * little less than twice as much time as generating 32 bits, since this can avoid some overhead via inlining.
018 * <br>
019 * The name can be construed as Pint-Size, since this has a small period and uses a smaller amount of space, or as
020 * Permuted Int, since this is based on PermutedRNG, changed to use 32-bit operations on ints.
021 * <br>
022 * Based on work by Melissa E. O'Neill for PCG-Random, though no code is actually shared with PCG-Random.
023 * Created by Tommy Ettinger on 11/15/2016.
024 */
025@Beta
026public class PintRNG implements RandomnessSource, StatefulRandomness, Serializable {
027
028    /** 2 raised to the 53, - 1. */
029    private static final long DOUBLE_MASK = ( 1L << 53 ) - 1;
030    /** 2 raised to the -53. */
031    private static final double NORM_53 = 1. / ( 1L << 53 );
032    /** 2 raised to the 24, -1. */
033    private static final long FLOAT_MASK = ( 1L << 24 ) - 1;
034    /** 2 raised to the -24. */
035    private static final double NORM_24 = 1. / ( 1L << 24 );
036
037    private static final long serialVersionUID = -374415589203474497L;
038
039    public int state; /* The state can be seeded with any value. */
040
041    /** Creates a new generator seeded using Math.random. */
042    public PintRNG() {
043        this((int)((Math.random() - 0.5) * 4.294967296E9));
044    }
045
046    public PintRNG( final long seed ) {
047        setState(seed);
048    }
049
050    public PintRNG(final int a)
051    {
052        state = a;
053    }
054
055    @Override
056    public int next( int bits ) {
057        return nextInt() >>> (32 - bits);
058    }
059
060    /**
061     * Can return any int, positive or negative, of any size permissible in a 32-bit signed integer.
062     * @return any int, all 32 bits are random
063     */
064    public int nextInt() {
065        //return stateA += (stateB ^= (stateB + 0x62E2AC0D + 0x85157AF5 * stateA));
066        //return stateB += ((stateB ^ (stateA += 0x62E2AC0D)) >>> 8) * 0x9E3779B9;
067
068        // increment  = 2891336453;
069        // multiplier = 747796405;
070
071        int p = state;
072        p ^= p >>> (4 + (p >>> 28));
073        state = state * 0x2C9277B5 + 0xAC564B05;
074        return ((p *= 277803737) >>> 22) ^ p;
075    }
076
077    /**
078     * Can return any long, positive or negative, of any size permissible in a 64-bit signed integer.
079     * Internally, generates two random 32-bit values and combines them into one random long.
080     * @return any long, all 64 bits are random
081     */
082    @Override
083    public long nextLong() {
084        int p = state;
085        p ^= p >>> (4 + (p >>> 28));
086        int q = (state = state * 0x2C9277B5 + 0xAC564B05);
087        q ^= q >>> (4 + (q >>> 28));
088        state = state * 0x2C9277B5 + 0xAC564B05;
089        return (((p *= 277803737) >>> 22) ^ p) | ((((q *= 277803737) >>> 22) ^ q) & 0xffffffffL) << 32;
090
091        //return 0x100000000L * nextInt() | nextInt();
092        /*
093        int p = stateA, q = stateB;
094        p ^= p >>> (4 + (p >>> 28));
095        p *= 277803737;
096        stateA = stateA * 0x2C9277B5 + 0xAC564B05;
097        q ^= q >>> (4 + (q >>> 28));
098        q *= 277803737;
099        stateB = stateB * 0x2C9277B5 + (p|1);
100        return 0x100000000L * (p ^ (p >>> 22)) | (q ^ (q >>> 22));
101        */
102    }
103
104    /**
105     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
106     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just needs to
107     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
108     *
109     * @return a copy of this RandomnessSource
110     */
111    @Override
112    public RandomnessSource copy() {
113        return new PintRNG(state);
114    }
115
116    /**
117     * Exclusive on the upper bound.  The lower bound is 0. Uses a technique that RNG calls
118     * {@link squidpony.squidmath.RNG#nextIntHasty(int)}, made known to me by Daniel Lemire.
119     * <br>
120     * Credit goes to Daniel Lemire, http://lemire.me/blog/2016/06/27/a-fast-alternative-to-the-modulo-reduction/
121     * @param bound the upper bound; should be positive
122     * @return a random int less than n and at least equal to 0
123     */
124    public int nextInt( final int bound ) {
125        return (bound <= 0) ? 0 : (int)((bound * (nextInt() & 0x7FFFFFFFL)) >> 31);
126    }
127    /**
128     * Inclusive lower, exclusive upper.
129     * @param lower the lower bound, inclusive, can be positive or negative
130     * @param upper the upper bound, exclusive, should be positive, must be greater than lower
131     * @return a random int at least equal to lower and less than upper
132     */
133    public int nextInt( final int lower, final int upper ) {
134        if ( upper - lower <= 0 ) throw new IllegalArgumentException("Upper bound must be greater than lower bound");
135        return lower + nextInt(upper - lower);
136    }
137
138    /**
139     * Gets a uniform random double in the range [0.0,1.0)
140     * @return a random double at least equal to 0.0 and less than 1.0
141     */
142    public double nextDouble() {
143        return ( nextLong() & DOUBLE_MASK ) * NORM_53;
144    }
145
146    /**
147     * Gets a uniform random double in the range [0.0,outer) given a positive parameter outer. If outer
148     * is negative, it will be the (exclusive) lower bound and 0.0 will be the (inclusive) upper bound.
149     * @param outer the exclusive outer bound, can be negative
150     * @return a random double between 0.0 (inclusive) and outer (exclusive)
151     */
152    public double nextDouble(final double outer) {
153        return nextDouble() * outer;
154    }
155
156    /**
157     * Gets a uniform random float in the range [0.0,1.0)
158     * @return a random float at least equal to 0.0 and less than 1.0
159     */
160    public float nextFloat() {
161        return (float)( ( nextLong() & FLOAT_MASK ) * NORM_24 );
162    }
163
164    /**
165     * Gets a random value, true or false.
166     * Calls nextInt() once.
167     * @return a random true or false value.
168     */
169    public boolean nextBoolean() {
170        return ( nextInt() & 1 ) != 0L;
171    }
172
173    /**
174     * Given a byte array as a parameter, this will fill the array with random bytes (modifying it
175     * in-place). Calls nextInt() {@code Math.ceil(bytes.length / 4.0)} times.
176     * @param bytes a byte array that will have its contents overwritten with random bytes.
177     */
178    public void nextBytes( final byte[] bytes ) {
179        int i = bytes.length, n = 0;
180        while( i != 0 ) {
181            n = Math.min( i, 4 );
182            for ( int bits = nextInt(); n-- != 0; bits >>>= 8 ) bytes[ --i ] = (byte)bits;
183        }
184    }
185
186
187
188    /**
189     * Sets the current state of this generator (an int) by XOR-ing the upper and lower halves of seed into one int.
190     * @param seed the seed to use for this PintRNG, as if it was constructed with this seed.
191     */
192    @Override
193    public void setState( final long seed ) {
194        state = (int)(seed>>>32 ^ seed);
195    }
196    /**
197     * Gets the current state of this generator.
198     * @return the current seed of this PintRNG, changed once per call to nextInt()
199     */
200    @Override
201    public long getState() {
202        return state;
203    }
204
205    @Override
206    public String toString() {
207        return "PintRNG with state 0x" + StringKit.hex(state);
208    }
209
210    @Override
211    public boolean equals(Object o) {
212        if (this == o) return true;
213        if (o == null || getClass() != o.getClass()) return false;
214
215        PintRNG pintRNG = (PintRNG) o;
216
217        return state == pintRNG.state;
218
219    }
220
221    @Override
222    public int hashCode() {
223        return 0x632BE5AB * state;
224    }
225
226    public static int determine(int state)
227    {
228        state ^= state >>> (4 + (state >>> 28));
229        return ((state *= 277803737) >>> 22) ^ state;
230    }
231
232    public static int determineBounded(int state, final int bound)
233    {
234        state ^= state >>> (4 + (state >>> 28));
235        return (int)((bound * ((((state *= 277803737) >>> 22) ^ state) & 0x7FFFFFFFL)) >> 31);
236    }
237}
