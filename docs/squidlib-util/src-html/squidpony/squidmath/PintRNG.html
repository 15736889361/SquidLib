001package squidpony.squidmath;
002
003import squidpony.StringKit;
004import squidpony.annotation.Beta;
005
006import java.io.Serializable;
007
008/**
009 * A different kind of RandomnessSource that operates internally on ints, which may have significant advantages on GWT
010 * since that toolchain has to emulate longs (slowly) for almost any math with them. It is, however, expectd to be
011 * slower on other platforms, at least platforms with 64-bit JVMs. Running on 32-bit desktop JVMs, 32-bit Dalvik on
012 * Android, or other increasingly-rare environments may benefit from this class, though.
013 * <br>
014 * Quality is uncertain at this point, but visual tests at least appear indistinguishable from other PRNGs. Not all
015 * seeds or combinations of seeds may be ideal, and the period of certain bits could easily be lower than the optimal
016 * period of 2^64 32-bit numbers that this might have.
017 * Created by Tommy Ettinger on 11/15/2016.
018 */
019@Beta
020public class PintRNG implements RandomnessSource, StatefulRandomness, Serializable {
021
022    /** 2 raised to the 53, - 1. */
023    private static final long DOUBLE_MASK = ( 1L << 53 ) - 1;
024    /** 2 raised to the -53. */
025    private static final double NORM_53 = 1. / ( 1L << 53 );
026    /** 2 raised to the 24, -1. */
027    private static final long FLOAT_MASK = ( 1L << 24 ) - 1;
028    /** 2 raised to the -24. */
029    private static final double NORM_24 = 1. / ( 1L << 24 );
030
031    private static final long serialVersionUID = -374415589203474497L;
032
033    public int stateA, stateB; /* The state can be seeded with any value. */
034
035    /** Creates a new generator seeded using Math.random. */
036    public PintRNG() {
037        this((int)((Math.random() - 0.5) * 4.294967296E9));
038    }
039
040    public PintRNG( final long seed ) {
041        setState(seed);
042    }
043
044    public PintRNG(final int a, final int b)
045    {
046        stateA = a;
047        stateB = b;
048    }
049
050    @Override
051    public int next( int bits ) {
052        return nextInt() >>> (32 - bits);
053    }
054
055    /**
056     * Can return any int, positive or negative, of any size permissible in a 32-bit signed integer.
057     * @return any int, all 32 bits are random
058     */
059    public int nextInt() {
060        //return stateA += (stateB ^= (stateB + 0x62E2AC0D + 0x85157AF5 * stateA));
061        return stateB += ((stateB ^ (stateA += 0x62E2AC0D)) >>> 8) * 0x9E3779B9;
062    }
063
064    /**
065     * Can return any long, positive or negative, of any size permissible in a 64-bit signed integer.
066     * Internally, generates two random 32-bit values and combines them into one random long.
067     * @return any long, all 64 bits are random
068     */
069    @Override
070    public long nextLong() {
071        return (0x100000000L * nextInt()) | nextInt();
072    }
073
074    /**
075     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
076     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
077     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
078     *
079     * @return a copy of this RandomnessSource
080     */
081    @Override
082    public RandomnessSource copy() {
083        return new PintRNG((0x100000000L * stateA) | stateB);
084    }
085
086    /**
087     * Exclusive on the upper bound.  The lower bound is 0. Uses a technique that RNG calls
088     * {@link squidpony.squidmath.RNG#nextIntHasty(int)}, made known to me by Daniel Lemire.
089     * <br>
090     * Credit goes to Daniel Lemire, http://lemire.me/blog/2016/06/27/a-fast-alternative-to-the-modulo-reduction/
091     * @param bound the upper bound; should be positive
092     * @return a random int less than n and at least equal to 0
093     */
094    public int nextInt( final int bound ) {
095        return (bound <= 0) ? 0 : (int)((bound * (nextInt() & 0x7FFFFFFFL)) >> 31);
096    }
097    /**
098     * Inclusive lower, exclusive upper.
099     * @param lower the lower bound, inclusive, can be positive or negative
100     * @param upper the upper bound, exclusive, should be positive, must be greater than lower
101     * @return a random int at least equal to lower and less than upper
102     */
103    public int nextInt( final int lower, final int upper ) {
104        if ( upper - lower <= 0 ) throw new IllegalArgumentException("Upper bound must be greater than lower bound");
105        return lower + nextInt(upper - lower);
106    }
107
108    /**
109     * Gets a uniform random double in the range [0.0,1.0)
110     * @return a random double at least equal to 0.0 and less than 1.0
111     */
112    public double nextDouble() {
113        return ( nextLong() & DOUBLE_MASK ) * NORM_53;
114    }
115
116    /**
117     * Gets a uniform random double in the range [0.0,outer) given a positive parameter outer. If outer
118     * is negative, it will be the (exclusive) lower bound and 0.0 will be the (inclusive) upper bound.
119     * @param outer the exclusive outer bound, can be negative
120     * @return a random double between 0.0 (inclusive) and outer (exclusive)
121     */
122    public double nextDouble(final double outer) {
123        return nextDouble() * outer;
124    }
125
126    /**
127     * Gets a uniform random float in the range [0.0,1.0)
128     * @return a random float at least equal to 0.0 and less than 1.0
129     */
130    public float nextFloat() {
131        return (float)( ( nextLong() & FLOAT_MASK ) * NORM_24 );
132    }
133
134    /**
135     * Gets a random value, true or false.
136     * Calls nextLong() once.
137     * @return a random true or false value.
138     */
139    public boolean nextBoolean() {
140        return ( nextLong() & 1 ) != 0L;
141    }
142
143    /**
144     * Given a byte array as a parameter, this will fill the array with random bytes (modifying it
145     * in-place). Calls nextLong() {@code Math.ceil(bytes.length / 8.0)} times.
146     * @param bytes a byte array that will have its contents overwritten with random bytes.
147     */
148    public void nextBytes( final byte[] bytes ) {
149        int i = bytes.length, n = 0;
150        while( i != 0 ) {
151            n = Math.min( i, 4 );
152            for ( int bits = nextInt(); n-- != 0; bits >>>= 8 ) bytes[ --i ] = (byte)bits;
153        }
154    }
155
156
157
158    /**
159     * Sets the current state of this generator (two ints) by splitting the given long into upper and lower values.
160     * @param seed the seed to use for this PinRNG, as if it was constructed with this seed.
161     */
162    @Override
163    public void setState( final long seed ) {
164        stateA = (int)(seed>>>32);
165        stateB = (int)seed;
166    }
167    /**
168     * Gets the current state of this generator.
169     * @return the current seed of this PintRNG, changed once per call to nextInt()
170     */
171    @Override
172    public long getState() {
173        return (0x100000000L * stateA) | stateB;
174    }
175
176    @Override
177    public String toString() {
178        return "PintRNG with stateA 0x" + StringKit.hex(stateA) + " and stateB 0x" + StringKit.hex(stateB);
179    }
180
181    @Override
182    public boolean equals(Object o) {
183        if (this == o) return true;
184        if (o == null || getClass() != o.getClass()) return false;
185
186        PintRNG pintRNG = (PintRNG) o;
187
188        return stateA == pintRNG.stateA && stateB == pintRNG.stateB;
189
190    }
191
192    @Override
193    public int hashCode() {
194        return 0x632BE5AB * stateA ^ stateB;
195    }
196}
