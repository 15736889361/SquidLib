001package squidpony.squidmath;
002
003import com.google.gwt.typedarrays.client.Float64ArrayNative;
004import com.google.gwt.typedarrays.client.Float32ArrayNative;
005import com.google.gwt.typedarrays.client.Int32ArrayNative;
006import com.google.gwt.typedarrays.client.Int8ArrayNative;
007import com.google.gwt.typedarrays.shared.Float64Array;
008import com.google.gwt.typedarrays.shared.Float32Array;
009import com.google.gwt.typedarrays.shared.Int32Array;
010import com.google.gwt.typedarrays.shared.Int8Array;
011
012/**
013 * Various numeric functions that are important to performance but need alternate implementations on GWT to obtain it.
014 * Super-sourced on GWT, but most things here are direct calls to JDK methods when on desktop or Android.
015 */
016public class NumberTools {
017
018    private static final Int8Array wba = Int8ArrayNative.create(8);
019    private static final Int32Array wia = Int32ArrayNative.create(wba.buffer(), 0, 2);
020    private static final Float32Array wfa = Float32ArrayNative.create(wba.buffer(), 0, 2);
021    private static final Float64Array wda = Float64ArrayNative.create(wba.buffer(), 0, 1);
022
023    public static long doubleToLongBits(double value) {
024        wda.set(0, value);
025        return ((long)wia.get(1) << 32) | (wia.get(0) & 0xffffffffL);
026    }
027
028    public static long doubleToRawLongBits(double value) {
029        wda.set(0, value);
030        return ((long)wia.get(1) << 32) | (wia.get(0) & 0xffffffffL);
031    }
032
033    public static double longBitsToDouble(long bits) {
034        wia.set(1, (int)(bits >>> 32));
035        wia.set(0, (int)(bits & 0xffffffffL));
036        return wda.get(0);
037    }
038
039    public static int doubleToLowIntBits(double value)
040    {
041        wda.set(0, value);
042        return wia.get(0);
043    }
044    public static int doubleToHighIntBits(double value)
045    {
046        wda.set(0, value);
047        return wia.get(1);
048    }
049
050    public static int floatToIntBits(float value) {
051        wfa.set(0, value);
052        return wia.get(0);
053    }
054    public static int floatToRawIntBits(float value) {
055        wfa.set(0, value);
056        return wia.get(0);
057    }
058
059    public static float intBitsToFloat(int bits) {
060        wia.set(0, bits);
061        return wfa.get(0);
062    }
063
064    static int hashWisp(final float[] data) {
065        if (data == null)
066            return 0;
067        int result = 0x9E3779B9, a = 0x632BE5AB;
068        final int len = data.length;
069        double t;
070        for (int i = 0; i < len; i++) {
071            result += (a ^= 0x85157AF5 * ((int) (-0xD0E8.9D2D311E289Fp-25f * (t = data[i]) + t * -0x1.39b4dce80194cp9f)));
072        }
073        return result * (a | 1) ^ (result >>> 11 | result << 21);
074    }
075
076    static int hashWisp(final double[] data)
077    {
078        if (data == null)
079            return 0;
080        int result = 0x9E3779B9, a = 0x632BE5AB;
081        final int len = data.length;
082        for (int i = 0; i < len; i++) {
083            wda.set(0, data[i]);
084            result += (a ^= 0x85157AF5 * wia.get(0)) + (a ^= 0x85157AF5 * wia.get(1));
085        }
086        return result * (a | 1) ^ (result >>> 11 | result << 21);
087    }
088}
