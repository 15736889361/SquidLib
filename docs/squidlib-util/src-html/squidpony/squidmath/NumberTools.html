001package squidpony.squidmath;
002
003import com.google.gwt.typedarrays.client.Float32ArrayNative;
004import com.google.gwt.typedarrays.client.Int32ArrayNative;
005import com.google.gwt.typedarrays.client.Int8ArrayNative;
006import com.google.gwt.typedarrays.shared.Float32Array;
007import com.google.gwt.typedarrays.shared.Int32Array;
008import com.google.gwt.typedarrays.shared.Int8Array;
009
010/**
011 * Various numeric functions that are important to performance but need alternate implementations on GWT to obtain it.
012 * Super-sourced on GWT, but most things here are direct calls to JDK methods when on desktop or Android.
013 */
014public class NumberTools {
015
016    private static final Int8Array wba = Int8ArrayNative.create(8);
017    private static final Int32Array wia = Int32ArrayNative.create(wba.buffer(), 0, 2);
018    private static final Float32Array wfa = Float32ArrayNative.create(wba.buffer(), 0, 2);
019    private static final Float64Array wda = Float64ArrayNative.create(wba.buffer(), 0, 1);
020
021    public static long doubleToLongBits(double value) {
022        wda.set(0, value);
023        return ((long)wia.get(1) << 32) | (wia.get(0) & 0xffffffffL);
024    }
025
026    public static long doubleToRawLongBits(double value) {
027        wda.set(0, value);
028        return ((long)wia.get(1) << 32) | (wia.get(0) & 0xffffffffL);
029    }
030
031    public static double longBitsToDouble(long bits) {
032        wia.set(1, (int)(bits >>> 32));
033        wia.set(0, (int)(bits & 0xffffffffL));
034        return wda.get(0);
035    }
036
037    public static int doubleToLowIntBits(double value)
038    {
039        wda.set(0, value);
040        return wia.get(0);
041    }
042    public static int doubleToHighIntBits(double value)
043    {
044        wda.set(0, value);
045        return wia.get(1);
046    }
047
048    public static int floatToIntBits(float value) {
049        wfa.set(0, value);
050        return wia.get(0);
051    }
052    public static int floatToRawIntBits(float value) {
053        wfa.set(0, value);
054        return wia.get(0);
055    }
056
057    public static float intBitsToFloat(int bits) {
058        wia.set(0, bits);
059        return wfa.get(0);
060    }
061
062    static int hashWisp(final double[] data)
063    {
064        if (data == null)
065            return 0;
066        int result = 0x9E3779B9, a = 0x632BE5AB;
067        final int len = data.length;
068        for (int i = 0; i < len; i++) {
069            wda.set(0, data[i]);
070            result += (a ^= 0x85157AF5 * wia.get(0)) + (a ^= 0x85157AF5 * wia.get(1));
071        }
072        return result * (a | 1) ^ (result >>> 11 | result << 21);
073    }
074}
