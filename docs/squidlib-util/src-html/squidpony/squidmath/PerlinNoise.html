003/**
004 * This is Ken Perlin's third revision of his noise function. It is sometimes
005 * referred to as "Simplex Noise". Results are bound by (-1, 1) inclusive.
006 *
007 *
008 * It is significantly faster than his earlier versions. This particular version
009 * was originally from Stefan Gustavson. This is much preferred to the earlier
010 * versions of Perlin Noise due to the reasons noted in the articles:
011 * <ul>
012 * <li>http://www.itn.liu.se/~stegu/simplexnoise/simplexnoise.pdf</li>
013 * <li>http://webstaff.itn.liu.se/~stegu/TNM022-2005/perlinnoiselinks/ch02.pdf</li>
014 * </ul>
015 * But, Gustavson's paper is not without its own issues, particularly for 2D noise.
016 * More detail is noted here,
017 * http://stackoverflow.com/questions/18885440/why-does-simplex-noise-seem-to-have-more-artifacts-than-classic-perlin-noise#21568753
018 * and some changes have been made to 2D noise generation to reduce angular artifacts.
019 * Specifically for the 2D gradient table, code based on Gustavson's paper used 12
020 * points, with some duplicates, and not all on the unit circle. In this version,
021 * points are used on the unit circle starting at (1,0) and moving along the circle
022 * in increments of 1.61803398875 radians, that is, the golden ratio phi, getting the
023 * sin and cosine of 15 points after the starting point and storing them as constants.
024 * This definitely doesn't have a noticeable 45 degree angle artifact, though it does
025 * have, to a lesser extent, some other minor artifacts.
026 */

147    /**
148     * 2D simplex noise.
149     * This doesn't use its parameters verbatim; xin and yin are both effectively divided by
150     * ({@link Math#E} * {@link Math#PI}), because without a step like that, any integer parameters would return 0 and
151     * only doubles with a decimal component would produce actual noise. This step allows integers to be passed in a
152     * arguments, and changes the cycle at which 0 is repeated to multiples of (E*PI).
153     *
154     * @param xin x input; works well if between 0.0 and 1.0, but anything is accepted
155     * @param yin y input; works well if between 0.0 and 1.0, but anything is accepted
156     * @return noise from -1.0 to 1.0, inclusive
157     */
158    public static double noise(double xin, double yin) {
159        xin *= epi;
160        yin *= epi;
161        double noise0, noise1, noise2; // from the three corners
162        // Skew the input space to determine which simplex cell we're in
163        double skew = (xin + yin) * F2; // Hairy factor for 2D
164        int i = (int) Math.floor(xin + skew);
165        int j = (int) Math.floor(yin + skew);
166        double t = (i + j) * G2;
167        double X0 = i - t; // Unskew the cell origin back to (x,y) space
168        double Y0 = j - t;
169        double x0 = xin - X0; // The x,y distances from the cell origin
170        double y0 = yin - Y0;
171        // For the 2D case, the simplex shape is an equilateral triangle.
172        // Determine which simplex we are in.
173        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
174        // coords
175        if (x0 > y0) {
176            i1 = 1;
177            j1 = 0;
178        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
179        else {
180            i1 = 0;
181            j1 = 1;
182        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
183        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
184        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
185        // where
186        // c = (3-sqrt(3))/6
187        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y)
188        // unskewed coords
189        double y1 = y0 - j1 + G2;
190        double x2 = x0 - 1.0 + 2.0 * G2; // Offsets for last corner in (x,y)
191        // unskewed coords
192        double y2 = y0 - 1.0 + 2.0 * G2;
193        // Work out the hashed gradient indices of the three simplex corners
194        int ii = i & 255;
195        int jj = j & 255;
196        int gi0 = perm[ii + perm[jj]] & 15;
197        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
198        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
199        // Calculate the contribution from the three corners
200        double t0 = 0.5 - x0 * x0 - y0 * y0;
201        if (t0 < 0) {
202            noise0 = 0.0;
203        } else {
204            t0 *= t0;
205            noise0 = t0 * t0 * dot(phiGrad2[gi0], x0, y0); // (x,y) of grad3 used
206            // for 2D gradient
207        }
208        double t1 = 0.5 - x1 * x1 - y1 * y1;
209        if (t1 < 0) {
210            noise1 = 0.0;
211        } else {
212            t1 *= t1;
213            noise1 = t1 * t1 * dot(phiGrad2[gi1], x1, y1);
214        }
215        double t2 = 0.5 - x2 * x2 - y2 * y2;
216        if (t2 < 0) {
217            noise2 = 0.0;
218        } else {
219            t2 *= t2;
220            noise2 = t2 * t2 * dot(phiGrad2[gi2], x2, y2);
221        }
222        // Add contributions from each corner to get the final noise value.
223        // The result is scaled to return values in the interval [-1,1].
224        return 70.0 * (noise0 + noise1 + noise2);
225    }

227    /**
228     * 3D simplex noise.
229     *
230     * @param xin X input
231     * @param yin Y input
232     * @param zin Z input
233     * @return noise from -1.0 to 1.0, inclusive
234     */
235    public static double noise(double xin, double yin, double zin) {
236        xin *= epi;
237        yin *= epi;
238        zin *= epi;
239        double n0, n1, n2, n3; // Noise contributions from the four corners
240        // Skew the input space to determine which simplex cell we're in
241        double s = (xin + yin + zin) * F3; // Very nice and simple skew
242        // factor for 3D
243        int i = (int) Math.floor(xin + s);
244        int j = (int) Math.floor(yin + s);
245        int k = (int) Math.floor(zin + s);
246        double t = (i + j + k) * G3;
247        double X0 = i - t; // Unskew the cell origin back to (x,y,z) space
248        double Y0 = j - t;
249        double Z0 = k - t;
250        double x0 = xin - X0; // The x,y,z distances from the cell origin
251        double y0 = yin - Y0;
252        double z0 = zin - Z0;
253        // For the 3D case, the simplex shape is a slightly irregular
254        // tetrahedron.
255        // Determine which simplex we are in.
256        int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k)
257        // coords
258        int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k)
259        // coords
260        if (x0 >= y0) {
261            if (y0 >= z0) {
262                i1 = 1;
263                j1 = 0;
264                k1 = 0;
265                i2 = 1;
266                j2 = 1;
267                k2 = 0;
268            } // X Y Z order
269            else if (x0 >= z0) {
270                i1 = 1;
271                j1 = 0;
272                k1 = 0;
273                i2 = 1;
274                j2 = 0;
275                k2 = 1;
276            } // X Z Y order
277            else {
278                i1 = 0;
279                j1 = 0;
280                k1 = 1;
281                i2 = 1;
282                j2 = 0;
283                k2 = 1;
284            } // Z X Y order
285        } else { // x0<y0
286            if (y0 < z0) {
287                i1 = 0;
288                j1 = 0;
289                k1 = 1;
290                i2 = 0;
291                j2 = 1;
292                k2 = 1;
293            } // Z Y X order
294            else if (x0 < z0) {
295                i1 = 0;
296                j1 = 1;
297                k1 = 0;
298                i2 = 0;
299                j2 = 1;
300                k2 = 1;
301            } // Y Z X order
302            else {
303                i1 = 0;
304                j1 = 1;
305                k1 = 0;
306                i2 = 1;
307                j2 = 1;
308                k2 = 0;
309            } // Y X Z order
310        }
311        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in
312        // (x,y,z),
313        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in
314        // (x,y,z), and
315        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in
316        // (x,y,z), where
317        // c = 1/6.
318        double x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z)
319        // coords
320        double y1 = y0 - j1 + G3;
321        double z1 = z0 - k1 + G3;
322        double x2 = x0 - i2 + 2.0 * G3; // Offsets for third corner in
323        // (x,y,z) coords
324        double y2 = y0 - j2 + 2.0 * G3;
325        double z2 = z0 - k2 + 2.0 * G3;
326        double x3 = x0 - 1.0 + 3.0 * G3; // Offsets for last corner in
327        // (x,y,z) coords
328        double y3 = y0 - 1.0 + 3.0 * G3;
329        double z3 = z0 - 1.0 + 3.0 * G3;
330        // Work out the hashed gradient indices of the four simplex corners
331        int ii = i & 255;
332        int jj = j & 255;
333        int kk = k & 255;
334        int gi0 = perm[ii + perm[jj + perm[kk]]] % 12;
335        int gi1 = perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]] % 12;
336        int gi2 = perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]] % 12;
337        int gi3 = perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]] % 12;
338        // Calculate the contribution from the four corners
339        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0;
340        if (t0 < 0) {
341            n0 = 0.0;
342        } else {
343            t0 *= t0;
344            n0 = t0 * t0 * dot(grad3[gi0], x0, y0, z0);
345        }
346        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1;
347        if (t1 < 0) {
348            n1 = 0.0;
349        } else {
350            t1 *= t1;
351            n1 = t1 * t1 * dot(grad3[gi1], x1, y1, z1);
352        }
353        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2;
354        if (t2 < 0) {
355            n2 = 0.0;
356        } else {
357            t2 *= t2;
358            n2 = t2 * t2 * dot(grad3[gi2], x2, y2, z2);
359        }
360        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3;
361        if (t3 < 0) {
362            n3 = 0.0;
363        } else {
364            t3 *= t3;
365            n3 = t3 * t3 * dot(grad3[gi3], x3, y3, z3);
366        }
367        // Add contributions from each corner to get the final noise value.
368        // The result is scaled to stay just inside [-1,1]
369        return 32.0 * (n0 + n1 + n2 + n3);
370    }

372    /**
373     * 4D simplex noise.
374     *
375     * @param x X position
376     * @param y Y position
377     * @param z Z position
378     * @param w Fourth-dimensional position. It is I, the Fourth-Dimensional Ziltoid the Omniscient!
379     * @return noise from -1.0 to 1.0, inclusive
380     */
381    public static double noise(double x, double y, double z, double w) {
382        x *= epi;
383        y *= epi;
384        z *= epi;
385        w *= epi;
386        // The skewing and unskewing factors are hairy again for the 4D case
387        double n0, n1, n2, n3, n4; // Noise contributions from the five
388        // corners
389        // Skew the (x,y,z,w) space to determine which cell of 24 simplices
390        // we're in
391        double s = (x + y + z + w) * F4; // Factor for 4D skewing
392        int i = (int) Math.floor(x + s);
393        int j = (int) Math.floor(y + s);
394        int k = (int) Math.floor(z + s);
395        int l = (int) Math.floor(w + s);
396        double t = (i + j + k + l) * G4; // Factor for 4D unskewing
397        double X0 = i - t; // Unskew the cell origin back to (x,y,z,w) space
398        double Y0 = j - t;
399        double Z0 = k - t;
400        double W0 = l - t;
401        double x0 = x - X0; // The x,y,z,w distances from the cell origin
402        double y0 = y - Y0;
403        double z0 = z - Z0;
404        double w0 = w - W0;
405        // For the 4D case, the simplex is a 4D shape I won't even try to
406        // describe.
407        // To find out which of the 24 possible simplices we're in, we need
408        // to
409        // determine the magnitude ordering of x0, y0, z0 and w0.
410        // The method below is a good way of finding the ordering of x,y,z,w
411        // and
412        // then find the correct traversal order for the simplex we’re in.
413        // First, six pair-wise comparisons are performed between each
414        // possible pair
415        // of the four coordinates, and the results are used to add up binary
416        // bits
417        // for an integer index.
418        int c1 = (x0 > y0) ? 32 : 0;
419        int c2 = (x0 > z0) ? 16 : 0;
420        int c3 = (y0 > z0) ? 8 : 0;
421        int c4 = (x0 > w0) ? 4 : 0;
422        int c5 = (y0 > w0) ? 2 : 0;
423        int c6 = (z0 > w0) ? 1 : 0;
424        int c = c1 + c2 + c3 + c4 + c5 + c6;
425        int i1, j1, k1, l1; // The integer offsets for the second simplex
426        // corner
427        int i2, j2, k2, l2; // The integer offsets for the third simplex
428        // corner
429        int i3, j3, k3, l3; // The integer offsets for the fourth simplex
430        // corner
431        // simplex[c] is a 4-vector with the numbers 0, 1, 2 and 3 in some
432        // order.
433        // Many values of c will never occur, since e.g. x>y>z>w makes x<z,
434        // y<w and x<w
435        // impossible. Only the 24 indices which have non-zero entries make
436        // any sense.
437        // We use a thresholding to set the coordinates in turn from the
438        // largest magnitude.
439        // The number 3 in the "simplex" array is at the position of the
440        // largest coordinate.
441        i1 = simplex[c][0] >= 3 ? 1 : 0;
442        j1 = simplex[c][1] >= 3 ? 1 : 0;
443        k1 = simplex[c][2] >= 3 ? 1 : 0;
444        l1 = simplex[c][3] >= 3 ? 1 : 0;
445        // The number 2 in the "simplex" array is at the second largest
446        // coordinate.
447        i2 = simplex[c][0] >= 2 ? 1 : 0;
448        j2 = simplex[c][1] >= 2 ? 1 : 0;
449        k2 = simplex[c][2] >= 2 ? 1 : 0;
450        l2 = simplex[c][3] >= 2 ? 1 : 0;
451        // The number 1 in the "simplex" array is at the second smallest
452        // coordinate.
453        i3 = simplex[c][0] >= 1 ? 1 : 0;
454        j3 = simplex[c][1] >= 1 ? 1 : 0;
455        k3 = simplex[c][2] >= 1 ? 1 : 0;
456        l3 = simplex[c][3] >= 1 ? 1 : 0;
457        // The fifth corner has all coordinate offsets = 1, so no need to
458        // look that up.
459        double x1 = x0 - i1 + G4; // Offsets for second corner in (x,y,z,w)
460        // coords
461        double y1 = y0 - j1 + G4;
462        double z1 = z0 - k1 + G4;
463        double w1 = w0 - l1 + G4;
464        double x2 = x0 - i2 + 2.0 * G4; // Offsets for third corner in
465        // (x,y,z,w) coords
466        double y2 = y0 - j2 + 2.0 * G4;
467        double z2 = z0 - k2 + 2.0 * G4;
468        double w2 = w0 - l2 + 2.0 * G4;
469        double x3 = x0 - i3 + 3.0 * G4; // Offsets for fourth corner in
470        // (x,y,z,w) coords
471        double y3 = y0 - j3 + 3.0 * G4;
472        double z3 = z0 - k3 + 3.0 * G4;
473        double w3 = w0 - l3 + 3.0 * G4;
474        double x4 = x0 - 1.0 + 4.0 * G4; // Offsets for last corner in
475        // (x,y,z,w) coords
476        double y4 = y0 - 1.0 + 4.0 * G4;
477        double z4 = z0 - 1.0 + 4.0 * G4;
478        double w4 = w0 - 1.0 + 4.0 * G4;
479        // Work out the hashed gradient indices of the five simplex corners
480        int ii = i & 255;
481        int jj = j & 255;
482        int kk = k & 255;
483        int ll = l & 255;
484        int gi0 = perm[ii + perm[jj + perm[kk + perm[ll]]]] & 31;
485        int gi1
486                = perm[ii + i1 + perm[jj + j1 + perm[kk + k1 + perm[ll + l1]]]] & 31;
487        int gi2
488                = perm[ii + i2 + perm[jj + j2 + perm[kk + k2 + perm[ll + l2]]]] & 31;
489        int gi3
490                = perm[ii + i3 + perm[jj + j3 + perm[kk + k3 + perm[ll + l3]]]] & 31;
491        int gi4
492                = perm[ii + 1 + perm[jj + 1 + perm[kk + 1 + perm[ll + 1]]]] & 31;
493        // Calculate the contribution from the five corners
494        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0 - w0 * w0;
495        if (t0 < 0) {
496            n0 = 0.0;
497        } else {
498            t0 *= t0;
499            n0 = t0 * t0 * dot(grad4[gi0], x0, y0, z0, w0);
500        }
501        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1 - w1 * w1;
502        if (t1 < 0) {
503            n1 = 0.0;
504        } else {
505            t1 *= t1;
506            n1 = t1 * t1 * dot(grad4[gi1], x1, y1, z1, w1);
507        }
508        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2 - w2 * w2;
509        if (t2 < 0) {
510            n2 = 0.0;
511        } else {
512            t2 *= t2;
513            n2 = t2 * t2 * dot(grad4[gi2], x2, y2, z2, w2);
514        }
515        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3 - w3 * w3;
516        if (t3 < 0) {
517            n3 = 0.0;
518        } else {
519            t3 *= t3;
520            n3 = t3 * t3 * dot(grad4[gi3], x3, y3, z3, w3);
521        }
522        double t4 = 0.6 - x4 * x4 - y4 * y4 - z4 * z4 - w4 * w4;
523        if (t4 < 0) {
524            n4 = 0.0;
525        } else {
526            t4 *= t4;
527            n4 = t4 * t4 * dot(grad4[gi4], x4, y4, z4, w4);
528        }
529        // Sum up and scale the result to cover the range [-1,1]
530        return 27.0 * (n0 + n1 + n2 + n3 + n4);
531    }
