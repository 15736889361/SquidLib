016        return t > 0 ? (int) t : (int) t - 1;

019    private static double lerp(double s, double v1, double v2) {
020        return v1 + s * (v2 - v1);

023    public static long rawNoise(final int x, final int y, long state)
024    {
025        long inter = 0x632BE59BD9B4E019L;
026        state += ((inter ^= x * 0xD0E89D2D311E289FL) + 0x9E3779B97F4A7C15L)
027                + ((inter ^= y * 0xD0E89D2D311E289FL) + 0x9E3779B97F4A7C15L);
028        return state ^ ((inter ^ state) >>> 16) * 0x9E3779B97F4A7C15L;
029    }

168    /**
169     * 2D simplex noise.
170     * This doesn't use its parameters verbatim; xin and yin are both effectively divided by
171     * ({@link Math#E} * {@link Math#PI}), because without a step like that, any integer parameters would return 0 and
172     * only doubles with a decimal component would produce actual noise. This step allows integers to be passed in a
173     * arguments, and changes the cycle at which 0 is repeated to multiples of (E*PI).
174     *
175     * @param xin x input; works well if between 0.0 and 1.0, but anything is accepted
176     * @param yin y input; works well if between 0.0 and 1.0, but anything is accepted
177     * @return noise from -1.0 to 1.0, inclusive
178     */
179    public static double noise(double xin, double yin) {
180        //xin *= epi;
181        //yin *= epi;
182        double noise0, noise1, noise2; // from the three corners
183        // Skew the input space to determine which simplex cell we're in
184        double skew = (xin + yin) * F2; // Hairy factor for 2D
185        int i = fastFloor(xin + skew);
186        int j = fastFloor(yin + skew);
187        double t = (i + j) * G2;
188        double X0 = i - t; // Unskew the cell origin back to (x,y) space
189        double Y0 = j - t;
190        double x0 = xin - X0; // The x,y distances from the cell origin
191        double y0 = yin - Y0;
192        // For the 2D case, the simplex shape is an equilateral triangle.
193        // Determine which simplex we are in.
194        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
195        // coords
196        if (x0 > y0) {
197            i1 = 1;
198            j1 = 0;
199        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
200        else {
201            i1 = 0;
202            j1 = 1;
203        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
204        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
205        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
206        // where
207        // c = (3-sqrt(3))/6
208        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y)
209        // unskewed coords
210        double y1 = y0 - j1 + G2;
211        double x2 = x0 - 1.0 + 2.0 * G2; // Offsets for last corner in (x,y)
212        // unskewed coords
213        double y2 = y0 - 1.0 + 2.0 * G2;
214        // Work out the hashed gradient indices of the three simplex corners
215        /*
216        int ii = i & 255;
217        int jj = j & 255;
218        int gi0 = perm[ii + perm[jj]] & 15;
219        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
220        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
221        */
222        /*
223        int hash = (int) rawNoise(i + (j * 0x9E3779B9),
224                i + i1 + ((j + j1) * 0x9E3779B9),
225                i + 1 + ((j + 1) * 0x9E3779B9),
226                seed);
227        int gi0 = hash & 15;
228        int gi1 = (hash >>>= 4) & 15;
229        int gi2 = (hash >>> 4) & 15;
230        */
231        int gi0 = (int) determine(i + determine(j)) & 15;
232        int gi1 = (int) determine(i + i1 + determine(j + j1)) & 15;
233        int gi2 = (int) determine(i + 1 + determine(j + 1)) & 15;
234
235        // Calculate the contribution from the three corners
236        double t0 = 0.5 - x0 * x0 - y0 * y0;
237        if (t0 < 0) {
238            noise0 = 0.0;
239        } else {
240            t0 *= t0;
241            noise0 = t0 * t0 * dot(grad2[gi0], x0, y0);
242            // for 2D gradient
243        }
244        double t1 = 0.5 - x1 * x1 - y1 * y1;
245        if (t1 < 0) {
246            noise1 = 0.0;
247        } else {
248            t1 *= t1;
249            noise1 = t1 * t1 * dot(grad2[gi1], x1, y1);
250        }
251        double t2 = 0.5 - x2 * x2 - y2 * y2;
252        if (t2 < 0) {
253            noise2 = 0.0;
254        } else {
255            t2 *= t2;
256            noise2 = t2 * t2 * dot(grad2[gi2], x2, y2);
257        }
258        // Add contributions from each corner to get the final noise value.
259        // The result is scaled to return values in the interval [-1,1].
260        return 70.0 * (noise0 + noise1 + noise2);
261    }

263    /**
264     * 3D simplex noise.
265     *
266     * @param xin X input
267     * @param yin Y input
268     * @param zin Z input
269     * @return noise from -1.0 to 1.0, inclusive
270     */
271    public static double noise(double xin, double yin, double zin) {
272        //xin *= epi;
273        //yin *= epi;
274        //zin *= epi;
275        double n0, n1, n2, n3; // Noise contributions from the four corners
276        // Skew the input space to determine which simplex cell we're in
277        double s = (xin + yin + zin) * F3; // Very nice and simple skew
278        // factor for 3D
279        int i = (int) Math.floor(xin + s);
280        int j = (int) Math.floor(yin + s);
281        int k = (int) Math.floor(zin + s);
282        double t = (i + j + k) * G3;
283        double X0 = i - t; // Unskew the cell origin back to (x,y,z) space
284        double Y0 = j - t;
285        double Z0 = k - t;
286        double x0 = xin - X0; // The x,y,z distances from the cell origin
287        double y0 = yin - Y0;
288        double z0 = zin - Z0;
289        // For the 3D case, the simplex shape is a slightly irregular
290        // tetrahedron.
291        // Determine which simplex we are in.
292        int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k)
293        // coords
294        int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k)
295        // coords
296        if (x0 >= y0) {
297            if (y0 >= z0) {
298                i1 = 1;
299                j1 = 0;
300                k1 = 0;
301                i2 = 1;
302                j2 = 1;
303                k2 = 0;
304            } // X Y Z order
305            else if (x0 >= z0) {
306                i1 = 1;
307                j1 = 0;
308                k1 = 0;
309                i2 = 1;
310                j2 = 0;
311                k2 = 1;
312            } // X Z Y order
313            else {
314                i1 = 0;
315                j1 = 0;
316                k1 = 1;
317                i2 = 1;
318                j2 = 0;
319                k2 = 1;
320            } // Z X Y order
321        } else { // x0<y0
322            if (y0 < z0) {
323                i1 = 0;
324                j1 = 0;
325                k1 = 1;
326                i2 = 0;
327                j2 = 1;
328                k2 = 1;
329            } // Z Y X order
330            else if (x0 < z0) {
331                i1 = 0;
332                j1 = 1;
333                k1 = 0;
334                i2 = 0;
335                j2 = 1;
336                k2 = 1;
337            } // Y Z X order
338            else {
339                i1 = 0;
340                j1 = 1;
341                k1 = 0;
342                i2 = 1;
343                j2 = 1;
344                k2 = 0;
345            } // Y X Z order
346        }
347        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in
348        // (x,y,z),
349        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in
350        // (x,y,z), and
351        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in
352        // (x,y,z), where
353        // c = 1/6.
354        double x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z)
355        // coords
356        double y1 = y0 - j1 + G3;
357        double z1 = z0 - k1 + G3;
358        double x2 = x0 - i2 + 2.0 * G3; // Offsets for third corner in
359        // (x,y,z) coords
360        double y2 = y0 - j2 + 2.0 * G3;
361        double z2 = z0 - k2 + 2.0 * G3;
362        double x3 = x0 - 1.0 + 3.0 * G3; // Offsets for last corner in
363        // (x,y,z) coords
364        double y3 = y0 - 1.0 + 3.0 * G3;
365        double z3 = z0 - 1.0 + 3.0 * G3;
366        // Work out the hashed gradient indices of the four simplex corners
367
368        /*
369        int ii = i & 255;
370        int jj = j & 255;
371        int kk = k & 255;
372
373        int gi0 = perm[ii + perm[jj + perm[kk]]] % 12;
374        int gi1 = perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]] % 12;
375        int gi2 = perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]] % 12;
376        int gi3 = perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]] % 12;
377        */
378
379        int gi0 = determineBounded(i + determine(j + determine(k)), 12);
380        int gi1 = determineBounded(i + i1 + determine(j + j1 + determine(k + k1)), 12);
381        int gi2 = determineBounded(i + i2 + determine(j + j2 + determine(k + k2)), 12);
382        int gi3 = determineBounded(i + 1 + determine(j + 1 + determine(k + 1)), 12);
383
384        /*
385        int hash = (int) rawNoise(i + ((j + k * 0x632BE5AB) * 0x9E3779B9),
386                i + i1 + ((j + j1 + (k + k1) * 0x632BE5AB) * 0x9E3779B9),
387                i + i2 + ((j + j2 + (k + k2) * 0x632BE5AB) * 0x9E3779B9),
388                i + 1 + ((j + 1 + ((k + 1) * 0x632BE5AB)) * 0x9E3779B9),
389                seed);
390        int gi0 = (hash >>>= 4) % 12;
391        int gi1 = (hash >>>= 4) % 12;
392        int gi2 = (hash >>>= 4) % 12;
393        int gi3 = (hash >>> 4) % 12;
394        */
395
396        //int hash = (int) rawNoise(i, j, k, seed);
397        //int gi0 = (hash >>>= 4) % 12, gi1 = (hash >>>= 4) % 12, gi2 = (hash >>>= 4) % 12, gi3 = (hash >>>= 4) % 12;
398        // Calculate the contribution from the four corners
399        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0;
400        if (t0 < 0) {
401            n0 = 0.0;
402        } else {
403            t0 *= t0;
404            n0 = t0 * t0 * dot(grad3[gi0], x0, y0, z0);
405        }
406        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1;
407        if (t1 < 0) {
408            n1 = 0.0;
409        } else {
410            t1 *= t1;
411            n1 = t1 * t1 * dot(grad3[gi1], x1, y1, z1);
412        }
413        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2;
414        if (t2 < 0) {
415            n2 = 0.0;
416        } else {
417            t2 *= t2;
418            n2 = t2 * t2 * dot(grad3[gi2], x2, y2, z2);
419        }
420        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3;
421        if (t3 < 0) {
422            n3 = 0.0;
423        } else {
424            t3 *= t3;
425            n3 = t3 * t3 * dot(grad3[gi3], x3, y3, z3);
426        }
427        // Add contributions from each corner to get the final noise value.
428        // The result is scaled to stay just inside [-1,1]
429        return 32.0 * (n0 + n1 + n2 + n3);
430    }
