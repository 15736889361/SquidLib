008/**
009 * Another experimental noise class. Extends PerlinNoise and should have similar quality, but can be faster and has less
010 * periodic results. This is still considered experimental because the exact output may change in future versions, along
011 * with the scale (potentially) and the parameters it takes. In general, {@link #noise(double, double)} and
012 * {@link #noise(double, double, double)} should have similar appearance to {@link PerlinNoise#noise(double, double)}
013 * and {@link PerlinNoise#noise(double, double, double)}, but are not forced to a zoomed-in scale like PerlinNoise makes
014 * its results, are less likely to repeat sections of noise, and are also somewhat faster (a 20% speedup can be expected
015 * over PerlinNoise using those two methods). Sound good? This also performs minimal (if any) arithmetic on 64-bit long
016 * numbers, instead using double and int, a style that is ideal for GWT (this is true of PerlinNoise too).
017 * <br>
018 * Created by Tommy Ettinger on 12/14/2016.
019 */
020@Beta

148    /**
149     * 2D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
150     * different when passing the same arguments to {@link PerlinNoise#noise(double, double)} and this method. Roughly
151     * 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in chunks because
152     * it uses a pseudo-random function (curiously, {@link PintRNG#determine(int)}, which is optimized for the
153     * limitations of GWT but is rather fast here) instead of a number chosen from a single 256-element array.
154     *
155     * @param xin x input; works well if between 0.0 and 1.0, but anything is accepted
156     * @param yin y input; works well if between 0.0 and 1.0, but anything is accepted
157     * @return noise from -1.0 to 1.0, inclusive
158     */
159    public static double noise(double xin, double yin) {
160        //xin *= epi;
161        //yin *= epi;
162        double noise0, noise1, noise2; // from the three corners
163        // Skew the input space to determine which simplex cell we're in
164        double skew = (xin + yin) * F2; // Hairy factor for 2D
165        int i = fastFloor(xin + skew);
166        int j = fastFloor(yin + skew);
167        double t = (i + j) * G2;
168        double X0 = i - t; // Unskew the cell origin back to (x,y) space
169        double Y0 = j - t;
170        double x0 = xin - X0; // The x,y distances from the cell origin
171        double y0 = yin - Y0;
172        // For the 2D case, the simplex shape is an equilateral triangle.
173        // Determine which simplex we are in.
174        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
175        // coords
176        if (x0 > y0) {
177            i1 = 1;
178            j1 = 0;
179        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
180        else {
181            i1 = 0;
182            j1 = 1;
183        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
184        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
185        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
186        // where
187        // c = (3-sqrt(3))/6
188        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y)
189        // unskewed coords
190        double y1 = y0 - j1 + G2;
191        double x2 = x0 - 1.0 + 2.0 * G2; // Offsets for last corner in (x,y)
192        // unskewed coords
193        double y2 = y0 - 1.0 + 2.0 * G2;
194        // Work out the hashed gradient indices of the three simplex corners
195        /*
196        int ii = i & 255;
197        int jj = j & 255;
198        int gi0 = perm[ii + perm[jj]] & 15;
199        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
200        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
201        */
202        /*
203        int hash = (int) rawNoise(i + (j * 0x9E3779B9),
204                i + i1 + ((j + j1) * 0x9E3779B9),
205                i + 1 + ((j + 1) * 0x9E3779B9),
206                seed);
207        int gi0 = hash & 15;
208        int gi1 = (hash >>>= 4) & 15;
209        int gi2 = (hash >>> 4) & 15;
210        */
211        int gi0 = determine(i + determine(j)) & 15;
212        int gi1 = determine(i + i1 + determine(j + j1)) & 15;
213        int gi2 = determine(i + 1 + determine(j + 1)) & 15;
214
215        // Calculate the contribution from the three corners
216        double t0 = 0.5 - x0 * x0 - y0 * y0;
217        if (t0 < 0) {
218            noise0 = 0.0;
219        } else {
220            t0 *= t0;
221            noise0 = t0 * t0 * dot(grad2[gi0], x0, y0);
222            // for 2D gradient
223        }
224        double t1 = 0.5 - x1 * x1 - y1 * y1;
225        if (t1 < 0) {
226            noise1 = 0.0;
227        } else {
228            t1 *= t1;
229            noise1 = t1 * t1 * dot(grad2[gi1], x1, y1);
230        }
231        double t2 = 0.5 - x2 * x2 - y2 * y2;
232        if (t2 < 0) {
233            noise2 = 0.0;
234        } else {
235            t2 *= t2;
236            noise2 = t2 * t2 * dot(grad2[gi2], x2, y2);
237        }
238        // Add contributions from each corner to get the final noise value.
239        // The result is scaled to return values in the interval [-1,1].
240        return 70.0 * (noise0 + noise1 + noise2);
241    }

245     * 2D simplex noise returning a float; slower than {@link #noise(double, double)}, so avoid this unless you
246     * absolutely need a float. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result
247     * will be different when passing the same arguments to {@link PerlinNoise#noise(double, double)} and this method.
248     *
249     * @param x x input; works well if between 0.0 and 1.0, but anything is accepted
250     * @param y y input; works well if between 0.0 and 1.0, but anything is accepted
251     * @return noise from -1.0 to 1.0, inclusive
252     */
253    public static float noiseAlt(double x, double y) {
254        //xin *= epi;
255        //yin *= epi;
256        float noise0, noise1, noise2; // from the three corners
257        float xin = (float)x, yin = (float)y;
258        // Skew the input space to determine which simplex cell we're in
259        float skew = (xin + yin) * F2f; // Hairy factor for 2D
260        int i = fastFloor(xin + skew);
261        int j = fastFloor(yin + skew);
262        float t = (i + j) * G2f;
263        float X0 = i - t; // Unskew the cell origin back to (x,y) space
264        float Y0 = j - t;
265        float x0 = xin - X0; // The x,y distances from the cell origin
266        float y0 = yin - Y0;
267        // For the 2D case, the simplex shape is an equilateral triangle.
268        // Determine which simplex we are in.
269        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
270        // coords
271        if (x0 > y0) {
272            i1 = 1;
273            j1 = 0;
274        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
275        else {
276            i1 = 0;
277            j1 = 1;
278        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
279        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
280        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
281        // where
282        // c = (3-sqrt(3))/6
283        float x1 = x0 - i1 + G2f; // Offsets for middle corner in (x,y)
284        // unskewed coords
285        float y1 = y0 - j1 + G2f;
286        float x2 = x0 - 1f + 2f * G2f; // Offsets for last corner in (x,y)
287        // unskewed coords
288        float y2 = y0 - 1f + 2f * G2f;
289        // Work out the hashed gradient indices of the three simplex corners
290        /*
291        int ii = i & 255;
292        int jj = j & 255;
293        int gi0 = perm[ii + perm[jj]] & 15;
294        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
295        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
296        */
297        /*
298        int hash = (int) rawNoise(i + (j * 0x9E3779B9),
299                i + i1 + ((j + j1) * 0x9E3779B9),
300                i + 1 + ((j + 1) * 0x9E3779B9),
301                seed);
302        int gi0 = hash & 15;
303        int gi1 = (hash >>>= 4) & 15;
304        int gi2 = (hash >>> 4) & 15;
305        */
306        int gi0 = PintRNG.determine(i + PintRNG.determine(j)) & 15;
307        int gi1 = PintRNG.determine(i + i1 + PintRNG.determine(j + j1)) & 15;
308        int gi2 = PintRNG.determine(i + 1 + PintRNG.determine(j + 1)) & 15;
309
310        // Calculate the contribution from the three corners
311        float t0 = 0.5f - x0 * x0 - y0 * y0;
312        if (t0 < 0) {
313            noise0 = 0f;
314        } else {
315            t0 *= t0;
316            noise0 = t0 * t0 * dotf(grad2f[gi0], x0, y0);
317            // for 2D gradient
318        }
319        float t1 = 0.5f - x1 * x1 - y1 * y1;
320        if (t1 < 0) {
321            noise1 = 0f;
322        } else {
323            t1 *= t1;
324            noise1 = t1 * t1 * dotf(grad2f[gi1], x1, y1);
325        }
326        float t2 = 0.5f - x2 * x2 - y2 * y2;
327        if (t2 < 0) {
328            noise2 = 0f;
329        } else {
330            t2 *= t2;
331            noise2 = t2 * t2 * dotf(grad2f[gi2], x2, y2);
332        }
333        // Add contributions from each corner to get the final noise value.
334        // The result is scaled to return values in the interval [-1,1].
335        return 70f * (noise0 + noise1 + noise2);
336    }

338    /**
339     * 3D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
340     * different when passing the same arguments to {@link PerlinNoise#noise(double, double, double)} and this method.
341     * Roughly 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in chunks
342     * because it uses a pseudo-random function (curiously, {@link PintRNG#determine(int)}, which is optimized for the
343     * limitations of GWT but is rather fast here) instead of a number chosen from a single 256-element array.
344     * @param xin X input
345     * @param yin Y input
346     * @param zin Z input
347     * @return noise from -1.0 to 1.0, inclusive
348     */
349    public static double noise(double xin, double yin, double zin) {
350        //xin *= epi;
351        //yin *= epi;
352        //zin *= epi;
353        double n0, n1, n2, n3; // Noise contributions from the four corners
354        // Skew the input space to determine which simplex cell we're in
355        double s = (xin + yin + zin) * F3; // Very nice and simple skew
356        // factor for 3D
357        int i = fastFloor(xin + s);
358        int j = fastFloor(yin + s);
359        int k = fastFloor(zin + s);
360        double t = (i + j + k) * G3;
361        double X0 = i - t; // Unskew the cell origin back to (x,y,z) space
362        double Y0 = j - t;
363        double Z0 = k - t;
364        double x0 = xin - X0; // The x,y,z distances from the cell origin
365        double y0 = yin - Y0;
366        double z0 = zin - Z0;
367        // For the 3D case, the simplex shape is a slightly irregular
368        // tetrahedron.
369        // Determine which simplex we are in.
370        int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k)
371        // coords
372        int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k)
373        // coords
374        if (x0 >= y0) {
375            if (y0 >= z0) {
376                i1 = 1;
377                j1 = 0;
378                k1 = 0;
379                i2 = 1;
380                j2 = 1;
381                k2 = 0;
382            } // X Y Z order
383            else if (x0 >= z0) {
384                i1 = 1;
385                j1 = 0;
386                k1 = 0;
387                i2 = 1;
388                j2 = 0;
389                k2 = 1;
390            } // X Z Y order
391            else {
392                i1 = 0;
393                j1 = 0;
394                k1 = 1;
395                i2 = 1;
396                j2 = 0;
397                k2 = 1;
398            } // Z X Y order
399        } else { // x0<y0
400            if (y0 < z0) {
401                i1 = 0;
402                j1 = 0;
403                k1 = 1;
404                i2 = 0;
405                j2 = 1;
406                k2 = 1;
407            } // Z Y X order
408            else if (x0 < z0) {
409                i1 = 0;
410                j1 = 1;
411                k1 = 0;
412                i2 = 0;
413                j2 = 1;
414                k2 = 1;
415            } // Y Z X order
416            else {
417                i1 = 0;
418                j1 = 1;
419                k1 = 0;
420                i2 = 1;
421                j2 = 1;
422                k2 = 0;
423            } // Y X Z order
424        }
425        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in
426        // (x,y,z),
427        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in
428        // (x,y,z), and
429        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in
430        // (x,y,z), where
431        // c = 1/6.
432        double x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z)
433        // coords
434        double y1 = y0 - j1 + G3;
435        double z1 = z0 - k1 + G3;
436        double x2 = x0 - i2 + F3; // Offsets for third corner in
437        // (x,y,z) coords
438        double y2 = y0 - j2 + F3;
439        double z2 = z0 - k2 + F3;
440        double x3 = x0 - 0.5; // Offsets for last corner in
441        // (x,y,z) coords
442        double y3 = y0 - 0.5;
443        double z3 = z0 - 0.5;
444        // Work out the hashed gradient indices of the four simplex corners
445
446        /*
447        int ii = i & 255;
448        int jj = j & 255;
449        int kk = k & 255;
450
451        int gi0 = perm[ii + perm[jj + perm[kk]]] % 12;
452        int gi1 = perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]] % 12;
453        int gi2 = perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]] % 12;
454        int gi3 = perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]] % 12;
455        */
456
457        int gi0 = determineBounded(i + determine(j + determine(k)), 12);
458        int gi1 = determineBounded(i + i1 + determine(j + j1 + determine(k + k1)), 12);
459        int gi2 = determineBounded(i + i2 + determine(j + j2 + determine(k + k2)), 12);
460        int gi3 = determineBounded(i + 1 + determine(j + 1 + determine(k + 1)), 12);
461
462        /*
463        int hash = (int) rawNoise(i + ((j + k * 0x632BE5AB) * 0x9E3779B9),
464                i + i1 + ((j + j1 + (k + k1) * 0x632BE5AB) * 0x9E3779B9),
465                i + i2 + ((j + j2 + (k + k2) * 0x632BE5AB) * 0x9E3779B9),
466                i + 1 + ((j + 1 + ((k + 1) * 0x632BE5AB)) * 0x9E3779B9),

510    /**
511     * 3D simplex noise returning a float; slower than {@link #noise(double, double, double)}, so avoid this unless you
512     * absolutely need a float.
513     *
514     * @param x X input
515     * @param y Y input
516     * @param z Z input
517     * @return noise from -1.0 to 1.0, inclusive
518     */
519    public static float noiseAlt(double x, double y, double z) {
520        //xin *= epi;
521        //yin *= epi;
522        //zin *= epi;
523        float xin = (float)x, yin = (float)y, zin = (float)z;
524        float n0, n1, n2, n3; // Noise contributions from the four corners
525        // Skew the input space to determine which simplex cell we're in
526        float s = (xin + yin + zin) * F3f; // Very nice and simple skew
527        // factor for 3D
528        int i = fastFloor(xin + s);
529        int j = fastFloor(yin + s);
530        int k = fastFloor(zin + s);
531        float t = (i + j + k) * G3f;
532        float X0 = i - t; // Unskew the cell origin back to (x,y,z) space
533        float Y0 = j - t;
534        float Z0 = k - t;
535        float x0 = xin - X0; // The x,y,z distances from the cell origin
536        float y0 = yin - Y0;
537        float z0 = zin - Z0;
538        // For the 3D case, the simplex shape is a slightly irregular
539        // tetrahedron.
540        // Determine which simplex we are in.
541        int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k)
542        // coords
543        int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k)
544        // coords
545        if (x0 >= y0) {
546            if (y0 >= z0) {
547                i1 = 1;
548                j1 = 0;
549                k1 = 0;
550                i2 = 1;
551                j2 = 1;
552                k2 = 0;
553            } // X Y Z order
554            else if (x0 >= z0) {
555                i1 = 1;
556                j1 = 0;
557                k1 = 0;
558                i2 = 1;
559                j2 = 0;
560                k2 = 1;
561            } // X Z Y order
562            else {
563                i1 = 0;
564                j1 = 0;
565                k1 = 1;
566                i2 = 1;
567                j2 = 0;
568                k2 = 1;
569            } // Z X Y order
570        } else { // x0<y0
571            if (y0 < z0) {
572                i1 = 0;
573                j1 = 0;
574                k1 = 1;
575                i2 = 0;
576                j2 = 1;
577                k2 = 1;
578            } // Z Y X order
579            else if (x0 < z0) {
580                i1 = 0;
581                j1 = 1;
582                k1 = 0;
583                i2 = 0;
584                j2 = 1;
585                k2 = 1;
586            } // Y Z X order
587            else {
588                i1 = 0;
589                j1 = 1;
590                k1 = 0;
591                i2 = 1;
592                j2 = 1;
593                k2 = 0;
594            } // Y X Z order
595        }
596        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in
597        // (x,y,z),
598        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in
599        // (x,y,z), and
600        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in
601        // (x,y,z), where
602        // c = 1/6.
603        float x1 = x0 - i1 + G3f; // Offsets for second corner in (x,y,z)
604        // coords
605        float y1 = y0 - j1 + G3f;
606        float z1 = z0 - k1 + G3f;
607        float x2 = x0 - i2 + F3f; // Offsets for third corner in
608        // (x,y,z) coords
609        float y2 = y0 - j2 + F3f;
610        float z2 = z0 - k2 + F3f;
611        float x3 = x0 - 0.5f; // Offsets for last corner in
612        // (x,y,z) coords
613        float y3 = y0 - 0.5f;
614        float z3 = z0 - 0.5f;
615        // Work out the hashed gradient indices of the four simplex corners
616
617        /*
618        int ii = i & 255;
619        int jj = j & 255;
620        int kk = k & 255;
621
622        int gi0 = perm[ii + perm[jj + perm[kk]]] % 12;
623        int gi1 = perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]] % 12;
624        int gi2 = perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]] % 12;
625        int gi3 = perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]] % 12;
626        */
627
628        int gi0 = PintRNG.determineBounded(i + PintRNG.determine(j + PintRNG.determine(k)), 12);
629        int gi1 = PintRNG.determineBounded(i + i1 + PintRNG.determine(j + j1 + PintRNG.determine(k + k1)), 12);
630        int gi2 = PintRNG.determineBounded(i + i2 + PintRNG.determine(j + j2 + PintRNG.determine(k + k2)), 12);
631        int gi3 = PintRNG.determineBounded(i + 1 + PintRNG.determine(j + 1 + PintRNG.determine(k + 1)), 12);
632
633        /*
634        int hash = (int) rawNoise(i + ((j + k * 0x632BE5AB) * 0x9E3779B9),
635                i + i1 + ((j + j1 + (k + k1) * 0x632BE5AB) * 0x9E3779B9),
636                i + i2 + ((j + j2 + (k + k2) * 0x632BE5AB) * 0x9E3779B9),
637                i + 1 + ((j + 1 + ((k + 1) * 0x632BE5AB)) * 0x9E3779B9),
638                seed);
639        int gi0 = (hash >>>= 4) % 12;
640        int gi1 = (hash >>>= 4) % 12;
641        int gi2 = (hash >>>= 4) % 12;
642        int gi3 = (hash >>> 4) % 12;
643        */
644
645        //int hash = (int) rawNoise(i, j, k, seed);
646        //int gi0 = (hash >>>= 4) % 12, gi1 = (hash >>>= 4) % 12, gi2 = (hash >>>= 4) % 12, gi3 = (hash >>>= 4) % 12;
647        // Calculate the contribution from the four corners
648        float t0 = 0.6f - x0 * x0 - y0 * y0 - z0 * z0;
649        if (t0 < 0) {
650            n0 = 0f;
651        } else {
652            t0 *= t0;
653            n0 = t0 * t0 * dotf(grad3[gi0], x0, y0, z0);
654        }
655        float t1 = 0.6f - x1 * x1 - y1 * y1 - z1 * z1;
656        if (t1 < 0) {
657            n1 = 0f;
658        } else {
659            t1 *= t1;
660            n1 = t1 * t1 * dotf(grad3[gi1], x1, y1, z1);
661        }
662        float t2 = 0.6f - x2 * x2 - y2 * y2 - z2 * z2;

664            n2 = 0f;

669        float t3 = 0.6f - x3 * x3 - y3 * y3 - z3 * z3;

676        // Add contributions from each corner to get the final noise value.

679    }
