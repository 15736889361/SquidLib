007 * if you find a need for such a thing, with {@link #hash(double...)}.
008 */

121    /**
122     * Hashes the input double array or vararg and produces a double with unpredictable value, between -1.0 inclusive
123     * and 1.0 exclusive.

128        if(data == null) return 0.0;
129        long a = 0x632BE59BD9B4E019L, result = 0x9E3779B97F4A7C94L;

131        int len = data.length;

136    }
