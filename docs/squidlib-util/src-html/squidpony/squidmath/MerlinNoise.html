065    public static int rawNoise2D(final int x, final int y) {
066        return ((x * 0x9E3779B9 ^ 0xC6BC2796) * (y * 0x92B5CC83 + 0xD9B4E019) - (x ^ ~y) * 0x632BE59B * (~x + y) * 0x7F4A7C15) >>> 8 & 0xff;
067    }

070        return (((z * 0xD0E89D2D + 0x311E289F) ^ x * 0x9E3779B9 ^ 0xC6BC2796 ^ (y * 0x92B5CC83 + 0xD9B4E019)) +
071                //(z * 0xD0E89D2D ^ x * 0xC6BC2796 ^ y * 0x92B5CC83 ^
072                (x ^ ~y + z) * 0x632BE59B + (z ^ ~x + y) * 0x7F4A7C15 + (y ^ ~z + x) * 0x9E3779B9) >>> 8 & 0xff; //0x311E289F
073    }

076     * 2D merlin noise.
077     *
078     * @param x x input
079     * @param y y input
080     * @return noise from 0 to 255, inclusive
081     */
082    public static int noise2D(int x, int y) {
083        return (rawNoise2D(x - 1, y - 1) & 43) + (rawNoise2D(x - 1, y + 1) & 43) +
084                (rawNoise2D(x + 1, y - 1) & 43) + (rawNoise2D(x + 1, y + 1) & 43) +
085                (rawNoise2D(x - 1, y) & 53) + (rawNoise2D(x + 1, y) & 53) +
086                (rawNoise2D(x, y - 1) & 53) + (rawNoise2D(x, y - 1) & 53) +
087                (rawNoise2D(x, y) & 127) >>> 1;
088    }

188        if (zoom < 1)
189            return 0;
190        int v = 0, a = x << 1, b = y << 1;
191        for (int s = zoom; s > 0; s--) {
192            v += noise2D((a += 379) / s, (b += 379) / s) + noise2D(a / s, (b + 1) / s) + noise2D((a + 1) / s, b / s);

200        adj *= adj;
201        return (int) (Math.pow(adj, 2.0 - 2.0 * adj) * 255.5);

256    /**
257     * Generates higher-quality continuous-style noise than the other methods, but requires pre-calculating a grid.
258     * Does allow taking a seed because internally it uses a ThunderRNG to quickly generate initial white noise before
259     * processing it into more continuous noise. This generates a lot of random numbers (at least 1 + 14 * height, or
260     * more if width is greater than 64), so ThunderRNG's high speed and presumed higher-than-2-to-the-64 period are
261     * both assets here.
262     * <br>
263     * The 2D int array this produces has ints ranging from 1 to 255, with extreme values very unlikely. Because 0 is
264     * impossible for this to generate, it should be fine to use values from this as denominators in division.
265     *
266     * @param width  the width of the 2D int array to generate
267     * @param height the height of the 2D int array to generate
268     * @param seed   the RNG seed to use when pseudo-randomly generating the initial white noise this then processes
269     * @return a 2D int array where each int should be between 1 and 255, inclusive
270     */
271    public static int[][] preCalcNoise2D(int width, int height, long seed) {
272        ThunderRNG random = new ThunderRNG(seed);
273        int w = (width << 1) + 2, h = (height << 1) + 2;
274        GreasedRegion[] regions = new GreasedRegion[]{
275                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
276                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
277                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
278                new GreasedRegion(random, w, h).retract().expand(3)
279        };
280        random.reseed(random.nextLong());
281        int[][] data = GreasedRegion.bitSum(regions);
282
283        regions = new GreasedRegion[]{
284                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
285                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
286                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
287                new GreasedRegion(random, w, h).retract().expand(3)
288        };
289        int[][] data2 = GreasedRegion.bitSum(regions), data3 = new int[width][height];
290        for (int x = 0; x < w; x++) {
291            for (int y = 0; y < h; y++) {
292                data[x][y] += 128 - data2[x][y];
293            }
294        }
295        for (int x = 0, dx = 1; x < width; x++, dx += 2) {
296            for (int y = 0, dy = 1; y < height; y++, dy += 2) {
297                data3[x][y] = ((data[dx][dy] << 2) + data[dx - 1][dy] + data[dx + 1][dy] + data[dx][dy + 1] + data[dx][dy - 1]) >>> 3;
298            }
299        }
300        return data3;
301        /*
302        int[][] data = GreasedRegion.bitSum(regions);
303        return GreasedRegion.selectiveNegate(data, new GreasedRegion(random, width, height), 0xff);
304        */
305    }
