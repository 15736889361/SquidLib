052/**
053 * Noise functions that try to conceal undesirable patterns, in 2D, 3D, 4D, and 6D, with the last two as options for
054 * generating seamlessly-tiling noise using {@link Noise#seamless2D(double[][], int, int, Noise.Noise4D)} and/or
055 * {@link Noise#seamless3D(double[][][], int, int, Noise.Noise6D)}. All functions can take an int seed that should
056 * significantly change the pattern of noise produced. Incorporates code from Joise; the full library is available at
057 * https://github.com/SudoPlayGames/Joise , and this class adds rather significant optimization in a few methods,
058 * especially 6D noise. Joise is derived from the Accidental Noise Library, available in C++ at
059 * http://accidentalnoise.sourceforge.net/index.html . Both Joise and ANL have many features that SquidLib has not (yet)
060 * incorporated, but now that SquidLib has seamless noise, that's a nice feature that would have needed Joise before.
061 * <br>
062 * The name comes from the Freemasons, who as a secret society, are very good at concealing things. MasonNoise also
063 * sounds a lot like MerlinNoise, which sounds a lot like PerlinNoise and WhirlingNoise. This class is very close in
064 * implementation to WhirlingNoise.
065 */

337    public static double noise(final float x, final float y, final float z, final float w, final int seed) {
338        double n0, n1, n2, n3, n4;
339        final float s = (x + y + z + w) * F4;
340        final int i = fastFloor(x + s), j = fastFloor(y + s), k = fastFloor(z + s), l = fastFloor(w + s);
341        final float[] gradient4DLUT = SeededNoise.gradient4DLUT;
342        final float t = (i + j + k + l) * G4,
343                X0 = i - t,
344                Y0 = j - t,
345                Z0 = k - t,
346                W0 = l - t,
347                x0 = x - X0,
348                y0 = y - Y0,
349                z0 = z - Z0,
350                w0 = w - W0;
351        final int c = (x0 > y0 ? 128 : 0) | (x0 > z0 ? 64 : 0) | (y0 > z0 ? 32 : 0) | (x0 > w0 ? 16 : 0) | (y0 > w0 ? 8 : 0) | (z0 > w0 ? 4 : 0);
352        final int i1 = SIMPLEX[c] >>> 2,
353                j1 = SIMPLEX[c | 1] >>> 2,
354                k1 = SIMPLEX[c | 2] >>> 2,
355                l1 = SIMPLEX[c | 3] >>> 2,
356                i2 = SIMPLEX[c] >>> 1 & 1,
357                j2 = SIMPLEX[c | 1] >>> 1 & 1,
358                k2 = SIMPLEX[c | 2] >>> 1 & 1,
359                l2 = SIMPLEX[c | 3] >>> 1 & 1,
360                i3 = SIMPLEX[c] & 1,
361                j3 = SIMPLEX[c | 1] & 1,
362                k3 = SIMPLEX[c | 2] & 1,
363                l3 = SIMPLEX[c | 3] & 1;
364        final float x1 = x0 - i1 + G4,
365                y1 = y0 - j1 + G4,
366                z1 = z0 - k1 + G4,
367                w1 = w0 - l1 + G4,
368                x2 = x0 - i2 + 2f * G4,
369                y2 = y0 - j2 + 2f * G4,

373                y3 = y0 - j3 + 3f * G4,
374                z3 = z0 - k3 + 3f * G4,
375                w3 = w0 - l3 + 3f * G4,
376                x4 = x0 - 1f + 4f * G4,
377                y4 = y0 - 1f + 4f * G4,
378                z4 = z0 - 1f + 4f * G4,
379                w4 = w0 - 1f + 4f * G4;
380        final int h0 = hash(i, j, k, l, seed) << 2,
381                h1 = hash(i + i1, j + j1, k + k1, l + l1, seed) << 2,
382                h2 = hash(i + i2, j + j2, k + k2, l + l2, seed) << 2,
383                h3 = hash(i + i3, j + j3, k + k3, l + l3, seed) << 2,
384                h4 = hash(i + 1, j + 1, k + 1, l + 1, seed) << 2;
385        float t0 = LIMIT4 - x0 * x0 - y0 * y0 - z0 * z0 - w0 * w0;
386        if (t0 < 0)
387            n0 = 0.0;
388        else {

395        else {

398        }
399        float t2 = LIMIT4 - x2 * x2 - y2 * y2 - z2 * z2 - w2 * w2;
400        if (t2 < 0)
401            n2 = 0.0;
402        else {
403            t2 *= t2;

405        }

407        if (t3 < 0)
408            n3 = 0.0;
409        else {

439    public static double noise(final float x, final float y, final float z, final float w, final float u, final float v, final int seed) {
440
441        final float s = (x + y + z + w + u + v) * F6;
442
443        final int skewX = fastFloor(x + s), skewY = fastFloor(y + s), skewZ = fastFloor(z + s),
444                skewW = fastFloor(w + s), skewU = fastFloor(u + s), skewV = fastFloor(v + s);
445        final int[] intLoc = MasonNoise.intLoc, distOrder = MasonNoise.distOrder;
446        intLoc[0] = skewX;
447        intLoc[1] = skewY;
448        intLoc[2] = skewZ;
449        intLoc[3] = skewW;
450        intLoc[4] = skewU;
451        intLoc[5] = skewV;
452
453        final float unskew = (skewX + skewY + skewZ + skewW + skewU + skewV) * G6;
454        final float[] cellDist = MasonNoise.cellDist, gradient6DLUT = SeededNoise.gradient6DLUT;
455        cellDist[0] = x - skewX + unskew;
456        cellDist[1] = y - skewY + unskew;
457        cellDist[2] = z - skewZ + unskew;
458        cellDist[3] = w - skewW + unskew;
459        cellDist[4] = u - skewU + unskew;
460        cellDist[5] = v - skewV + unskew;
