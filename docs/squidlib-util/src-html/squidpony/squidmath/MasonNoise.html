054/**
055 * Noise functions that try to conceal undesirable patterns, in 2D, 3D, 4D, and 6D, with the last two as options for
056 * generating seamlessly-tiling noise using {@link Noise#seamless2D(double[][], int, int, Noise.Noise4D)} and/or
057 * {@link Noise#seamless3D(double[][][], int, int, Noise.Noise6D)}. All functions can take an int seed that should
058 * significantly change the pattern of noise produced. Incorporates code from Joise; the full library is available at
059 * https://github.com/SudoPlayGames/Joise , and this class adds rather significant optimization in a few methods,
060 * especially 6D noise. Joise is derived from the Accidental Noise Library, available in C++ at
061 * http://accidentalnoise.sourceforge.net/index.html . Both Joise and ANL have many features that SquidLib has not (yet)
062 * incorporated, but now that SquidLib has seamless noise, that's a nice feature that would have needed Joise before.
063 * <br>
064 * The name comes from the Freemasons, who as a secret society, are very good at concealing things. MasonNoise also
065 * sounds a lot like MerlinNoise, which sounds a lot like PerlinNoise and WhirlingNoise. This class is very close in
066 * implementation to WhirlingNoise.
067 */

360    public static double noise(final double xPos, final double yPos, int seed) {
361        final float x = (float)xPos, y = (float)yPos;
362        final float s = (x + y) * F2;
363        final int i = fastFloor(x + s),
364                j = fastFloor(y + s);
365        final float t = (i + j) * G2,
366                X0 = i - t,
367                Y0 = j - t,
368                x0 = x - X0,
369                y0 = y - Y0;
370        int i1, j1;
371        if (x0 > y0) {
372            i1 = 1;
373            j1 = 0;
374        } else {
375            i1 = 0;
376            j1 = 1;
377        }
378        final float
379                x1 = x0 - i1 + G2,
380                y1 = y0 - j1 + G2,
381                x2 = x0 - 1f + 2f * G2,
382                y2 = y0 - 1f + 2f * G2;
383        final int h0 = mix(i, j, seed),
384                h1 = mix(i + i1, j + j1, seed),
385                h2 = mix(i + 1, j + 1, seed);
386        float n0, n1, n2;
387        float t0 = 0.5f - x0 * x0 - y0 * y0;
388        if (t0 < 0)
389            n0 = 0;
390        else {
391            t0 *= t0;
392            n0 = t0 * t0 * ((x0 * randomSignedFloat(h0)) + (y0 * randomSignedFloat(h0 + 0x9E3779B9)));
393        }
394        float t1 = 0.5f - x1 * x1 - y1 * y1;
395        if (t1 < 0)
396            n1 = 0;
397        else {
398            t1 *= t1;
399            n1 = t1 * t1 * (x1 * randomSignedFloat(h1) + y1 * randomSignedFloat(h1 + 0x9E3779B9));
400        }
401        float t2 = 0.5f - x2 * x2 - y2 * y2;
402        if (t2 < 0)
403            n2 = 0;
404        else {
405            t2 *= t2;
406            n2 = t2 * t2 * (x2 * randomSignedFloat(h2) + y2 * randomSignedFloat(h2 + 0x9E3779B9));
407        }
408        return (50f * (n0 + n1 + n2)) * 1.42188695 + 0.001054489;
409    }

411    public static double noise(final double xPos, final double yPos, final double zPos, final int seed) {
412        final float x = (float)xPos, y = (float)yPos, z = (float)zPos;
413        double n0, n1, n2, n3;
414
415        final float s = (x + y + z) * F3;
416        final int i = fastFloor(x + s),
417                j = fastFloor(y + s),
418                k = fastFloor(z + s);
419
420        final float t = (i + j + k) * G3;
421        final float X0 = i - t, Y0 = j - t, Z0 = k - t,
422                x0 = x - X0, y0 = y - Y0, z0 = z - Z0;
423
424        int i1, j1, k1;
425        int i2, j2, k2;
426
427        if (x0 >= y0) {
428            if (y0 >= z0) {
429                i1 = 1;
430                j1 = 0;
431                k1 = 0;
432                i2 = 1;
433                j2 = 1;
434                k2 = 0;
435            } else if (x0 >= z0) {
436                i1 = 1;
437                j1 = 0;
438                k1 = 0;
439                i2 = 1;
440                j2 = 0;
441                k2 = 1;
442            } else {
443                i1 = 0;
444                j1 = 0;
445                k1 = 1;
446                i2 = 1;
447                j2 = 0;
448                k2 = 1;
449            }
450        } else {
451            if (y0 < z0) {
452                i1 = 0;
453                j1 = 0;
454                k1 = 1;
455                i2 = 0;
456                j2 = 1;
457                k2 = 1;
458            } else if (x0 < z0) {
459                i1 = 0;
460                j1 = 1;
461                k1 = 0;
462                i2 = 0;
463                j2 = 1;
464                k2 = 1;
465            } else {
466                i1 = 0;
467                j1 = 1;
468                k1 = 0;
469                i2 = 1;
470                j2 = 1;
471                k2 = 0;
472            }
473        }
474
475        float x1 = x0 - i1 + G3,
476              y1 = y0 - j1 + G3,
477              z1 = z0 - k1 + G3,
478              x2 = x0 - i2 + 2f * G3,
479              y2 = y0 - j2 + 2f * G3,
480              z2 = z0 - k2 + 2f * G3,
481              x3 = x0 - 1f + 3f * G3,
482              y3 = y0 - 1f + 3f * G3,
483              z3 = z0 - 1f + 3f * G3;
484
485        final int h0 = mix(i, j, k, seed),
486                h1 = mix(i + i1, j + j1, k + k1, seed),
487                h2 = mix(i + i2, j + j2, k + k2, seed),
488                h3 = mix(i + 1, j + 1, k + 1, seed);
489
490        float t0 = 0.6f - x0 * x0 - y0 * y0 - z0 * z0;
491        if (t0 < 0f)
492            n0 = 0f;
493        else {
494            t0 *= t0;
495            n0 = t0 * t0 * (x0 * randomSignedFloat(h0) + y0 * randomSignedFloat(h0 + 0x9E3779B9) + z0 * randomSignedFloat(h0 + 0x3C6EF372));
496        }
497
498        float t1 = 0.6f - x1 * x1 - y1 * y1 - z1 * z1;
499        if (t1 < 0f)
500            n1 = 0f;
501        else {
502            t1 *= t1;
503            n1 = t1 * t1 * (x1 * randomSignedFloat(h1) + y1 * randomSignedFloat(h1 + 0x9E3779B9) + z1 * randomSignedFloat(h1 + 0x3C6EF372));
504        }
505
506        float t2 = 0.6f - x2 * x2 - y2 * y2 - z2 * z2;
507        if (t2 < 0)
508            n2 = 0f;
509        else {
510            t2 *= t2;
511            n2 = t2 * t2 * (x2 * randomSignedFloat(h2) + y2 * randomSignedFloat(h2 + 0x9E3779B9) + z2 * randomSignedFloat(h2 + 0x3C6EF372));
512        }
513
514        float t3 = 0.6f - x3 * x3 - y3 * y3 - z3 * z3;
