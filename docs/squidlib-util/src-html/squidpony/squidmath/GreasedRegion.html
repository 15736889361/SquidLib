344    public GreasedRegion insertRectangle(int startX, int startY, int rectangleWidth, int rectangleHeight)
345    {
346        if(rectangleWidth < 1 || rectangleHeight < 1 || ySections <= 0)
347            return this;
348        if(startX < 0)
349            startX = 0;
350        else if(startX >= width)
351            startX = width - 1;
352        if(startY < 0)
353            startY = 0;
354        else if(startY >= height)
355            startY = height - 1;
356        int endX = Math.min(width, startX + rectangleWidth) - 1,
357                endY = Math.min(height, startY + rectangleHeight) - 1,
358                startSection = (startY >> 6), endSection = (endY >> 6);
359        if(startSection < endSection)
360        {
361            long startMask = (-1L << (startY & 63)),
362                    endMask = (-1L >>> (~endY & 63));
363            for (int a = startX * ySections + startSection; a <= endX * ySections; a += ySections) {
364                data[a] |= startMask;
365            }
366            if(endSection - startSection > 1)
367            {
368                for (int b = 1; b < endSection - startSection; b++) {
369                    for (int a = startX * ySections + startSection + b; a < endX * ySections + ySections; a += ySections) {
370                        data[a] = -1;
371                    }
372                }
373            }
374            for (int a = startX * ySections + endSection; a <= endX * ySections + ySections; a += ySections) {
375                data[a] |= endMask;
376            }
377        }
378        else
379        {
380            long mask = (-1L << (startY & 63)) & (-1L >>> (~endY & 63));
381            for (int a = startX * ySections + startSection; a <= endX * ySections + startSection; a += ySections) {
382                data[a] |= mask;
383            }
384        }
385
386        if(ySections > 0 && yEndMask != -1) {
387            for (int a = ySections - 1; a < data.length; a += ySections) {
388                data[a] &= yEndMask;
389            }
390        }
391        return this;
392    }
