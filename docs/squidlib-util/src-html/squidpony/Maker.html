011/**
012 * Utility methods for more easily constructing data structures, particularly those in Java's standard library.

018    /**
019     * Stores any information relating to non-fatal issues, such as caught and handled Exceptions that still change the
020     * behavior of methods. Typically shouldn't be cleared while debugging, since it could be useful later on, and
021     * hopefully won't need to be written to in a release build.
022     */
023    public static final StringBuilder issueLog = new StringBuilder(1024);
024    /**
025     * Makes a LinkedHashMap (LHM) with key and value types inferred from the types of k0 and v0, and considers all
026     * parameters key-value pairs, casting the Objects at positions 0, 2, 4... etc. to K and the objects at positions
027     * 1, 3, 5... etc. to V. If rest has an odd-number length, then it discards the last item. If any pair of items in
028     * rest cannot be cast to the correct type of K or V, then this inserts nothing for that pair and logs information
029     * on the problematic pair to the static Maker.issueLog field.
030     * @param k0 the first key; used to infer the types of other keys if generic parameters aren't specified.
031     * @param v0 the first value; used to infer the types of other values if generic parameters aren't specified.
032     * @param rest an array or vararg of keys and values in pairs; should contain alternating K, V, K, V... elements
033     * @param <K> the type of keys in the returned LinkedHashMap; if not specified, will be inferred from k0
034     * @param <V> the type of values in the returned LinkedHashMap; if not specified, will be inferred from v0
035     * @return a freshly-made LinkedHashMap with K keys and V values, using k0, v0, and the contents of rest to fill it
036     */
037    @SuppressWarnings("unchecked")
038    public static <K, V> LinkedHashMap<K, V> makeLHM(K k0, V v0, Object... rest)
039    {
040        if(rest == null)
041            return makeLHM(k0, v0);
042        LinkedHashMap<K, V> lhm = new LinkedHashMap<>(1 + (rest.length / 2));
043        lhm.put(k0, v0);
044
045        for (int i = 0; i < rest.length - 1; i+=2) {
046            try {
047                lhm.put((K) rest[i], (V) rest[i + 1]);
048            }catch (ClassCastException cce) {
049                issueLog.append("makeLHM call had a casting problem with pair at rest[");
050                issueLog.append(i);
051                issueLog.append("] and/or rest[");
052                issueLog.append(i + 1);
053                issueLog.append("], with contents: ");
054                issueLog.append(rest[i]);
055                issueLog.append(", ");
056                issueLog.append(rest[i+1]);
057                issueLog.append(".\n\nException messages:\n");
058                issueLog.append(cce);
059                String msg = cce.getMessage();
060                if (msg != null) {
061                    issueLog.append('\n');
062                    issueLog.append(msg);
063                }
064                issueLog.append('\n');
065            }
066        }
067        return lhm;
068    }

070    /**
071     * Makes an empty LinkedHashMap (LHM); needs key and value types to be specified in order to work. For an empty
072     * LinkedHashMap with String keys and Coord values, you could use {@code Maker.<String, Coord>makeLHM();}. Using
073     * the new keyword is probably just as easy in this case; this method is provided for completeness relative to
074     * makeLHM() with 2 or more parameters.
075     * @param <K> the type of keys in the returned LinkedHashMap; cannot be inferred and must be specified
076     * @param <V> the type of values in the returned LinkedHashMap; cannot be inferred and must be specified
077     * @return an empty LinkedHashMap with the given key and value types.
078     */
079    public static <K, V> LinkedHashMap<K, V> makeLHM()
080    {
081        return new LinkedHashMap<>();
082    }

084    /**
085     * Makes an ArrayList of T given an array or vararg of T elements.
086     * @param elements an array or vararg of T
087     * @param <T> just about any non-primitive type
088     * @return a newly-allocated ArrayList containing all of elements, in order
089     */
090    public static <T> ArrayList<T> makeList(T... elements) {
091        if(elements == null) return null;
092        ArrayList<T> list = new ArrayList<>(elements.length);
093        Collections.addAll(list, elements);
094        return list;
095    }
096
097    /**
098     * Makes a LinkedHashSet (LHS) of T given an array or vararg of T elements. Duplicate items in elements will have
099     * all but one item discarded, using the later item in elements.
100     * @param elements an array or vararg of T
101     * @param <T> just about any non-primitive type
102     * @return a newly-allocated LinkedHashSet containing all of the non-duplicate items in elements, in order
103     */
104    public static <T> LinkedHashSet<T> makeLHS(T... elements) {
105        if(elements == null) return null;
106        LinkedHashSet<T> set = new LinkedHashSet<>(elements.length);
107        Collections.addAll(set, elements);
108        return set;
109    }

111    /**
112     * Makes an OrderedMap (OM) with key and value types inferred from the types of k0 and v0, and considers all
113     * parameters key-value pairs, casting the Objects at positions 0, 2, 4... etc. to K and the objects at positions
114     * 1, 3, 5... etc. to V. If rest has an odd-number length, then it discards the last item. If any pair of items in
115     * rest cannot be cast to the correct type of K or V, then this inserts nothing for that pair and logs information
116     * on the problematic pair to the static Maker.issueLog field.
117     * @param k0 the first key; used to infer the types of other keys if generic parameters aren't specified.
118     * @param v0 the first value; used to infer the types of other values if generic parameters aren't specified.
119     * @param rest an array or vararg of keys and values in pairs; should contain alternating K, V, K, V... elements
120     * @param <K> the type of keys in the returned OrderedMap; if not specified, will be inferred from k0
121     * @param <V> the type of values in the returned OrderedMap; if not specified, will be inferred from v0
122     * @return a freshly-made OrderedMap with K keys and V values, using k0, v0, and the contents of rest to fill it
123     */
124    @SuppressWarnings("unchecked")
125    public static <K, V> OrderedMap<K, V> makeOM(K k0, V v0, Object... rest)
126    {
127        return makeOM(0.625f, k0, v0, rest);
128    }
129    /**
130     * Makes an OrderedMap (OM) with the given load factor (which should be between 0.1 and 0.9), key and value types
131     * inferred from the types of k0 and v0, and considers all remaining parameters key-value pairs, casting the Objects
132     * at positions 0, 2, 4... etc. to K and the objects at positions 1, 3, 5... etc. to V. If rest has an odd-number
133     * length, then it discards the last item. If any pair of items in rest cannot be cast to the correct type of K or
134     * V, then this inserts nothing for that pair and logs information on the problematic pair to the static
135     * Maker.issueLog field.
136     * @param factor the load factor; should be between 0.1 and 0.9, and 0.75f is a safe choice
137     * @param k0 the first key; used to infer the types of other keys if generic parameters aren't specified.
138     * @param v0 the first value; used to infer the types of other values if generic parameters aren't specified.
139     * @param rest an array or vararg of keys and values in pairs; should contain alternating K, V, K, V... elements
140     * @param <K> the type of keys in the returned OrderedMap; if not specified, will be inferred from k0
141     * @param <V> the type of values in the returned OrderedMap; if not specified, will be inferred from v0
142     * @return a freshly-made OrderedMap with K keys and V values, using k0, v0, and the contents of rest to fill it
143     */
144    @SuppressWarnings("unchecked")
145    public static <K, V> OrderedMap<K, V> makeOM(float factor, K k0, V v0, Object... rest)
146    {
147        if(rest == null)
148            rest = new Object[0];
149        OrderedMap<K, V> om = new OrderedMap<>(1 + (rest.length / 2), factor);
150        om.put(k0, v0);
151
152        for (int i = 0; i < rest.length - 1; i+=2) {
153            try {
154                om.put((K) rest[i], (V) rest[i + 1]);
155            }catch (ClassCastException cce) {
156                issueLog.append("makeOM call had a casting problem with pair at rest[");
157                issueLog.append(i);
158                issueLog.append("] and/or rest[");
159                issueLog.append(i + 1);
160                issueLog.append("], with contents: ");
161                issueLog.append(rest[i]);
162                issueLog.append(", ");
163                issueLog.append(rest[i+1]);
164                issueLog.append(".\n\nException messages:\n");
165                issueLog.append(cce);
166                String msg = cce.getMessage();
167                if (msg != null) {
168                    issueLog.append('\n');
169                    issueLog.append(msg);
170                }
171                issueLog.append('\n');
172            }
173        }
174        return om;
175    }

188        return new OrderedMap<>();

191    /**
192     * Makes a OrderedSet (OS) of T given an array or vararg of T elements. Duplicate items in elements will have
193     * all but one item discarded, using the later item in elements.
194     * @param elements an array or vararg of T
195     * @param <T> just about any non-primitive type
196     * @return a newly-allocated OrderedSet containing all of the non-duplicate items in elements, in order
197     */
198    public static <T> OrderedSet<T> makeOS(T... elements) {
199        if(elements == null) return null;
200        return new OrderedSet<T>(elements);
