2538     * Produces a FakeLanguageGen by mixing this FakeLanguageGen with one or more other FakeLanguageGen objects. Takes
2539     * a weight for this, another FakeLanguageGen, a weight for that FakeLanguageGen, then a possibly-empty group of
2540     * FakeLanguageGen parameters and the weights for those parameters. If other1 is null or if pairs has been given a
2541     * value of null instead of the normal (possibly empty) array of Objects, then this simply returns a copy of this
2542     * FakeLanguageGen. Otherwise, it will at least mix this language with other1 using the given weights for each.
2543     * If pairs is not empty, it has special requirements for what types it allows and in what order, but does no type
2544     * checking. Specifically, pairs requires the first Object to be a FakeLanguageGen, the next to be a number of some
2545     * kind that will be the weight for the previous FakeLanguageGen(this method can handle non-Double weights, and
2546     * converts them to Double if needed), and every two parameters after that to follow the same order and pattern
2547     * (FakeLanguageGen, then number, then FakeLanguageGen, then number...). Weights are absolute, and don't depend on
2548     * earlier weights, which is the case when chaining the {@link #mix(FakeLanguageGen, double)} method. This makes
2549     * reasoning about the ideal weights for multiple mixed languages easier; to mix 3 languages equally you can use
2550     * 3 equal weights with this, whereas with mix chaining you would need to mix the first two with 0.5 and the third
2551     * with 0.33 .
2552     * <br>
2553     * Unlike the static method {@link #mixAll(Object...)}, this _is_ intended for external use, in part because the
2554     * technique for mixing languages by weight is so much more intuitive, but also because this assigns valid data for
2555     * serializing and deserializing this FakeLanguageGen that allows it to use significantly less space (less than 1/72
2556     * the bytes used in one not-quite-simple test).

2558     * @param other1 another FakeLanguageGen to mix in; if null, this method will abort and return {@link #copy()}
2559     * @param weight1 the weight to assign other1 in the mix
2560     * @param pairs may be empty, not null; otherwise must alternate between FakeLanguageGen and number (weight) elements
2561     * @return a FakeLanguageGen produced by mixing this with any FakeLanguageGen arguments by the given weights
2562     */

2564    {
2565        if(other1 == null || pairs == null)

2568        FakeLanguageGen mixer = removeModifiers();
2569        FakeLanguageGen[] languages = new FakeLanguageGen[2 + (pairs.length >>> 1)];
2570        double[] weights = new double[languages.length];
2571        String[] summaries = new String[languages.length];
2572        boolean summarize = true;
2573        double total = 0.0, current, weight;
2574        languages[0] = mixer;
2575        total += (weights[0] = myWeight);
2576        if((summaries[0] = mixer.summary) == null) summarize = false;
2577        mods.addAll(other1.modifiers);
2578        languages[1] = other1.removeModifiers();
2579        total += (weights[1] = weight1);
2580        if(summarize && (summaries[1] = languages[1].summary) == null) summarize = false;
2581        for (int i = 1, p = 2; i < pairs.length; i+=2, p++) {
2582            if(pairs[i] == null || pairs[i-1] == null)
2583                continue;
2584            languages[p] = ((FakeLanguageGen) (pairs[i-1])).removeModifiers();
2585            total += (weights[p] = readDouble(pairs[i]));
2586            if(summarize && (summaries[p] = languages[p].summary) == null) summarize = false;
2587        }
2588        if(total == 0)
2589            return copy();

2591        for (int i = 1; i < languages.length; i++) {
2592            if((weight = weights[i]) > 0)
2593                mixer = mixer.mix(languages[i], (weight / total) / (current += weight / total));
2594        }
2595        if(summarize) {

2597            String c;
2598            int idx;
2599            for (int i = 0; i < summaries.length; i++) {
2600                c = summaries[i];
2601                idx = c.indexOf('@');
2602                if(idx >= 0)
2603                {
2604                    brief.append(c.substring(0, idx + 1)).append(weights[i]);

2607                }
2608            }

2610                brief.append('\016').append(mods.get(i).serializeToString());
2611            }
