029    /**
030     * Stupidly simple convenience method that produces a range from 0 to end, not including end, as an int array.
031     *
032     * @param end the exclusive upper bound on the range
033     * @return the range of ints as an int array
034     */
035    public static int[] range(int end) {
036        if (end <= 0)
037            return new int[0];
038        int[] r = new int[end];
039        for (int i = 0; i < end; i++) {
040            r[i] = i;
041        }
042        return r;
043    }

045    /**
046     * Stupidly simple convenience method that produces a range from start to end, not including end, as an int array.
047     *
048     * @param start the inclusive lower bound on the range
049     * @param end   the exclusive upper bound on the range
050     * @return the range of ints as an int array
051     */
052    public static int[] range(int start, int end) {
053        if (end - start <= 0)
054            return new int[0];
055        int[] r = new int[end - start];
056        for (int i = 0, n = start; n < end; i++, n++) {
057            r[i] = n;
058        }
059        return r;
060    }

062    /**
063     * Stupidly simple convenience method that produces a char range from start to end, including end, as a char array.
064     *
065     * @param start the inclusive lower bound on the range, such as 'a'
066     * @param end   the inclusive upper bound on the range, such as 'z'
067     * @return the range of chars as a char array
068     */
069    public static char[] charSpan(char start, char end) {
070        if (end - start <= 0)
071            return empty;
072        if (end == 0xffff) {
073
074            char[] r = new char[0x10000 - start];
075            for (char i = 0, n = start; n < end; i++, n++) {
076                r[i] = n;
077            }
078            r[0xffff - start] = 0xffff;
079            return r;
080        }
081        char[] r = new char[end - start + 1];
082        for (char i = 0, n = start; n <= end; i++, n++) {
083            r[i] = n;
084        }
085        return r;
086    }

088    /**
089     * Stupidly simple convenience method that produces a char array containing only letters that can be reasonably
090     * displayed (with SquidLib's default text display assets, at least). The letters are copied from a single source
091     * of 256 chars; if you need more chars or you don't need pure letters, you can use {@link #charSpan(char, char)}.
092     *
093     * @param charCount the number of letters to return in an array; the maximum this will produce is 256
094     * @return the range of letters as a char array
095     */
096    public static char[] letterSpan(int charCount) {
097        if (charCount <= 0)
098            return empty;
099        char[] r = new char[Math.min(charCount, 256)];
100        System.arraycopy(letters, 0, r, 0, r.length);
101        return r;
102    }

105     * Gets a copy of the 2D char array, source, that has the same data but shares no references with source.
106     *
107     * @param source a 2D char array
108     * @return a copy of source, or null if source is null
109     */
110    public static char[][] copy(char[][] source) {
111        if (source == null)
112            return null;
113        if (source.length < 1)
114            return new char[0][0];
115        char[][] target = new char[source.length][source[0].length];
116        for (int i = 0; i < source.length && i < target.length; i++) {
117            System.arraycopy(source[i], 0, target[i], 0, source[i].length);
118        }
119        return target;
120    }

176    /**
177     * Inserts as much of source into target at the given x,y position as target can hold or source can supply.
178     * Modifies target in-place and also returns target for chaining.
179     * Used primarily to place a smaller array into a different position in a larger array, often freshly allocated.
180     *
181     * @param source a 2D char array that will be copied and inserted into target
182     * @param target a 2D char array that will be modified by receiving as much of source as it can hold
183     * @param x      the x position in target to receive the items from the first cell in source
184     * @param y      the y position in target to receive the items from the first cell in source
185     * @return a modified copy of target with source inserted into it at the given position
186     */
187    public static char[][] insert(char[][] source, char[][] target, int x, int y) {
188        if (source == null || target == null)
189            return target;
190        if (source.length < 1 || source[0].length < 1)
191            return copy(target);
192        for (int i = 0; i < source.length && x + i < target.length; i++) {
193            System.arraycopy(source[i], 0, target[x + i], y, Math.min(source[i].length, target[x + i].length - y));
194        }
195        return target;
196    }

405    /**
406     * Rearranges an ArrayList to use the given ordering, returning a copy; random orderings can be produced with
407     * {@link squidpony.squidmath.RNG#randomOrdering(int)} or
408     * {@link squidpony.squidmath.RNG#randomOrdering(int, int[])}. These orderings will never repeat an earlier element,
409     * and the returned ArrayList may be shorter than the original if {@code ordering} isn't as long as {@code list}.
410     * Using a random ordering is like shuffling, but allows you to repeat the shuffle exactly on other collections of
411     * the same size. A reordering can also be inverted with {@link #invertOrdering(int[])} or
412     * {@link #invertOrdering(int[], int[])}, getting the change that will undo another ordering.
413     *
414     * @param list     an ArrayList that you want a reordered version of; will not be modified.
415     * @param ordering an ordering, typically produced by one of RNG's randomOrdering methods.
416     * @param <T>      any generic type
417     * @return a modified copy of {@code list} with its ordering changed to match {@code ordering}.
418     */
419    public static <T> ArrayList<T> reorder(ArrayList<T> list, int... ordering) {
420        int ol;
421        if (list == null || ordering == null || (ol = Math.min(list.size(), ordering.length)) == 0)
422            return list;
423        ArrayList<T> alt = new ArrayList<T>(ol);
424        for (int i = 0; i < ol; i++) {
425            alt.add(list.get((ordering[i] % ol + ol) % ol));
426        }
427        return alt;
428    }

430    /**
431     * Given an ordering such as one produced by {@link squidpony.squidmath.RNG#randomOrdering(int, int[])}, this finds
432     * its inverse, able to reverse the reordering and vice versa.
433     *
434     * @param ordering the ordering to find the inverse for
435     * @return the inverse of ordering
436     */
437    public static int[] invertOrdering(int[] ordering) {
438        int ol = 0;
439        if (ordering == null || (ol = ordering.length) == 0) return ordering;
440        int[] next = new int[ol];
441        for (int i = 0; i < ol; i++) {
442            if (ordering[i] < 0 || ordering[i] >= ol) return next;
443            next[ordering[i]] = i;
444        }
445        return next;
446    }

448    /**
449     * Given an ordering such as one produced by {@link squidpony.squidmath.RNG#randomOrdering(int, int[])}, this finds
450     * its inverse, able to reverse the reordering and vice versa. This overload doesn't allocate a new int
451     * array, and instead relies on having an int array of the same size as ordering passed to it as an
452     * additional argument.
453     *
454     * @param ordering the ordering to find the inverse for
455     * @param dest     the int array to put the inverse reordering into; should have the same length as ordering
456     * @return the inverse of ordering; will have the same value as dest
457     */
458    public static int[] invertOrdering(int[] ordering, int[] dest) {
459        int ol = 0;
460        if (ordering == null || dest == null || (ol = Math.min(ordering.length, dest.length)) == 0)
461            return ordering;
462        for (int i = 0; i < ol; i++) {
463            if (ordering[i] < 0 || ordering[i] >= ol) return dest;
464            dest[ordering[i]] = i;
465        }
466        return dest;
467    }
