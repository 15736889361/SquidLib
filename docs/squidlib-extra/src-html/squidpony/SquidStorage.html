016/**
017 * Helps games store information in libGDX's Preferences class as Strings, then get it back out.
018 * Created by Tommy Ettinger on 9/16/2016.
019 */
020@Beta
021public class SquidStorage {
022    public final Preferences preferences;
023    public final String storageName;
024    public final Json json;
025    protected OrderedMap<String, String> contents;
026    public boolean compress = true;
027
028    public SquidStorage()
029    {
030        this("nameless");
031    }
032    public SquidStorage(final String name)
033    {
034        storageName = name;
035        preferences = Gdx.app.getPreferences(storageName);
036        json = new Json(JsonWriter.OutputType.minimal);
037
038        json.setSerializer(Pattern.class, new Json.Serializer<Pattern>() {
039            @Override
040            public void write(Json json, Pattern object, Class knownType) {
041                if(object == null)
042                {
043                    json.writeValue(null);
044                    return;
045                }
046                json.writeValue(object.serializeToString());
047            }
048
049            @Override
050            public Pattern read(Json json, JsonValue jsonData, Class type) {
051                if(jsonData == null || jsonData.isNull()) return null;
052                String data = jsonData.asString();
053                if(data == null || data.length() < 2) return null;
054                return Pattern.deserializeFromString(data);
055            }
056        });
057
058        json.setSerializer(GreasedRegion.class, new Json.Serializer<GreasedRegion>() {
059            @Override
060            public void write(Json json, GreasedRegion object, Class knownType) {
061                if(object == null)
062                {
063                    json.writeValue(null);
064                    return;
065                }
066                json.writeObjectStart();
067                json.writeValue("w", object.width);
068                json.writeValue("h", object.height);
069                json.writeValue("d", object.data);
070                json.writeObjectEnd();
071            }
072
073            @Override
074            public GreasedRegion read(Json json, JsonValue jsonData, Class type) {
075                if(jsonData == null || jsonData.isNull()) return null;
076                return new GreasedRegion(jsonData.get("d").asLongArray(), jsonData.getInt("w"), jsonData.getInt("h"));
077            }
078        });
079
080        json.setSerializer(IntDoubleOrderedMap.class, new Json.Serializer<IntDoubleOrderedMap>() {
081            @Override
082            public void write(Json json, IntDoubleOrderedMap object, Class knownType) {
083                if(object == null)
084                {
085                    json.writeValue(null);
086                    return;
087                }
088                json.writeObjectStart();
089                json.writeArrayStart("k");
090                IntDoubleOrderedMap.KeyIterator ki = object.keySet().iterator();
091                while (ki.hasNext())
092                    json.writeValue(ki.nextInt());
093                json.writeArrayEnd();
094                json.writeArrayStart("v");
095                IntDoubleOrderedMap.DoubleIterator vi = object.values().iterator();
096                while (vi.hasNext())
097                    json.writeValue(vi.nextDouble());
098                json.writeArrayEnd();
099                json.writeValue("f", object.f);
100                json.writeObjectEnd();
101            }
102
103            @Override
104            public IntDoubleOrderedMap read(Json json, JsonValue jsonData, Class type) {
105                if(jsonData == null || jsonData.isNull()) return null;
106                return new IntDoubleOrderedMap(jsonData.get("k").asIntArray(), jsonData.get("v").asDoubleArray(), jsonData.getFloat("f"));
107            }
108        });
109
110        json.setSerializer(OrderedMap.class, new Json.Serializer<OrderedMap>() {
111            @Override
112            public void write(Json json, OrderedMap object, Class knownType) {
113                if(object == null)
114                {
115                    json.writeValue(null);
116                    return;
117                }
118                json.writeObjectStart();
119                int sz = object.size();
120                json.writeValue("k", object.keysAsOrderedSet(), OrderedSet.class);
121                json.writeValue("v", object.valuesAsList(), ArrayList.class);
122                json.writeValue("f", object.f);
123                json.writeObjectEnd();
124            }
125
126            @Override
127            @SuppressWarnings("unchecked")
128            public OrderedMap read(Json json, JsonValue jsonData, Class type) {
129                if(jsonData == null || jsonData.isNull()) return null;
130                return new OrderedMap(json.readValue(OrderedSet.class, jsonData.get("k")),
131                        json.readValue(ArrayList.class, jsonData.get("v")), jsonData.getFloat("f"));
132            }
133        });
134        json.setSerializer(char[][].class, new Json.Serializer<char[][]>() {
135            @Override
136            public void write(Json json, char[][] object, Class knownType) {
137                if(object == null)
138                {
139                    json.writeValue(null);
140                    return;
141                }
142                int sz = object.length;
143                json.writeArrayStart();
144                for (int i = 0; i < sz; i++) {
145                    json.writeValue(String.valueOf(object[i]));
146                }
147                json.writeArrayEnd();
148            }
149
150            @Override
151            public char[][] read(Json json, JsonValue jsonData, Class type) {
152                if(jsonData == null || jsonData.isNull())
153                    return null;
154                int sz = jsonData.size;
155                char[][] data = new char[sz][];
156                JsonValue c = jsonData.child();
157                for (int i = 0; i < sz && c != null; i++, c = c.next()) {
158                    data[i] = c.asString().toCharArray();
159                }
160                return data;
161            }
162        });
163        json.setSerializer(FakeLanguageGen.class, new Json.Serializer<FakeLanguageGen>() {
164            @Override
165            public void write(Json json, FakeLanguageGen object, Class knownType) {
166                if(object == null)
167                {
168                    json.writeValue(null);
169                    return;
170                }
171                json.writeValue(object.serializeToString());
172            }
173
174            @Override
175            public FakeLanguageGen read(Json json, JsonValue jsonData, Class type) {
176                if(jsonData == null || jsonData.isNull())
177                    return null;
178                return FakeLanguageGen.deserializeFromString(jsonData.asString());
179            }
180        });
181        json.addClassTag("#St", String.class);
182        json.addClassTag("#Z", Boolean.class);
183        json.addClassTag("#z", boolean.class);
184        json.addClassTag("#B", Byte.class);
185        json.addClassTag("#b", byte.class);
186        json.addClassTag("#S", Short.class);
187        json.addClassTag("#s", short.class);
188        json.addClassTag("#C", Character.class);
189        json.addClassTag("#c", char.class);
190        json.addClassTag("#I", Integer.class);
191        json.addClassTag("#i", int.class);
192        json.addClassTag("#F", Float.class);
193        json.addClassTag("#f", float.class);
194        json.addClassTag("#L", Long.class);
195        json.addClassTag("#l", long.class);
196        json.addClassTag("#D", Double.class);
197        json.addClassTag("#d", double.class);
198        json.addClassTag("#Grea", GreasedRegion.class);
199        json.addClassTag("#IDOM", IntDoubleOrderedMap.class);
200        json.addClassTag("#Lang", FakeLanguageGen.class);
201        json.addClassTag("#LnAl", FakeLanguageGen.Alteration.class);
202        json.addClassTag("#LnMd", FakeLanguageGen.Modifier.class);
203        json.addClassTag("#OMap", OrderedMap.class);
204        json.addClassTag("#OSet", OrderedSet.class);
205        json.addClassTag("#IVLA", IntVLA.class);
206        json.addClassTag("#SVLA", ShortVLA.class);
207        json.addClassTag("#RNG", RNG.class);
208        json.addClassTag("#SRNG", StatefulRNG.class);
209        json.addClassTag("#EdiR", EditRNG.class);
210        json.addClassTag("#DhaR", DharmaRNG.class);
211        json.addClassTag("#DecR", DeckRNG.class);
212        json.addClassTag("#Ligh", LightRNG.class);
213        json.addClassTag("#LonP", LongPeriodRNG.class);
214        json.addClassTag("#Thun", ThunderRNG.class);
215        json.addClassTag("#XoRo", XoRoRNG.class);
216        json.addClassTag("#XorR", XorRNG.class);
217        json.addClassTag("#Strm", CrossHash.Storm.class);
218
219
220
221        contents = new OrderedMap<>(16, 0.2f);
222    }
223    public SquidStorage put(String innerName, Object o)
224    {
225        contents.put(innerName, json.toJson(o));
226        return this;
227    }
228    public SquidStorage store(String outerName)
229    {
230        if(compress)
231            preferences.putString(outerName, LZSEncoding.compressToUTF16(json.toJson(contents, OrderedMap.class)));
232        else
233            preferences.putString(outerName, json.toJson(contents, OrderedMap.class));
234        preferences.flush();
235        return this;
236    }
237    public String show()
238    {
239        if(compress)
240            return LZSEncoding.compressToUTF16(json.toJson(contents, OrderedMap.class));
241        else
242            return json.toJson(contents, OrderedMap.class);
243    }
244    public SquidStorage clear()
245    {
246        contents.clear();
247        return this;
248    }
249    public SquidStorage remove(String innerName)
250    {
251        contents.remove(innerName);
252        return this;
253    }
254    @SuppressWarnings("unchecked")
255    public <T> T get(String outerName, String innerName, Class<T> type)
256    {
257        OrderedMap<String, String> om;
258        if(compress)
259            om = json.fromJson(OrderedMap.class, LZSEncoding.decompressFromUTF16(preferences.getString(outerName)));
260        else
261            om = json.fromJson(OrderedMap.class, preferences.getString(outerName));
262        if(om == null) return null;
263        return json.fromJson(type, om.get(innerName));
264    }
265    public int preferencesSize()
266    {
267        Map<String, ?> p = preferences.get();
268        int byteSize = 0;
269        for(String k : p.keySet())
270        {
271            byteSize += k.length();
272            byteSize += preferences.getString(k, "").length();
273        }
274        return byteSize * 2;
275    }
