017/**
018 * Helps games store information in libGDX's Preferences class as Strings, then get it back out.
019 * Created by Tommy Ettinger on 9/16/2016.
020 */
021@Beta
022public class SquidStorage {
023    public final Preferences preferences;
024    public final String storageName;
025    public final Json json;
026    protected OrderedMap<String, String> contents;
027    public boolean compress = true;
028
029    /**
030     * Please don't use this constructor if possible; it simply calls {@link #SquidStorage(String)} with the constant
031     * String "nameless". This could easily overlap with other files/sections in Preferences, so you should always
032     * prefer giving a String argument to the constructor, typically the name of the game.
033     * @see #SquidStorage(String) the recommended constructor to use
034     */
035    public SquidStorage()
036    {
037        this("nameless");
038    }
039
040    /**
041     * Creates a SquidStorage with the given fileName to save using Preferences from libGDX. The name should generally
042     * be the name of this game or application, and must be a valid name for a file (so no slashes, backslashes, colons,
043     * semicolons, or commas for certain, and other non-alphanumeric characters are also probably invalid). You should
044     * not assume anything is present in the Preferences storage unless you have put it there, and this applies doubly
045     * to games or applications other than your own; you should avoid values for fileName that might overlap with
046     * another game's Preferences values.
047     * <br>
048     * To organize saved data into sub-sections, you specify logical units (like different players' saved games) with a
049     * String outerName when you call {@link #store(String)}, and can further distinguish data under the outerName when
050     * you call {@link #put(String, Object)} to put each individual item into the saved storage with its own innerName.
051     * <br>
052     * Calling this also sets up custom serializers for several important types in SquidLib; char[][], OrderedMap,
053     * IntDoubleOrderedMap, FakeLanguageGen, GreasedRegion, and notably Pattern from RegExodus all have smaller
054     * serialized representations than the default. OrderedMap allows non-String keys, which gets around a limitation in
055     * JSON maps normally, and both FakeLanguageGen and Pattern are amazingly smaller with the custom representation.
056     * The custom char[][] representation is about half the normal size by omitting commas after each char.
057     * @param fileName the valid file name to create or open from Preferences; typically the name of the game/app.
058     */
059    public SquidStorage(final String fileName)
060    {
061        storageName = fileName;
062        preferences = Gdx.app.getPreferences(storageName);
063        json = new Json(JsonWriter.OutputType.minimal);
064
065        json.setSerializer(Pattern.class, new Json.Serializer<Pattern>() {
066            @Override
067            public void write(Json json, Pattern object, Class knownType) {
068                if(object == null)
069                {
070                    json.writeValue(null);
071                    return;
072                }
073                json.writeValue(object.serializeToString());
074            }
075
076            @Override
077            public Pattern read(Json json, JsonValue jsonData, Class type) {
078                if(jsonData == null || jsonData.isNull()) return null;
079                String data = jsonData.asString();
080                if(data == null || data.length() < 2) return null;
081                return Pattern.deserializeFromString(data);
082            }
083        });
084
085        json.setSerializer(GreasedRegion.class, new Json.Serializer<GreasedRegion>() {
086            @Override
087            public void write(Json json, GreasedRegion object, Class knownType) {
088                if(object == null)
089                {
090                    json.writeValue(null);
091                    return;
092                }
093                json.writeObjectStart();
094                json.writeValue("w", object.width);
095                json.writeValue("h", object.height);
096                json.writeValue("d", object.data);
097                json.writeObjectEnd();
098            }
099
100            @Override
101            public GreasedRegion read(Json json, JsonValue jsonData, Class type) {
102                if(jsonData == null || jsonData.isNull()) return null;
103                return new GreasedRegion(jsonData.get("d").asLongArray(), jsonData.getInt("w"), jsonData.getInt("h"));
104            }
105        });
106
107        json.setSerializer(IntDoubleOrderedMap.class, new Json.Serializer<IntDoubleOrderedMap>() {
108            @Override
109            public void write(Json json, IntDoubleOrderedMap object, Class knownType) {
110                if(object == null)
111                {
112                    json.writeValue(null);
113                    return;
114                }
115                json.writeObjectStart();
116                json.writeArrayStart("k");
117                IntDoubleOrderedMap.KeyIterator ki = object.keySet().iterator();
118                while (ki.hasNext())
119                    json.writeValue(ki.nextInt());
120                json.writeArrayEnd();
121                json.writeArrayStart("v");
122                IntDoubleOrderedMap.DoubleIterator vi = object.values().iterator();
123                while (vi.hasNext())
124                    json.writeValue(vi.nextDouble());
125                json.writeArrayEnd();
126                json.writeValue("f", object.f);
127                json.writeObjectEnd();
128            }
129
130            @Override
131            public IntDoubleOrderedMap read(Json json, JsonValue jsonData, Class type) {
132                if(jsonData == null || jsonData.isNull()) return null;
133                return new IntDoubleOrderedMap(jsonData.get("k").asIntArray(), jsonData.get("v").asDoubleArray(), jsonData.getFloat("f"));
134            }
135        });
136
137        json.setSerializer(OrderedMap.class, new Json.Serializer<OrderedMap>() {
138            @Override
139            public void write(Json json, OrderedMap object, Class knownType) {
140                if(object == null)
141                {
142                    json.writeValue(null);
143                    return;
144                }
145                json.writeObjectStart();
146                json.writeValue("k", object.keysAsOrderedSet(), OrderedSet.class);
147                json.writeValue("v", object.valuesAsList(), ArrayList.class);
148                json.writeValue("f", object.f);
149                json.writeObjectEnd();
150            }
151
152            @Override
153            @SuppressWarnings("unchecked")
154            public OrderedMap read(Json json, JsonValue jsonData, Class type) {
155                if(jsonData == null || jsonData.isNull()) return null;
156                return new OrderedMap(json.readValue(OrderedSet.class, jsonData.get("k")),
157                        json.readValue(ArrayList.class, jsonData.get("v")), jsonData.getFloat("f"));
158            }
159        });
160
161        json.setSerializer(Arrangement.class, new Json.Serializer<Arrangement>() {
162            @Override
163            public void write(Json json, Arrangement object, Class knownType) {
164                if(object == null)
165                {
166                    json.writeValue(null);
167                    return;
168                }
169                json.writeObjectStart();
170                json.writeValue("k", object.keysAsOrderedSet(), OrderedSet.class);
171                json.writeValue("f", object.f);
172                json.writeObjectEnd();
173            }
174
175            @Override
176            @SuppressWarnings("unchecked")
177            public Arrangement read(Json json, JsonValue jsonData, Class type) {
178                if(jsonData == null || jsonData.isNull()) return null;
179                return new Arrangement(json.readValue(OrderedSet.class, jsonData.get("k")), jsonData.getFloat("f"));
180            }
181        });
182
183        json.setSerializer(TwoKey.class, new Json.Serializer<TwoKey>() {
184            @Override
185            public void write(Json json, TwoKey object, Class knownType) {
186                if(object == null)
187                {
188                    json.writeValue(null);
189                    return;
190                }
191                json.writeObjectStart();
192                json.writeValue("a", object.getSetA(), SortedSet.class);
193                json.writeValue("b", object.getSetB(), SortedSet.class);
194                json.writeObjectEnd();
195            }
196
197            @Override
198            @SuppressWarnings("unchecked")
199            public TwoKey read(Json json, JsonValue jsonData, Class type) {
200                if(jsonData == null || jsonData.isNull()) return null;
201                return new TwoKey(json.readValue(SortedSet.class, jsonData.get("a")), json.readValue(SortedSet.class, jsonData.get("b")));
202            }
203        });
204
205        json.setSerializer(char[][].class, new Json.Serializer<char[][]>() {
206            @Override
207            public void write(Json json, char[][] object, Class knownType) {
208                if(object == null)
209                {
210                    json.writeValue(null);
211                    return;
212                }
213                int sz = object.length;
214                json.writeArrayStart();
215                for (int i = 0; i < sz; i++) {
216                    json.writeValue(String.valueOf(object[i]));
217                }
218                json.writeArrayEnd();
219            }
220
221            @Override
222            public char[][] read(Json json, JsonValue jsonData, Class type) {
223                if(jsonData == null || jsonData.isNull())
224                    return null;
225                int sz = jsonData.size;
226                char[][] data = new char[sz][];
227                JsonValue c = jsonData.child();
228                for (int i = 0; i < sz && c != null; i++, c = c.next()) {
229                    data[i] = c.asString().toCharArray();
230                }
231                return data;
232            }
233        });
234        json.setSerializer(FakeLanguageGen.class, new Json.Serializer<FakeLanguageGen>() {
235            @Override
236            public void write(Json json, FakeLanguageGen object, Class knownType) {
237                if(object == null)
238                {
239                    json.writeValue(null);
240                    return;
241                }
242                json.writeValue(object.serializeToString());
243            }
244
245            @Override
246            public FakeLanguageGen read(Json json, JsonValue jsonData, Class type) {
247                if(jsonData == null || jsonData.isNull())
248                    return null;
249                return FakeLanguageGen.deserializeFromString(jsonData.asString());
250            }
251        });

268        json.addClassTag("#d", double.class);

270        json.addClassTag("#Patt", Pattern.class);
271        /*

276        json.addClassTag("#Grea", GreasedRegion.class);

278        json.addClassTag("#Lang", FakeLanguageGen.class);
279        json.addClassTag("#LnAl", FakeLanguageGen.Alteration.class);
280        json.addClassTag("#LnMd", FakeLanguageGen.Modifier.class);

299        contents = new OrderedMap<>(16, 0.2f);
300    }

302    /**
303     * Prepares to store the Object {@code o} to be retrieved with {@code innerName} in the current group of objects.
304     * Does not write to a permanent location until {@link #store(String)} is called. The innerName used to store an
305     * object is required to get it back again, and can also be used to remove it before storing (or storing again).
306     * @param innerName one of the two Strings needed to retrieve this later
307     * @param o the Object to prepare to store
308     * @return this for chaining
309     */
310    public SquidStorage put(String innerName, Object o)
311    {
312        contents.put(innerName, json.toJson(o));
313        return this;
314    }
315
316    /**
317     * Actually stores all objects that had previously been prepared with {@link #put(String, Object)}, with
318     * {@code outerName} used as a key to retrieve any object in the current group. Flushes the preferences, making the
319     * changes permanent (until overwritten), but does not change the current group (you may want to call this method
320     * again with additional items in the current group, and that would simply involve calling put() again). If you want
321     * to clear the current group, use {@link #clear()}. If you want to remove just one object from the current group,
322     * use {@link #remove(String)}.
323     * @param outerName one of the two Strings needed to retrieve any of the objects in the current group
324     * @return this for chaining
325     */
326    public SquidStorage store(String outerName)
327    {
328        if(compress)
329            preferences.putString(outerName, LZSEncoding.compressToUTF16(json.toJson(contents, OrderedMap.class)));
330        else
331            preferences.putString(outerName, json.toJson(contents, OrderedMap.class));
332        preferences.flush();
333        return this;
334    }
335
336    /**
337     * Gets a String representation of the data that would be saved when {@link #store(String)} is called. This can be
338     * useful for finding particularly problematic objects that require unnecessary space when serialized.
339     * @return a String that previews what would be stored permanently when {@link #store(String)} is called
340     */
341    public String show()
342    {
343        if(compress)
344            return LZSEncoding.compressToUTF16(json.toJson(contents, OrderedMap.class));
345        else
346            return json.toJson(contents, OrderedMap.class);
347    }
348
349    /**
350     * Clears the current group of objects; recommended if you intend to store under multiple outerName keys.
351     * @return this for chaining
352     */
353    public SquidStorage clear()
354    {
355        contents.clear();
356        return this;
357    }

359    /**
360     * Removes one object from the current group by the {@code innerName} it was prepared with using
361     * {@link #put(String, Object)}. This does not affect already-stored objects unless {@link #store(String)} is called
362     * after this, in which case the new version of the current group, without the object this removed, is stored.
363     * @param innerName the String key used to put an object in the current group with {@link #put(String, Object)}
364     * @return this for chaining
365     */
366    public SquidStorage remove(String innerName)
367    {
368        contents.remove(innerName);
369        return this;
370    }

372    /**
373     * Gets an object from the storage by the given {@code outerName} key from {@link #store(String)} and
374     * {@code innerName} key from {@link #put(String, Object)}, and uses the class given by {@code type} for the
375     * returned value, assuming it matches the object that was originally put with those keys. If no such object is
376     * present, returns null. Results are undefined if {@code type} doesn't match the actual class of the stored object.
377     * @param outerName the key used to store the group of objects with {@link #store(String)}
378     * @param innerName the key used to store the specific object with {@link #put(String, Object)}
379     * @param type the class of the value; for a class like RNG, use {@code RNG.class}, but changed to fit
380     * @param <T> the type of the value to retrieve; if type was {@code RNG.class}, this would be {@code RNG}
381     * @return the retrieved value if successful, or null otherwise
382     */
383    @SuppressWarnings("unchecked")
384    public <T> T get(String outerName, String innerName, Class<T> type)
385    {
386        OrderedMap<String, String> om;
387        String got;
388        if(compress)
389            got = LZSEncoding.decompressFromUTF16(preferences.getString(outerName));
390        else
391            got = preferences.getString(outerName);
392        if(got == null) return null;
393        om = json.fromJson(OrderedMap.class, got);
394        if(om == null) return null;
395        return json.fromJson(type, om.get(innerName));
396    }

398    /**
399     * Gets the approximate size of the currently-stored preferences. This assumes UTF-16 storage, which is the case for
400     * GWT's LocalStorage. Since GWT is restricted to the size the browser permits for LocalStorage, and this limit can
401     * be rather small (about 5 MB, sometimes more but not reliably), this method is especially useful there, but it may
402     * yield inaccurate sizes on other platforms that save Preferences data differently.
403     * @return the size, in bytes, of the already-stored preferences
404     */
405    public int preferencesSize()
406    {
407        Map<String, ?> p = preferences.get();
408        int byteSize = 0;
409        for(String k : p.keySet())
410        {
411            byteSize += k.length();
412            byteSize += preferences.getString(k, "").length();
413        }
414        return byteSize * 2;
415    }
