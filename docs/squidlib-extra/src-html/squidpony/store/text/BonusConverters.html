001package squidpony.store.text;
002
003import com.badlogic.gdx.utils.reflect.ClassReflection;
004import squidpony.Converters;
005import squidpony.StringConvert;
006import squidpony.StringKit;
007import squidpony.squidgrid.mapping.SpillWorldMap;
008import squidpony.squidmath.OrderedMap;
009import squidpony.squidmath.StatefulRNG;
010import squidpony.squidmath.StatefulRandomness;
011
012/**
013 * Created by Tommy Ettinger on 4/22/2017.
014 */
015public class BonusConverters {
016    public static final StringConvert<OrderedMap<Character, String>> convertMapCharString =
017            Converters.convertOrderedMap(Converters.convertChar, Converters.convertString);
018    public static final StringConvert<StatefulRNG> convertStatefulRNG = new StringConvert<StatefulRNG>("StatefulRNG") {
019        @Override
020        public String stringify(StatefulRNG item) {
021            return StringKit.hex(item.getState()) + ':' + (item.getRandomness().getClass().getSimpleName());
022        }
023
024        @Override
025        public StatefulRNG restore(String text) {
026            long state = StringKit.longFromHex(text);
027            try {
028                StatefulRandomness sr = (StatefulRandomness) (ClassReflection.forName(text.substring(text.indexOf(':') + 1)).newInstance());
029                sr.setState(state);
030                return new StatefulRNG(sr);
031            }catch (Exception re)
032            {
033                return new StatefulRNG(state);
034            }
035        }
036    };
037    public static final StringConvert<SpillWorldMap> convertSpillWorldMap = new StringConvert<SpillWorldMap>("SpillWorldMap") {
038        @Override
039        public String stringify(SpillWorldMap item) {
040            return item.width
041                    + "\t" + item.height
042                    + "\t" + item.name
043                    + '\t' + Converters.convertArrayInt2D.stringify(item.heightMap)
044                    + '\t' + Converters.convertArrayCoord.stringify(item.mountains)
045                    + '\t' + Converters.convertArrayChar2D.stringify(item.politicalMap)
046                    + '\t' + convertMapCharString.stringify(item.atlas)
047                    + '\t' + convertStatefulRNG.stringify(item.rng);
048        }
049
050        @Override
051        public SpillWorldMap restore(String text) {
052            int pos;
053            SpillWorldMap swm = new SpillWorldMap(
054                    Integer.decode(text.substring(0, (pos = text.indexOf('\t')))),
055                    Integer.decode(text.substring(pos+1, (pos = text.indexOf('\t', pos+1)))),
056                    text.substring(pos+1, (pos = text.indexOf('\t', pos+1)))
057            );
058            swm.heightMap = Converters.convertArrayInt2D.restore(text.substring(pos+1, (pos = text.indexOf('\t', pos+1))));
059            swm.mountains = Converters.convertArrayCoord.restore(text.substring(pos+1, (pos = text.indexOf('\t', pos+1))));
060            swm.politicalMap = Converters.convertArrayChar2D.restore(text.substring(pos+1, (pos = text.indexOf('\t', pos+1))));
061            swm.atlas.clear();
062            swm.atlas.putAll(convertMapCharString.restore(text.substring(pos+1, (pos = text.indexOf('\t', pos+1)))));
063            swm.rng = convertStatefulRNG.restore(text.substring(pos+1));
064            return swm;
065        }
066    };
067
068
069}
