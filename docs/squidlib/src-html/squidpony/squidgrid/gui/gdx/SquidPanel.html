025/**

027 * 
028 * Grid width and height settings are in terms of number of cells. Cell width and height
029 * are in terms of number of pixels.
030 *
031 * When text is placed, the background color is set separately from the foreground character. When moved, only the
032 * foreground character is moved.
033 *
034 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
035 */

118     * Builds a panel with the given grid size and all other parameters determined by the factory. Even if sprite images
119     * are being used, a TextCellFactory is still needed to perform sizing and other utility functions.
120     *
121     * If the TextCellFactory has not yet been initialized, then it will be sized at 12x12 px per cell. If it is null
122     * then a default one will be created and initialized.
123     *
124     * @param gridWidth the number of cells horizontally
125     * @param gridHeight the number of cells vertically
126     * @param factory the factory to use for cell rendering
127     * @param center
128     *                  The color center to use. Can be {@code null}, but then must be set later on with
129     *          {@link #setColorCenter(IColorCenter)}.
130     */
131    public SquidPanel(int gridWidth, int gridHeight, TextCellFactory factory, IColorCenter<Color> center,
132                      float xOffset, float yOffset) {
133        this.gridWidth = gridWidth;
134        this.gridHeight = gridHeight;
135        if(center == null)
136            scc = DefaultResources.getSCC();
137        else
138            scc = center;
139
140        if (factory == null) {
141            textFactory = new TextCellFactory();
142        }
143        else
144            textFactory = factory;
145        if (!textFactory.initialized()) {
146            textFactory.initByFont();
147        }
148
149        cellWidth = MathUtils.round(textFactory.actualCellWidth);
150        cellHeight = MathUtils.round(textFactory.actualCellHeight);
151
152        contents = new String[gridWidth][gridHeight];
153        colors = new Color[gridWidth][gridHeight];
154        for (int i = 0; i < gridWidth; i++) {
155            Arrays.fill(colors[i], scc.filter(Color.CLEAR));
156        }
157
158
159        int w = gridWidth * cellWidth;
160        int h = gridHeight * cellHeight;
161        this.xOffset = xOffset;
162        this.yOffset = yOffset;
163        setSize(w, h);
164        animatedEntities = new OrderedSet<>();
165    }

480    @Override
481    public void draw (Batch batch, float parentAlpha) {
482        textFactory.configureShader(batch);
483        Color tmp;
484        int inc = onlyRenderEven ? 2 : 1;
485        for (int x = gridOffsetX; x < gridWidth; x += inc) {
486            for (int y = gridOffsetY; y < gridHeight; y += inc) {
487                tmp = scc.filter(colors[x][y]);
488                textFactory.draw(batch, contents[x][y], tmp, xOffset + /*- getX() + 1f * */ x * cellWidth,
489                        yOffset + /*- getY() + 1f * */ (gridHeight - y) * cellHeight + 1f);
490            }
491        }
492        super.draw(batch, parentAlpha);
493        for(AnimatedEntity ae : animatedEntities)
494        {
495            ae.actor.act(Gdx.graphics.getDeltaTime());
496        }
497    }

1373     * Like {@link #tint(int, int, Color, float)}, but waits for {@code delay}
1374     * (in seconds) before performing it. Additionally, enqueue {@code postRunnable}
1375     * for running after the created action ends.
1376     * @param delay how long to wait in milliseconds before starting the effect
1377     * @param x the x-coordinate of the cell to tint
1378     * @param y the y-coordinate of the cell to tint
1379     * @param color what to transition ae's color towards, and then transition back from
1380     * @param duration how long the total "round-trip" transition should take in milliseconds
1381     * @param postRunnable a Runnable to execute after the tint completes; may be null to do nothing.
1382     */
1383
1384    public void tint(float delay, int x, int y, Color color, float duration, Runnable postRunnable) {
1385        final Actor a = cellToActor(x, y);
1386        if(a == null)
1387            return;
1388        duration = clampDuration(duration);
1389        animationCount++;
1390
1391        Color ac = scc.filter(a.getColor());
1392
1393        final int nbActions = 3 + (0 < delay ? 1 : 0) + (postRunnable == null ? 0 : 1);
1394        final Action[] sequence = new Action[nbActions];
1395        int index = 0;
1396        if (0 < delay)
1397            sequence[index++] = Actions.delay(delay);
1398        sequence[index++] = Actions.color(color, duration * 0.3f);
1399        sequence[index++] = Actions.color(ac, duration * 0.7f);
1400        if(postRunnable != null)
1401        {
1402            sequence[index++] = Actions.run(postRunnable);
1403        }
1404        /* Do this one last, so that hasActiveAnimations() returns true during 'postRunnable' */
1405        sequence[index] = Actions.run(new Runnable() {
1406            @Override
1407            public void run() {
1408                recallActor(a, true);
1409            }
1410        });
1411
1412        a.addAction(Actions.sequence(sequence));
1413    }
