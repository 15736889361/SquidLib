026/**

028 * 
029 * Grid width and height settings are in terms of number of cells. Cell width and height
030 * are in terms of number of pixels.
031 *
032 * When text is placed, the background color is set separately from the foreground character. When moved, only the
033 * foreground character is moved.
034 *
035 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
036 */

119     * Builds a panel with the given grid size and all other parameters determined by the factory. Even if sprite images
120     * are being used, a TextCellFactory is still needed to perform sizing and other utility functions.
121     *
122     * If the TextCellFactory has not yet been initialized, then it will be sized at 12x12 px per cell. If it is null
123     * then a default one will be created and initialized.
124     *
125     * @param gridWidth the number of cells horizontally
126     * @param gridHeight the number of cells vertically
127     * @param factory the factory to use for cell rendering
128     * @param center
129     *                  The color center to use. Can be {@code null}, but then must be set later on with
130     *          {@link #setColorCenter(IColorCenter)}.
131     */
132    public SquidPanel(int gridWidth, int gridHeight, TextCellFactory factory, IColorCenter<Color> center,
133                      float xOffset, float yOffset) {
134        this.gridWidth = gridWidth;
135        this.gridHeight = gridHeight;
136        if(center == null)
137            scc = DefaultResources.getSCC();
138        else
139            scc = center;
140
141        if (factory == null) {
142            textFactory = new TextCellFactory();
143        }
144        else
145            textFactory = factory;
146        if (!textFactory.initialized()) {
147            textFactory.initByFont();
148        }
149
150        cellWidth = MathUtils.round(textFactory.actualCellWidth);
151        cellHeight = MathUtils.round(textFactory.actualCellHeight);
152
153        contents = new String[gridWidth][gridHeight];
154        colors = ArrayTools.fill(scc.filter(Color.CLEAR).toFloatBits(), gridWidth, gridHeight);
155
156        int w = gridWidth * cellWidth;
157        int h = gridHeight * cellHeight;
158        this.xOffset = xOffset;
159        this.yOffset = yOffset;
160        setSize(w, h);
161        animatedEntities = new OrderedSet<>();
162    }

496    @Override
497    public void draw (Batch batch, float parentAlpha) {
498        textFactory.configureShader(batch);
499        int inc = onlyRenderEven ? 2 : 1;
500        for (int x = gridOffsetX; x < gridWidth; x += inc) {
501            for (int y = gridOffsetY; y < gridHeight; y += inc) {
502                textFactory.draw(batch, contents[x][y],
503                        colors[x][y],
504                        xOffset + /*- getX() + 1f * */ x * cellWidth,
505                        yOffset + /*- getY() + 1f * */ (gridHeight - y) * cellHeight + 1f);
506            }
507        }
508        super.draw(batch, parentAlpha);
509        for(AnimatedEntity ae : animatedEntities)
510        {
511            ae.actor.act(Gdx.graphics.getDeltaTime());
512        }
513    }

1774     * rotationStrength can default to 1 if you want some rotation (which looks good) or 0 if you want the Actors to
1775     * start at the correct rotation and not change that rotation over the course of the effect, but can be between 0
1776     * and 1 or higher than 1 (negative values may also work).
1777     * @param delay amount of time, in seconds, to wait before starting the effect
1778     * @param x the starting, center, x-position to create all Actors at
1779     * @param y the starting, center, y-position to create all Actors at
1780     * @param distance how far away, in cells, to move each actor from the center (Chebyshev distance, forming a square)
1781     * @param eightWay if true, creates 8 Actors and moves them away in a square, otherwise, 4 Actors in a diamond
1782     * @param glyph the char to make a rotate-able Actor of; should definitely be visible
1783     * @param startColor the color to start the effect with
1784     * @param endColor the color to end the effect on
1785     * @param doubleWidth true if this should use the double-width-cell technique, false in most cases
1786     * @param rotationStrength how strongly to rotate the Actors; 0 is no rotation, 1 is a normal rotation
1787     * @param duration how long, in seconds, the effect should last
1788     */
1789    public void burst(float delay, int x, int y, int distance, boolean eightWay, char glyph,
1790                      Color startColor, Color endColor, boolean doubleWidth,
1791                      float rotationStrength, float duration)
1792    {
1793        Direction d;
1794        if(eightWay)
1795        {
1796            for (int i = 0; i < 8; i++) {
1797                d = Direction.CLOCKWISE[i];
1798                summon(delay, x, y, x - d.deltaX * distance, y + d.deltaY * distance,
1799                        glyph, startColor, endColor, doubleWidth,
1800                        45f * i, 45f * (i - rotationStrength),
1801                        duration);
1802            }
1803        }
1804        else
1805        {
1806            for (int i = 0; i < 4; i++) {
1807                d = Direction.CARDINALS_CLOCKWISE[i];
1808                summon(delay, x, y, x - d.deltaX * distance, y + d.deltaY * distance,
1809                        glyph, startColor, endColor, doubleWidth,
1810                        90f * i, 90f * (i - rotationStrength),
1811                        duration);
1812            }
1813
1814        }
1815    }
