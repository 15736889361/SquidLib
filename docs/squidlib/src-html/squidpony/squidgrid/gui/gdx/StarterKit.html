001package squidpony.squidgrid.gui.gdx;
002
003import com.badlogic.gdx.graphics.Color;
004import com.badlogic.gdx.graphics.g2d.SpriteBatch;
005import com.badlogic.gdx.scenes.scene2d.Stage;
006import com.badlogic.gdx.utils.viewport.StretchViewport;
007import com.badlogic.gdx.utils.viewport.Viewport;
008
009/**
010 * A convenience class that groups several commonly-used GUI classes into one object and provides ways to
011 * initialize these kits for specific purposes, some of which would be a challenge to write without this code.
012 * Created by Tommy Ettinger on 8/11/2016.
013 */
014public class StarterKit {
015    /**
016     * One of the more critical parts of rendering text is what font to use, and textFactory should usually
017     * not be reassigned during a game because so many things depend on this value or a copy of it (so the
018     * change might not affect what it was expected to, and might break other things).
019     */
020    public TextCellFactory textFactory;
021    /**
022     * The main way to interact with a text-based grid as for roguelikes. A SquidLayers object stores a
023     * background and foreground SquidPanel, and this configures them as requested.
024     */
025    public SquidLayers layers;
026    /**
027     * The number of grid spaces on the x axis.
028     */
029    public int gridWidth;
030    /**
031     * The number of grid spaces on the y axis.
032     */
033    public int gridHeight;
034    /**
035     * The width of a cell that holds one char, in "relative pixels," where the screen is expected to stretch so
036     * one relative pixel does not generally refer to one actual screen pixel (since high-DPI phones and
037     * laptops may make a single pixel virtually impossible to see with the naked eye).
038     * <br>
039     * By default, this value is doubled to make stretching look more smooth.
040     */
041    public int cellWidth;
042    /**
043     * The height of a cell that holds one char, in "relative pixels," where the screen is expected to stretch
044     * so one relative pixel does not generally refer to one actual screen pixel (since high-DPI phones and
045     * laptops may make a single pixel virtually impossible to see with the naked eye).
046     * <br>
047     * By default, this value is doubled to make stretching look more smooth.
048     */
049    public int cellHeight;
050
051    /**
052     * All visible parts of this class are in the Stage, and if you add additional widget or scene2d.ui Actor
053     * values to your game, they should probably be added to this Stage.
054     */
055    public Stage stage;
056    /**
057     * Used to draw lots of things, but mostly handled internally by the Stage.
058     * You may need to call {@code batch.begin()} and {@code batch.end()} in some cases where you want to
059     * render something that isn't a child of stage but is an Actor or similar render-able object.
060     */
061    public SpriteBatch batch;
062    /**
063     * An important part of how this will be displayed; the viewport defaults to a displayed width of
064     * {@code cellWidth * gridWidth} and a displayed height of {@code cellHeight * gridHeight}, after cellWidth
065     * and cellHeight were doubled by default, and will be stretched or shrunk to fit the actual screen size.
066     */
067    public Viewport viewport;
068
069    public StarterKit(int gridWidth, int gridHeight, int cellWidth, int cellHeight)
070    {
071        this(DefaultResources.getStretchableFont(), gridWidth, gridHeight, cellWidth, cellHeight);
072    }
073
074    public StarterKit(TextCellFactory textFactory, int gridWidth, int gridHeight, int cellWidth, int cellHeight) {
075        this(textFactory, gridWidth, gridHeight, cellWidth, cellHeight, 0, 0);
076    }
077    public StarterKit(TextCellFactory textFactory, int gridWidth, int gridHeight, int cellWidth, int cellHeight,
078                      int additionalWidth, int additionalHeight) {
079        this.cellWidth = cellWidth * 2;
080        this.cellHeight = cellHeight * 2;
081        this.textFactory = DefaultResources.getStretchableFont().setSmoothingMultiplier(2f / 3f)
082                .width(this.cellWidth).height(this.cellHeight).initBySize();
083        layers = new SquidLayers(gridWidth, gridHeight, this.cellWidth, this.cellHeight, textFactory);
084        layers.setTextSize(cellWidth, cellHeight + 2);
085        this.gridWidth = gridWidth;
086        this.gridHeight = gridHeight;
087        batch = new SpriteBatch();
088        viewport = new StretchViewport(this.cellWidth * gridWidth + additionalWidth, this.cellHeight * gridHeight + additionalHeight);
089        stage = new Stage(viewport, batch);
090        stage.addActor(layers);
091    }
092
093    /**
094     * Not a complete drawing solution; so much of the logic related to drawing is specific to each game, like
095     * FOV being used to make certain things not render if they are out of sight, that this doesn't even try to
096     * guess at what a particular game needs for its rendering code. You should probably draw any AnimatedEntity
097     * objects, like what {@link SquidLayers#animateActor(int, int, char, Color)} returns, separately and after
098     * calling this method. The recommended way to draw those objects is with {@link #drawEntity(AnimatedEntity)},
099     * which must be called between SpriteBatch's begin() and end() methods, while this method cannot be called
100     * between those SpriteBatch methods. The solution, usually, is to call this method, then call
101     * {@link SpriteBatch#begin()}, do any logic to determine what AnimatedEntity objects need to be shown (are
102     * they in FOV, are they alive, etc.), draw the ones that should be shown with drawEntity(), and finally
103     * call {@link SpriteBatch#end()} when no more AnimatedEntity objects need to be drawn. Note that this
104     * method renders all of {@link #stage}, which may include other GUI elements using scene2d.ui, but the
105     * AnimatedEntity objects in a SquidLayers aren't part of any Stage to allow them to be rendered as special
106     * cases for visibility.
107     * <br>
108     * Specifically, this applies the current viewport to the stage, draws the stage, and makes any actions or
109     * events related to the stage take effect. Should not be called inside a {@link SpriteBatch#begin()} block,
110     * since this calls it itself by drawing the stage.
111     */
112    public void draw()
113    {
114        stage.getViewport().apply(true);
115        stage.draw();
116        stage.act();
117    }
118
119    /**
120     * Draws an AnimatedEntity object; must be called between {@link SpriteBatch#begin()} and {@link SpriteBatch#end()}.
121     * You can obtain the correct batch with the {@link #batch} field, and ideally all calls to this method will be
122     * inside a single block of one begin() and one end(), that is, the batch shouldn't start and end for each entity
123     * to draw.
124     * @param entity an AnimatedEntity to draw, usually obtained through one of many methods in {@link SquidLayers}
125     */
126    public void drawEntity(AnimatedEntity entity)
127    {
128        if(batch.isDrawing())
129            layers.drawActor(batch, 1.0f, entity);
130    }
131
132
133    /**
134     * Not a complete resize method; this is meant to handle the resizing of this StarterKit only and should be
135     * called inside your main Game, ApplicationListener, etc. class' resize method.
136     * @param width the new width of the screen; should be a parameter from the other resize() method
137     * @param height the new height of the screen; should be a parameter from the other resize() method
138     */
139    public void resize(int width, int height) {
140        viewport.update(width, height, true);
141    }
142
143}
