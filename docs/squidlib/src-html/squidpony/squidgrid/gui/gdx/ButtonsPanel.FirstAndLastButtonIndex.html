021/**
022 * A panel that layouts buttons vertically. It offers various features:
023 * 
024 * <ul>
025 * <li>It can display margins around buttons, with various {@link CornerStyle
026 * styles}.</li>
027 * <li>It returns a handler to detect mouse clicks on button or add it to
028 * {@code this} (if you want the handler to be called from the enclosing
029 * {@link Stage}).</li>
030 * <li>If {@link #shortcutCharacterColor configured}, the handler also handles
031 * key shortcuts. In that case this panel highlights the shortcuts in text
032 * automatically.</li>
033 * <li>To handle what happens when buttons are clicked, you should implement the
034 * abstract method {@link #selectedButton(int)}.</li>
035 * </ul>
036 * 
037 * <p>
038 * The panel is configured via its {@code public} fields. Configuration must
039 * happen in-between creation and the call to {@link #putAll(boolean, boolean)}.
040 * </p>
041 * 
042 * <p>
043 * This class has two different behaviors w.r.t to the backing panels. They can
044 * either be given at creation time or they can be created on the fly. If you're
045 * doing a full screen menu, you should likely give the panels at creation time
046 * (because you'll compute theirs sizes so that they fit the whole screen). If
047 * you're doing an in-game menu, that is quickly dispatched, you should likely
048 * let this class create the panels (it'll create panels as small as possible,
049 * yet that suffice to display the buttons correctly). There is a dedicated
050 * subtype for the first usage: {@link PreAllocatedPanels}.
051 * </p>
052 * 
053 * <p>
054 * Here's a full screen example of this class:
055 * 
056 * <br/>
057 * <br/>
058 * 
059 * <img src="http://i.imgur.com/AQgWeic.png"/>
060 * 
061 * <br/>
062 * <br/>
063 * 
064 * and this shows a non-full screen example (the Drink/Throw/Drop menu):
065 * 
066 * <br/>
067 * <br/>
068 * 
069 * <img src="http://i.imgur.com/dyd7IoN.png"/>
070 * </p>
071 * 
072 * @author smelC
073 * 
074 * @param <T>
075 *            The type of colors.
076 */
077@Deprecated

360        /**
361         * Displays this panel. You should very likely call this method exactly
362         * once.
363         * 
364         * @param putBordersAndMargins
365         *            Whether to draw margins and the border. This should be
366         *            {@code true} if {@code this}'s position (I mean, in terms of
367         *            {@link #setPosition(float, float)}) is set already. If that's
368         *            not the case (for example, because you need this method to be
369         *            called to compute the position from {@link #getHCells()} and
370         *            {@link #getVCells()}), give {@code false}.
371         * @return The {@link InputProcessor} to plug if you want
372         *         {@link #selectedButton(int)} to be called.
373         * 
374         *         <p>
375         *         See {@link #y_gdxToSquid()} to configure the processor's
376         *         behavior.
377         *         </p>
378         * 
379         *         <p>
380         *         If this panel is behind a {@link Stage} (i.e. you're not setting
381         *         the returned processor to {@link Gdx#input}) and you want it to
382         *         receive keyboard events, don't forget to call
383         *         {@link Stage#setKeyboardFocus(Actor)} by giving {@code this}.
384         *         </p>
385         * @throws NullPointerException
386         *             If the text of buttons wasn't given at creation time, and
387         *             {@link #init(List)} wasn't called since then.
388         * @throws IllegalStateException
389         *             If {@link #enableScrolling} is ON but {@link #bg} isn't set.
390         */
391        public InputProcessor putAll(boolean addListener, boolean putBordersAndMargins) {
392                /*
393                 * The number of cells that this panel can span. It is taken from the
394                 * panels if they are preallocated. If that's the case, the panel will
395                 * draw vertical dots that allow to "scroll". This is only done if
396                 * {@link #interButtonMarginn} is 0.
397                 */
398                if (enableScrolling && bg == null)
399                        throw new IllegalStateException("Panels must be preallocated if scrolling is enabled");
400                final int nbVDisplayedCells = enableScrolling ? bg.gridHeight() : Integer.MAX_VALUE;
401                final int nbHDisplayedCells = enableScrolling ? bg.gridWidth() : Integer.MAX_VALUE;
402                if (nbVDisplayedCells < Integer.MAX_VALUE)
403                        Gdx.app.log(SquidTags.LAYOUT,
404                                        "Available rectangle (in cells): " + nbHDisplayedCells + "x" + nbVDisplayedCells);
405                {
406                        hcells = computeRequiredCellsWidth();
407                        vcells = enableScrolling ? nbVDisplayedCells : computeRequiredCellsHeight();
408
409                        if (bg == null) {
410                                /* Panels weren't given at creation time */
411                                assert fg == null;
412                                setPanels(buildPanel(hcells, vcells), buildPanel(hcells, vcells));
413                        }
414
415                        /*
416                         * We don't include buttons margins, because they are rendered using
417                         * ShapeRenderer, whereas the "important" space is the one for the
418                         * SquidPanel, i.e. the text of buttons, the paddings, and the inter
419                         * button margins.
420                         */
421                        /* The call to #cellWidth() requires #bg to be set */
422                        final int width = hcells * cellWidth();
423                        final int availableW = pixelsWidth();
424                        if (availableW < width)
425                                Gdx.app.log("layout",
426                                                "Cannot layout " + getClass().getSimpleName() + " correctly. Required pixels width: "
427                                                                + width + ". Pixels width available: " + availableW);
428
429                        /* The call to #cellHeight() requires #bg to be set */
430                        final int height = vcells * cellHeight();
431                        final int availableH = pixelsHeight();
432                        if (!enableScrolling && availableH < height)
433                                Gdx.app.log("layout",
434                                                "Cannot layout " + getClass().getSimpleName() + " correctly. Required pixels height: "
435                                                                + height + ". Pixels height available: " + availableH);
436                }
437
438                final int gridHeight = bg.gridHeight();
439
440                final int totalVerticalMargin = gridHeight - vcells;
441                /*
442                 * Inexact division is floored (3/2 = 1), hence this aligns to the top.
443                 */
444                this.topMargin = totalVerticalMargin / 2;
445
446                if (buttonsTexts == null)
447                        throw new NullPointerException(
448                                        "The text of buttons must be set before displaying a " + getClass().getSimpleName());
449
450                this.buttons = new ArrayList<>(buttonsTexts.size());
451
452                return putAll0(addListener, putBordersAndMargins, 0, false);
453        }

455        /**
456         * Missing doc: see {@link #putAll(boolean, boolean)}.
457         * 
458         * @param scroll
459         *            Whether this is a scrolling request.
460         */
461        private InputProcessor putAll0(boolean addListener, boolean putBordersAndMargins, int buttonStartingIndex,
462                        boolean scroll) {
463                if (bgColor != null) {
464                        /* Paint whole background */
465                        fill(ICombinedPanel.What.BG, bgColor);
466                }
467
468                if (scroll) {
469                        /*
470                         * Repaint whole foreground, to avoid leaving the end of longest
471                         * entries visible.
472                         */
473                        fill(ICombinedPanel.What.FG, bgColor);
474                }
475
476                final int nbButtons = buttonsTexts.size();
477                final int nbVDisplayedCells = enableScrolling ? bg.gridHeight() : Integer.MAX_VALUE;
478
479                int m = topMargin;
480                /* In number of cells */
481                final int buttonHeight = buttonCellsHeight();
482                int nbb = 0;
483                int i = buttonStartingIndex;
484                boolean first = true;
485                for (; i < nbButtons; i++, nbb++) {
486                        final int alignment;
487                        final int nextm = m + buttonHeight + interButtonMargin;
488                        if (buttonsAlignment != null && i < buttonsAlignment.length)
489                                alignment = buttonsAlignment[i];
490                        else
491                                /* The default: centering */
492                                alignment = 0;
493                        /*
494                         * Lhs: '...' for 'scroll up' (first button). Rhs: '...' for scoll
495                         * down (last button).
496                         */
497                        final boolean last = i == nbButtons - 1;
498                        final boolean firstDots = first && 0 < i;
499                        final boolean lastDots = !last && (nbVDisplayedCells <= nextm);
500                        putButton(i, m, alignment, putBordersAndMargins, firstDots || lastDots, nbb, scroll);
501                        if (firstDots || lastDots)
502                                i--;
503                        m = nextm;
504                        first = false;
505                        if (nbVDisplayedCells <= m)
506                                break;
507                }
508
509                firstLastButtonIndexes = new FirstAndLastButtonIndex(buttonStartingIndex, i);
510                if ((firstLastButtonIndexes.last - firstLastButtonIndexes.start) < buttonsTexts.size()) {
511                        Gdx.app.log(SquidTags.LAYOUT, "Displaying buttons " + firstLastButtonIndexes + ", out of [0,"
512                                        + (buttonsTexts.size() - 1) + "].");
513                }
514
515                if (putBordersAndMargins)
516                        putBorder();
517
518                final InputMultiplexer result = new InputMultiplexer();
519                /*
520                 * First put the key handler. It'd be nice to use SquidLib's KeyHandler
521                 * interface (it'd be more convenient in #selectedButton for complex
522                 * inputs), but it requires more work.
523                 */
524                result.addProcessor(buildKeyInputProcessor());
525                /*
526                 * Then the mouse handler. SquidInput is required here, since it does
527                 * the translation from libgdx's coordinates to SquidLib coordinates.
528                 */
529                result.addProcessor(new SquidInput(new SquidMouse(bg.cellWidth(), bg.cellHeight(), bg.gridWidth(),
530                                bg.gridHeight(), 0, 0, buildMouseInputProcessor())));
531
532                if (addListener) {
533                        addListener(new InputListener() {
534
535                                @Override
536                                public boolean keyDown(InputEvent event, int keycode) {
537                                        return result.keyDown(keycode);
538                                }
539
540                                @Override
541                                public boolean keyUp(InputEvent event, int keycode) {
542                                        return result.keyUp(keycode);
543                                }
544
545                                @Override
546                                public boolean touchDown(InputEvent event, float x, float y, int pointer, int button) {
547                                        return result.touchDown(MathUtils.round(x), MathUtils.round(y), pointer, button);
548                                }
549
550                                @Override
551                                public void touchUp(InputEvent event, float x, float y, int pointer, int button) {
552                                        result.touchUp(MathUtils.round(x), MathUtils.round(y), pointer, buttonHeight);
553                                }
554
555                                @Override
556                                public boolean keyTyped(InputEvent event, char character) {
557                                        return result.keyTyped(character);
558                                }
559
560                        });
561                }
562
563                putHook();
564
565                return result;
566        }

568        /**
569         * Draws the margins around the buttons and the border around the panel.
570         * Because this uses a {@link ShapeRenderer}, margins do not move if
571         * {@code this} is moved (such as with
572         * {@link Actions#moveTo(float, float, float)}). This method's purpose is to
573         * draw margins in such a case.
574         * 
575         * <p>
576         * Remember that, when using {@link SquidPanel}s; margins (more generally:
577         * anything done with {@link ShapeRenderer}) should always be drawn after
578         * everything else, so when sliding/moving, your code should be like:
579         * 
580         * <pre>
581         * stage.act()   <- if the panel is moving
582         * stage.draw(); <- of course, 'stage' is the Stage containing {@code this}
583         * buttonsPanel.putMarginsAndBorder();
584         * </pre>
585         * </p>
586         */
587        public void putButtonsMarginsAndBorder() {
588                if (buttons == null)
589                        /* Not set yet */
590                        return;
591
592                /* Margins around buttons */
593                final int bound = buttons.size();
594                /* index-based loop, to avoid allocating the Iterator */
595                for (int i = 0; i < bound; i++) {
596                        final Rectangle r = buttons.get(i);
597                        displayMarginsAround(r.botLeftX, r.botLeftY, r.width, r.height);
598                }
599
600                /* The border */
601                putBorder();
602        }

634        protected InputProcessor buildMouseInputProcessor() {
635                return new InputAdapter() {
636
637                        @Override
638                        public boolean touchDown(int screenX, int screenY, int pointer, int button) {
639                                /* Because we wanna receive touchUp */
640                                return true;
641                        }
642
643                        @Override
644                        public boolean touchUp(int x, int gdxy, int pointer, int button) {
645                                final int y;
646                                if (y_gdxToSquid())
647                                        y = getGridHeight() - (gdxy + 1);
648                                else
649                                        y = gdxy;
650                                final int bound = buttons.size();
651                                for (int i = 0; i < bound; i++) {
652                                        final Rectangle r = buttons.get(i);
653                                        if (x < r.botLeftX)
654                                                /* Too much to the left */
655                                                continue;
656                                        if (y < r.botLeftY - r.height)
657                                                /* Too high */
658                                                continue;
659                                        if (r.botLeftY <= y)
660                                                /* Too low */
661                                                continue;
662                                        if (r.botLeftX + r.width <= x)
663                                                /* Too much to the right */
664                                                continue;
665
666                                        /* It's a hit! */
667
668                                        if (i == 0 && hasScrollUp()) {
669                                                final boolean b = scrollUp(Input.Keys.UP);
670                                                assert b;
671                                                return true;
672                                        } else if (i == scrollDownButtonIndex()) {
673                                                final boolean b = scrollDown(Input.Keys.DOWN);
674                                                assert b;
675                                                return true;
676                                        }
677
678                                        final int j = i + firstLastButtonIndexes.start + (hasScrollUp() ? -1 : 0);
679
680                                        if (doNotBind == null || !doNotBind.contains(j)) {
681                                                if (j < buttonsTexts.size()) {
682                                                        /* Send event */
683                                                        selectedButton(j);
684                                                } else {
685                                                        /* Should not happen */
686                                                        Gdx.app.log(SquidTags.LAYOUT,
687                                                                        "Skipping invalid button index in " + getClass().getSimpleName() + ": "
688                                                                                        + j + ". Maximum index is " + (buttons.size() - 1)
689                                                                                        + ". This is a bug.");

862        /**
863         * @param buttonIndex
864         *            The index of the button for which the text is being built.
865         * @param text
866         * @return The text to display, with the coloring to highlight the shortcut.
867         */
868        protected IColoredString<T> putShortcut(int buttonIndex, IColoredString<T> text) {
869                assert 0 <= buttonIndex && buttonIndex < buttonsTexts.size();
870                if (shortcutCharacterColor == null || (doNotBind != null && doNotBind.contains(buttonIndex)))
871                        /*
872                         * Shortcuts are disabled or no shortcut should be put on this
873                         * button
874                         */
875                        return text;
876
877                /* @Nullable */ Character prevShortcut = null;
878
879                if (shortcuts == null)
880                        shortcuts = new HashMap<>();
881                else {
882                        for (Map.Entry<Character, Integer> entry : shortcuts.entrySet()) {
883                                if (entry.getValue() == buttonIndex) {
884                                        /* Shortcut was decided already, let's reuse it */
885                                        prevShortcut = entry.getKey();
886                                        break;
887                                }
888                        }
889                }
890
891                final IColoredString<T> result = new IColoredString.Impl<>();
892                boolean set = false;
893                for (IColoredString.Bucket<T> bucket : text) {
894                        final String bucketText = bucket.getText();
895                        final T bucketColor = bucket.getColor();
896                        if (set) {
897                                /* Append the bucket to the result */
898                                result.append(bucketText, bucketColor);
899                        } else {
900                                final int bound = bucketText.length();
901                                /* Iterate over the bucket's text */
902                                for (int i = 0; i < bound; i++) {
903                                        final char c = bucketText.charAt(i);
904                                        if (!set && prevShortcut != null && Character.toLowerCase(c) == prevShortcut) {
905                                                /* Shortcut was assigned already, let's display it */
906                                                result.append(c, shortcutCharacterColor);
907                                                /* To avoid coming here in next rolls */
908                                                set = true;
909                                        } else if (set || shortcuts.containsKey(Character.toLowerCase(c))
910                                                        || !canBeShortcut(c) || (unbindable != null && unbindable.contains(c))
911                                                        || (enableScrolling && (c == 'j' || c == 'k'))) {
912                                                /*
913                                                 * Shortcut already used or we went into the 'else'
914                                                 * already, or character is inadequate, or character can
915                                                 * be used to scroll (in vim-style).
916                                                 */
917                                                result.append(c, bucketColor);
918                                        } else {
919                                                /* Let's use 'c' as the shortcut! */
920                                                result.append(c, shortcutCharacterColor);
921                                                shortcuts.put(Character.toLowerCase(c), buttonIndex);
922                                                set = true;
923                                        }
924                                }
925                        }
926                }
927                return result;
928        }

941        protected boolean scrollDown(int keycode) {
942                if (enableScrolling
943                                && (keycode == Input.Keys.DOWN || keycode == Input.Keys.NUMPAD_2
944                                                || /* vim-style */ keycode == Input.Keys.J)
945                                && firstLastButtonIndexes.last < buttonsTexts.size() - 1) {
946                        /*
947                         * A request to scroll down && last displayed button is not last
948                         * button. If this is the first scroll, we straight go to the third
949                         * item (if possible, that's why there's the Math.min call), to
950                         * avoid just replacing the first one by '...'.
951                         */
952                        final int nbButtons = buttonsTexts.size();
953                        final int i = Math.min(firstLastButtonIndexes.start + (firstLastButtonIndexes.start == 0 ? 2 : 1),
954                                        nbButtons - 1);
955                        Gdx.app.log(SquidTags.LAYOUT, "Putting buttons of " + ButtonsPanel.this.getClass().getSimpleName()
956                                        + " from index " + i + " when scrolling down");
957                        ButtonsPanel.this.putAll0(false, false, i, true);
958                        return true;
959                } else
960                        return false;
961        }

1039        /**
1040         * @param i
1041         *            The button index with {@link #buttonsTexts}.
1042         * @param y
1043         *            The y offset from the panel's top.
1044         * @param alignment
1045         *            The alignment, in the format of {@link #buttonsAlignment}.
1046         * @param displayVDots
1047         *            if {@link #scrollText} must be displayed instead of the
1048         *            button's text.
1049         * @param nbb
1050         *            The index with {@link #buttons}.
1051         * @param scroll
1052         *            Whether this is a scrolling request.
1053         */
1054        private void putButton(int i, int y, int alignment, boolean putMargins, boolean displayVDots, int nbb,
1055                        boolean scroll) {
1056                final IColoredString<T> text = displayVDots ? IColoredString.Impl.<T> create(scrollText, null)
1057                                : buttonsTexts.get(i);
1058                final int textLength = text.length();
1059
1060                final int insideWidth = textLength + (xpadding * 2);
1061                final int insideHeight = 1 + (ypadding * 2);
1062
1063                final int gridWidth = getGridWidth();
1064                final int center = gridWidth / 2;
1065                /* The leftmost cell of the button's inside */
1066                final int left;
1067                /* The left most cell of the button's text */
1068                final int textLeft;
1069                if (alignment == -1) {
1070                        /* Align to left */
1071                        left = 0;
1072                        textLeft = 0;
1073                } else if (alignment == 1) {
1074                        /* Align to right */
1075                        left = gridWidth - insideWidth;
1076                        textLeft = gridWidth - textLength;
1077                } else {
1078                        /* Align to center */
1079                        left = center - (insideWidth / 2);
1080                        textLeft = center - textLength / 2;
1081                }
1082
1083                /* Paint the button's inside's background */
1084                if (insideButtonBGColor != null) {
1085                        for (int x = 0; x < insideWidth; x++) {
1086                                for (int w = 0; w < insideHeight; w++)
1087                                        putBG(x + left, w + y, insideButtonBGColor);
1088                        }
1089                }
1090
1091                /* Now place text */
1092                putFG(textLeft, y + (insideHeight / 2), putShortcut(i, text));
1093
1094                final int botLeftY = y + insideHeight;
1095
1096                if (putMargins)
1097                        /* Finally, paint margins */
1098                        displayMarginsAround(left, botLeftY, insideWidth, insideHeight);
1099
1100                if (this.buttons == null)
1101                        throw new NullPointerException(
1102                                        getClass().getSimpleName() + ": this.buttons should not be null at this moment");
1103
1104                if (scroll)
1105                        this.buttons.get(nbb).setWidth(insideWidth);
1106                else
1107                        this.buttons.add(new Rectangle(left, botLeftY, insideWidth, insideHeight));
1108        }
